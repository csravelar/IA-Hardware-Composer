27 VKAPI_ATTR VkBool32 VKAPI_CALL VulkanDebugReportCallback(
28  VkDebugReportFlagsEXT flags, VkDebugReportObjectTypeEXT objectType,
29  uint64_t object, size_t location, int32_t messageCode,
30  const char *pLayerPrefix, const char *pMessage, void *pUserData) {
31  (void)flags;
32  (void)objectType;
33  (void)object;
34  (void)location;
35  (void)messageCode;
36  (void)pLayerPrefix;
37  (void)pUserData;
38  if (objectType == VK_DEBUG_REPORT_OBJECT_TYPE_SEMAPHORE_EXT)
39  return VK_FALSE;
40  if (objectType == VK_DEBUG_REPORT_OBJECT_TYPE_UNKNOWN_EXT &&
41  messageCode == 3 && strcmp(pLayerPrefix, "ParameterValidation") == 0 &&
42  strcmp(pMessage,
43  "vkCreateImage: value of pCreateInfo->pNext must be NULL") == 0)
44  return VK_FALSE;
45  ETRACE("vulkan(%d): %s\n", (int)objectType, pMessage);
46  return VK_FALSE;
47 }

49 uint32_t VKRenderer::GetMemoryTypeIndex(uint32_t mem_type_bits,
50  uint32_t required_props) {
51  for (uint32_t type_index = 0; type_index < 32;
52  type_index++, mem_type_bits >>= 1) {
53  if (mem_type_bits & 1) {
54  if ((device_mem_props_.memoryTypes[type_index].propertyFlags &
55  required_props) == required_props) {
56  return type_index;
57  }
58  }
59  }
60  return 32;
61 }

63 VkBuffer VKRenderer::UploadBuffer(size_t data_size, const uint8_t *data,
64  VkBufferUsageFlags usage) {
65  VkResult res;
66  VkBufferCreateInfo buffer_create = {};
67  buffer_create.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
68  buffer_create.size = data_size;
69  buffer_create.usage = VK_BUFFER_USAGE_TRANSFER_SRC_BIT;
70 
71  VkBuffer src_buffer;
72  res = vkCreateBuffer(dev_, &buffer_create, NULL, &src_buffer);
73  if (res != VK_SUCCESS) {
74  ETRACE("vkCreateBuffer failed (%d)\n", res);
75  return NULL;
76  }
77 
78  VkMemoryRequirements mem_requirements;
79  vkGetBufferMemoryRequirements(dev_, src_buffer, &mem_requirements);
80  VkMemoryAllocateInfo mem_allocate = {};
81  mem_allocate.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
82  mem_allocate.allocationSize = mem_requirements.size;
83  mem_allocate.memoryTypeIndex = GetMemoryTypeIndex(
84  mem_requirements.memoryTypeBits, VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT);
85  if (mem_allocate.memoryTypeIndex >= 32) {
86  ETRACE("Failed to find suitable staging device memory\n");
87  return NULL;
88  }
89 
90  VkDeviceMemory host_mem;
91  res = vkAllocateMemory(dev_, &mem_allocate, NULL, &host_mem);
92  if (res != VK_SUCCESS) {
93  ETRACE("vkAllocateMemory failed (%d)\n", res);
94  return NULL;
95  }
96 
97  res = vkBindBufferMemory(dev_, src_buffer, host_mem, 0);
98  if (res != VK_SUCCESS) {
99  ETRACE("vkBindBufferMemory failed (%d)\n", res);
100  return NULL;
101  }
102 
103  uint8_t *src_ptr;
104  res = vkMapMemory(dev_, host_mem, 0, mem_allocate.allocationSize, 0,
105  (void **)&src_ptr);
106  if (res != VK_SUCCESS) {
107  ETRACE("vkMapMemory failed (%d)\n", res);
108  return NULL;
109  }
110 
111  memcpy(src_ptr, data, data_size);
112 
113  vkUnmapMemory(dev_, host_mem);
114 
115  buffer_create.usage = usage | VK_BUFFER_USAGE_TRANSFER_DST_BIT;
116  VkBuffer dst_buffer;
117  res = vkCreateBuffer(dev_, &buffer_create, NULL, &dst_buffer);
118  if (res != VK_SUCCESS) {
119  ETRACE("vkCreateBuffer failed (%d)\n", res);
120  return NULL;
121  }
122  vkGetBufferMemoryRequirements(dev_, dst_buffer, &mem_requirements);
123  mem_allocate.allocationSize = mem_requirements.size;
124  mem_allocate.memoryTypeIndex = GetMemoryTypeIndex(
125  mem_requirements.memoryTypeBits, VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
126  if (mem_allocate.memoryTypeIndex >= 32) {
127  ETRACE("Failed to find suitable buffer device memory");
128  return NULL;
129  }
130 
131  VkDeviceMemory device_mem;
132  res = vkAllocateMemory(dev_, &mem_allocate, NULL, &device_mem);
133  if (res != VK_SUCCESS) {
134  ETRACE("vkAllocateMemory failed (%d)\n", res);
135  return NULL;
136  }
137  res = vkBindBufferMemory(dev_, dst_buffer, device_mem, 0);
138  if (res != VK_SUCCESS) {
139  ETRACE("vkBindBufferMemory failed (%d)\n", res);
140  return NULL;
141  }
142 
143  VkCommandBuffer cmd_buffer;
144  VkCommandBufferAllocateInfo cmd_buffer_allocate = {};
145  cmd_buffer_allocate.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
146  cmd_buffer_allocate.commandPool = cmd_pool_;
147  cmd_buffer_allocate.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
148  cmd_buffer_allocate.commandBufferCount = 1;
149 
150  res = vkAllocateCommandBuffers(dev_, &cmd_buffer_allocate, &cmd_buffer);
151  if (res != VK_SUCCESS) {
152  ETRACE("vkAllocateCommandBuffers failed (%d)\n", res);
153  return NULL;
154  }
155 
156  VkCommandBufferBeginInfo begin_info = {};
157  begin_info.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
158  begin_info.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
159 
160  res = vkBeginCommandBuffer(cmd_buffer, &begin_info);
161  if (res != VK_SUCCESS) {
162  ETRACE("vkBeginCommandBuffer failed (%d)\n", res);
163  return NULL;
164  }
165 
166  VkBufferCopy buffer_copy = {};
167  buffer_copy.size = data_size;
168 
169  vkCmdCopyBuffer(cmd_buffer, src_buffer, dst_buffer, 1, &buffer_copy);
170 
171  res = vkEndCommandBuffer(cmd_buffer);
172  if (res != VK_SUCCESS) {
173  ETRACE("vkEndCommandBuffer failed (%d)\n", res);
174  return NULL;
175  }
176 
177  VkSubmitInfo submit = {};
178  submit.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
179  submit.commandBufferCount = 1;
180  submit.pCommandBuffers = &cmd_buffer;
181 
182  res = vkQueueSubmit(queue_, 1, &submit, VK_NULL_HANDLE);
183  if (res != VK_SUCCESS) {
184  ETRACE("%d: vkQueueSubmit failed (%d)\n", __LINE__, res);
185  return NULL;
186  }
187 
188  res = vkQueueWaitIdle(queue_);
189  if (res != VK_SUCCESS) {
190  ETRACE("vkQueueWaitIdle failed (%d)\n", res);
191  return NULL;
192  }
193 
194  vkFreeCommandBuffers(dev_, cmd_pool_, 1, &cmd_buffer);
195  vkFreeMemory(dev_, host_mem, NULL);
196 
197  return dst_buffer;
198 }

468 bool VKRenderer::Draw(const std::vector<RenderState> &render_states,
469  NativeSurface *surface) {
470  VkResult res;
471  uint32_t frame_width = surface->GetWidth();
472  uint32_t frame_height = surface->GetHeight();
473  surface->MakeCurrent();
474 
475  src_image_infos_.clear();
476  ub_allocs_.clear();
477  std::vector<VkDescriptorSetLayout> desc_layouts;
478  std::vector<VkDescriptorSet> desc_sets(render_states.size());
479  std::vector<VkDescriptorBufferInfo> ub_infos;
480  for (const RenderState &state : render_states) {
481  unsigned size = state.layer_state_.size();
482  if (size == 0)
483  break;
484 
485  VKProgram *program = GetProgram(size);
486  if (!program)
487  continue;
488 
489  desc_layouts.emplace_back(program->getDescLayout());
490 
491  program->UseProgram(state, frame_width, frame_height);
492 
493  ub_infos.emplace_back(program->getVertUBInfo());
494  ub_infos.emplace_back(program->getFragUBInfo());
495  }
496 
497  VkDescriptorSetAllocateInfo alloc_desc_set = {};
498  alloc_desc_set.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
499  alloc_desc_set.descriptorPool = desc_pool_;
500  alloc_desc_set.descriptorSetCount = (uint32_t)desc_layouts.size();
501  alloc_desc_set.pSetLayouts = desc_layouts.data();
502 
503  res = vkAllocateDescriptorSets(dev_, &alloc_desc_set, desc_sets.data());
504  if (res != VK_SUCCESS) {
505  ETRACE("vkAllocateDescriptorSets failed (%d)\n", res);
506  return false;
507  }
508 
509  std::vector<VkWriteDescriptorSet> write_desc_sets;
510  size_t src_image_infos_offset = 0;
511  for (size_t cmd_index = 0; cmd_index < render_states.size(); cmd_index++) {
512  const RenderState &state = render_states[cmd_index];
513  size_t layer_count = state.layer_state_.size();
514  VkDescriptorSet desc_set = desc_sets[cmd_index];
515 
516  VkWriteDescriptorSet write_desc_set = {};
517  write_desc_set.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
518  write_desc_set.dstSet = desc_set;
519  write_desc_set.dstBinding = 0;
520  write_desc_set.descriptorCount = 1;
521  write_desc_set.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
522  write_desc_set.pBufferInfo = &ub_infos[cmd_index * 2 + 0];
523  write_desc_sets.emplace_back(write_desc_set);
524 
525  write_desc_set = {};
526  write_desc_set.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
527  write_desc_set.dstSet = desc_set;
528  write_desc_set.dstBinding = 1;
529  write_desc_set.descriptorCount = 1;
530  write_desc_set.descriptorType = VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER;
531  write_desc_set.pBufferInfo = &ub_infos[cmd_index * 2 + 1];
532  write_desc_sets.emplace_back(write_desc_set);
533 
534  write_desc_set = {};
535  write_desc_set.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
536  write_desc_set.dstSet = desc_set;
537  write_desc_set.dstBinding = 2;
538  write_desc_set.descriptorCount = (uint32_t)layer_count;
539  write_desc_set.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
540  write_desc_set.pImageInfo = &src_image_infos_[src_image_infos_offset];
541  write_desc_sets.emplace_back(write_desc_set);
542 
543  src_image_infos_offset += layer_count;
544  }
545 
546  vkUpdateDescriptorSets(dev_, write_desc_sets.size(), write_desc_sets.data(),
547  0, NULL);
548 
549  VkCommandBufferAllocateInfo cmd_buffer_alloc = {};
550  cmd_buffer_alloc.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
551  cmd_buffer_alloc.commandPool = cmd_pool_;
552  cmd_buffer_alloc.commandBufferCount = 1;
553 
554  VkCommandBuffer cmd_buffer;
555  res = vkAllocateCommandBuffers(dev_, &cmd_buffer_alloc, &cmd_buffer);
556  if (res != VK_SUCCESS) {
557  ETRACE("vkAllocateCommandBuffer failed (%d)\n", res);
558  return false;
559  }
