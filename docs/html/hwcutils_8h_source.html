100 inline HwcRect<int> Intersection(const hwcomposer::HwcRect<int>& rect1,
101  const hwcomposer::HwcRect<int>& rect2) {
102  HwcRect<int> rect = {0, 0, 0, 0};
103 
104  int lmax = std::max(rect1.left, rect2.left);
105  int rmax = std::max(rect1.right, rect2.right);
106  int tmax = std::max(rect1.top, rect2.top);
107  int bmax = std::max(rect1.bottom, rect2.bottom);
108 
109  int lmin = std::min(rect1.left, rect2.left);
110  int rmin = std::min(rect1.right, rect2.right);
111  int tmin = std::min(rect1.top, rect2.top);
112  int bmin = std::min(rect1.bottom, rect2.bottom);
113 
114  if (lmax == rmax || tmax == bmax)
115  return rect;
116 
117  if (lmin == rmin || tmin == bmin)
118  return rect;
119 
120  rect.left = lmax;
121  rect.right = rmin;
122  rect.top = tmax;
123  rect.bottom = bmin;
124 
125  return rect;
126 }
