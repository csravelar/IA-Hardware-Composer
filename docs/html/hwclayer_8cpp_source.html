96 void HwcLayer::SetSurfaceDamage(const HwcRegion& surface_damage) {
97  uint32_t rects = surface_damage.size();
98  state_ |= kLayerContentChanged;
99  HwcRect<int> rect;
100  ResetRectToRegion(surface_damage, rect);
101  if (rects == 1) {
102  if ((rect.top == 0) && (rect.bottom == 0) && (rect.left == 0) &&
103  (rect.right == 0)) {
104  state_ &= ~kLayerContentChanged;
105  state_ &= ~kSurfaceDamageChanged;
106  if (!surface_damage_.empty()) {
107  damage_dirty_ = true;
108  surface_damage_.reset();
109  }
110  return;
111  }
112  } else if (rects == 0) {
113  rect = display_frame_;
114  }
115 
116  if ((surface_damage_.left == rect.left) &&
117  (surface_damage_.top == rect.top) &&
118  (surface_damage_.right == rect.right) &&
119  (surface_damage_.bottom == rect.bottom)) {
120  return;
121  }
122 
123  state_ |= kSurfaceDamageChanged;
124  damage_dirty_ = true;
125 
126  surface_damage_ = rect;
127 }

129 void HwcLayer::SetVisibleRegion(const HwcRegion& visible_region) {
130  uint32_t rects = visible_region.size();
131  const HwcRect<int>& new_region = visible_region.at(0);
132  HwcRect<int> new_visible_rect = new_region;
133  state_ |= kVisibleRegionSet;
134  state_ &= ~kVisibleRegionChanged;
135 
136  for (uint32_t r = 1; r < rects; r++) {
137  const HwcRect<int>& rect = visible_region.at(r);
138  new_visible_rect.left = std::min(new_region.left, rect.left);
139  new_visible_rect.top = std::min(new_region.top, rect.top);
140  new_visible_rect.right = std::max(new_region.right, rect.right);
141  new_visible_rect.bottom = std::max(new_region.bottom, rect.bottom);
142  }
143 
144  if ((visible_rect_.left == new_visible_rect.left) &&
145  (visible_rect_.top == new_visible_rect.top) &&
146  (visible_rect_.right == new_visible_rect.right) &&
147  (visible_rect_.bottom == new_visible_rect.bottom)) {
148  return;
149  }
150 
151  state_ |= kVisibleRegionChanged;
152  damage_dirty_ = true;
153  visible_rect_ = new_visible_rect;
154 
155  if ((visible_rect_.top == 0) && (visible_rect_.bottom == 0) &&
156  (visible_rect_.left == 0) && (visible_rect_.right == 0)) {
157  state_ &= ~kVisible;
158  } else {
159  state_ |= kVisible;
160  }
161 }

328 const HwcRect<int>& HwcLayer::GetLayerDamage() {
329  if (!damage_dirty_) {
330  return current_rendering_damage_;
331  }
332 
333  current_rendering_damage_.reset();
334 
335  int ox = 0, oy = 0;
336  HwcRect<int> translated_damage =
337  TranslateRect(surface_damage_, -source_crop_.left, -source_crop_.top);
338 
339  if (transform_ == hwcomposer::HWCTransform::kTransform270) {
340  ox = display_frame_.left;
341  oy = display_frame_.bottom;
342  current_rendering_damage_.left = ox + translated_damage.top;
343  current_rendering_damage_.top = oy - translated_damage.right;
344  current_rendering_damage_.right = ox + translated_damage.bottom;
345  current_rendering_damage_.bottom = oy - translated_damage.left;
346  } else if (transform_ == hwcomposer::HWCTransform::kTransform180) {
347  ox = display_frame_.right;
348  oy = display_frame_.bottom;
349  current_rendering_damage_.left = ox - translated_damage.right;
350  current_rendering_damage_.top = oy - translated_damage.bottom;
351  current_rendering_damage_.right = ox - translated_damage.left;
352  current_rendering_damage_.bottom = oy - translated_damage.top;
353  } else if (transform_ & hwcomposer::HWCTransform::kTransform90) {
354  if (transform_ & hwcomposer::HWCTransform::kReflectX) {
355  ox = display_frame_.left;
356  oy = display_frame_.top;
357  current_rendering_damage_.left = ox + translated_damage.top;
358  current_rendering_damage_.top = oy + translated_damage.left;
359  current_rendering_damage_.right = ox + translated_damage.bottom;
360  current_rendering_damage_.bottom = oy + translated_damage.right;
361  } else if (transform_ & hwcomposer::HWCTransform::kReflectY) {
362  ox = display_frame_.right;
363  oy = display_frame_.bottom;
364  current_rendering_damage_.left = ox - translated_damage.bottom;
365  current_rendering_damage_.top = oy - translated_damage.right;
366  current_rendering_damage_.right = ox - translated_damage.top;
367  current_rendering_damage_.bottom = oy - translated_damage.left;
368  } else {
369  ox = display_frame_.right;
370  oy = display_frame_.top;
371  current_rendering_damage_.left = ox - translated_damage.bottom;
372  current_rendering_damage_.top = oy + translated_damage.left;
373  current_rendering_damage_.right = ox - translated_damage.top;
374  current_rendering_damage_.bottom = oy + translated_damage.right;
375  }
376  } else if (transform_ == 0) {
377  ox = display_frame_.left;
378  oy = display_frame_.top;
379  current_rendering_damage_.left = ox + translated_damage.left;
380  current_rendering_damage_.top = oy + translated_damage.top;
381  current_rendering_damage_.right = ox + translated_damage.right;
382  current_rendering_damage_.bottom = oy + translated_damage.bottom;
383  }
384 
385  if (state_ & kVisibleRegionSet) {
386  current_rendering_damage_ =
387  Intersection(current_rendering_damage_, visible_rect_);
388  }
389 
390  damage_dirty_ = false;
391 
392  return current_rendering_damage_;
393 }
