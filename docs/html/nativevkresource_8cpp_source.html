23 bool NativeVKResource::PrepareResources(
24  const std::vector<OverlayBuffer*>& buffers) {
25  VkResult res;
26 
27  Reset();
28 
29  VkImageSubresourceRange clear_range = {};
30  clear_range.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
31  clear_range.levelCount = 1;
32  clear_range.layerCount = 1;
33 
34  for (auto& buffer : buffers) {
35  const struct vk_import& import = buffer->GetGpuResource(dev_, true);
36  if (import.image_ == VK_NULL_HANDLE) {
37  ETRACE("Failed to make import image\n");
38  return false;
39  }
40  src_images_.emplace_back(import.image_);
41  src_image_memory_.emplace_back(import.memory_);
42 
43  VkImageMemoryBarrier barrier = {};
44  barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
45  barrier.srcAccessMask = 0;
46  barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
47  barrier.oldLayout = VK_IMAGE_LAYOUT_UNDEFINED;
48  barrier.newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
49  barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
50  barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
51  barrier.image = import.image_;
52  barrier.subresourceRange = clear_range;
53  src_barrier_before_clear_.emplace_back(barrier);
54 
55  VkImageViewCreateInfo view_create = {};
56  view_create.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
57  view_create.image = import.image_;
58  view_create.viewType = VK_IMAGE_VIEW_TYPE_2D;
59  view_create.format = NativeToVkFormat(buffer->GetFormat());
60  view_create.components = {};
61  view_create.components.r = VK_COMPONENT_SWIZZLE_R;
62  view_create.components.g = VK_COMPONENT_SWIZZLE_G;
63  view_create.components.b = VK_COMPONENT_SWIZZLE_B;
64  view_create.components.a = VK_COMPONENT_SWIZZLE_A;
65  view_create.subresourceRange = {};
66  view_create.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
67  view_create.subresourceRange.levelCount = 1;
68  view_create.subresourceRange.layerCount = 1;
69 
70  VkImageView image_view;
71  res = vkCreateImageView(dev_, &view_create, NULL, &image_view);
72  if (res != VK_SUCCESS) {
73  ETRACE("vkCreateImageView failed (%d)\n", res);
74  return false;
75  }
76  src_image_views_.emplace_back(image_view);
77 
78  struct vk_resource resource;
79  resource.image = import.image_;
80  resource.image_view = image_view;
81  layer_textures_.emplace_back(resource);
82  }
83  return true;
84 }
