25 bool NativeGLResource::PrepareResources(
26  const std::vector<OverlayBuffer*>& buffers) {
27  std::vector<GLuint>().swap(layer_textures_);
28  layer_textures_.reserve(buffers.size());
29  EGLDisplay egl_display = eglGetCurrentDisplay();
30  for (auto& buffer : buffers) {
31  // Create EGLImage.
32  const ResourceHandle& import_image =
33  buffer->GetGpuResource(egl_display, true);
34 
35  if (import_image.image_ == EGL_NO_IMAGE_KHR) {
36  ETRACE("Failed to make import image.");
37  return false;
38  }
39 
40  layer_textures_.emplace_back(import_image.texture_);
41  }
42 
43  return true;
44 }

49 void NativeGLResource::ReleaseGPUResources(
50  const std::vector<ResourceHandle>& handles) {
51  size_t purged_size = handles.size();
52  EGLDisplay egl_display = eglGetCurrentDisplay();
53  glBindFramebuffer(GL_FRAMEBUFFER, 0);
54  glBindTexture(GL_TEXTURE_EXTERNAL_OES, 0);
55  std::vector<GLuint> textures;
56  std::vector<GLuint> fbs;
57 
58  for (size_t i = 0; i < purged_size; i++) {
59  const ResourceHandle& handle = handles.at(i);
60  if (handle.image_) {
61  eglDestroyImageKHR(egl_display, handle.image_);
62  }
63 
64  if (handle.texture_) {
65  textures.emplace_back(handle.texture_);
66  }
67 
68  if (handle.fb_) {
69  fbs.emplace_back(handle.fb_);
70  }
71  }
72 
73  uint32_t textures_size = textures.size();
74  if (textures_size > 0) {
75  glDeleteTextures(textures_size, textures.data());
76  }
77 
78  uint32_t fb_size = fbs.size();
79 
80  if (fb_size > 0) {
81  glDeleteFramebuffers(fb_size, fbs.data());
82  }
83 }
