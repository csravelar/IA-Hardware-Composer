32 int HwfLayer::InitFromHwcLayer(hwf_layer_t *sf_layer) {
33  if (!hwc_layer_) {
34  hwc_layer_ = new hwcomposer::HwcLayer();
35  }
36 
37  bool surface_damage = true;
38 
39  if (hwc_layer_->GetNativeHandle() &&
40  (hwc_layer_->GetNativeHandle()->target_ == sf_layer->target))
41  surface_damage = false;
42 
43  native_handle_.target_ = sf_layer->target;
44  hwc_layer_->SetNativeHandle(&native_handle_);
45  hwc_layer_->SetAlpha(sf_layer->globalAlpha);
46  hwc_layer_->SetSourceCrop(hwcomposer::HwcRect<float>(
47  sf_layer->srcRect.left, sf_layer->srcRect.top, sf_layer->srcRect.right,
48  sf_layer->srcRect.bottom));
49 
50  hwc_layer_->SetDisplayFrame(
51  hwcomposer::HwcRect<int>(sf_layer->destRect.left, sf_layer->destRect.top,
52  sf_layer->destRect.right,
53  sf_layer->destRect.bottom),
54  0, 0);
55 
56  uint32_t transform = 0;
57  if (sf_layer->transform == HWF_TRANSFORM_ROT_270) {
58  transform = hwcomposer::HWCTransform::kTransform270;
59  } else if (sf_layer->transform == HWF_TRANSFORM_ROT_180) {
60  transform = hwcomposer::HWCTransform::kTransform180;
61  } else {
62  if (sf_layer->transform & HWF_TRANSFORM_FLIP_H)
63  transform |= hwcomposer::HWCTransform::kReflectX;
64  if (sf_layer->transform & HWF_TRANSFORM_FLIP_V)
65  transform |= hwcomposer::HWCTransform::kReflectY;
66  if (sf_layer->transform & HWF_TRANSFORM_ROT_90)
67  transform |= hwcomposer::HWCTransform::kTransform90;
68  }
69 
70  hwc_layer_->SetTransform(transform);
71  hwc_layer_->SetAcquireFence(dup(sf_layer->acquireSyncFd));
72 
73  switch (sf_layer->blendMode) {
74  case HWF_BLENDING_NONE:
75  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingNone);
76  break;
77 
78  case HWF_BLENDING_PREMULT:
79  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingPremult);
80  break;
81 
82  case HWF_BLENDING_COVERAGE:
83  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingCoverage);
84  break;
85 
86  default:
87  LOG_E("Invalid blendMode in hwc_layer_1_t %d", sf_layer->blendMode);
88  return -EINVAL;
89  }
90 
91  // TODO: Add the damage detection.
92  if (surface_damage) {
93  hwcomposer::HwcRegion hwc_region;
94 #if 0
95  uint32_t num_rects = sf_layer->surfaceDamage.numRects;
96  for (size_t rect = 0; rect < num_rects; ++rect) {
97  hwc_region.emplace_back(sf_layer->surfaceDamage.rects[rect].left,
98  sf_layer->surfaceDamage.rects[rect].top,
99  sf_layer->surfaceDamage.rects[rect].right,
100  sf_layer->surfaceDamage.rects[rect].bottom);
101  }
102 #endif
103  hwc_layer_->SetSurfaceDamage(hwc_region);
104  } else {
105  hwcomposer::HwcRegion hwc_region;
106  hwc_region.emplace_back(0, 0, 0, 0);
107  hwc_layer_->SetSurfaceDamage(hwc_region);
108  }
109 
110  uint32_t num_rects = sf_layer->visibleRegion.num;
111  hwcomposer::HwcRegion visible_region;
112 
113  for (size_t rect = 0; rect < num_rects; ++rect) {
114  visible_region.emplace_back(sf_layer->visibleRegion.rects[rect].left,
115  sf_layer->visibleRegion.rects[rect].top,
116  sf_layer->visibleRegion.rects[rect].right,
117  sf_layer->visibleRegion.rects[rect].bottom);
118  }
119 
120  hwc_layer_->SetVisibleRegion(visible_region);
121 
122  return 0;
123 }

236 int HwfDevice::flip(struct hwf_device_t *device, int dispCount,
237  hwf_display_t **displays) {
238  CTRACE();
239  LOG_I("HwfDevice::flip --> enter.\n");
240  LOG_I("HwfDevice::flip --> dispCount: %d\n", dispCount);
241 
242  HwfDevice *hwf_device = (HwfDevice *)device;
243 
244  for (int i = 0; i < dispCount; ++i) {
245  LOG_I("\tflip --> display[%d] -- begin.\n", i);
246  hwf_display_t *dc = displays[i];
247  if (!dc || i == HWF_DISPLAY_VIRTUAL)
248  continue;
249 
250  size_t num_dc_layers = dc->numLayers;
251  HwfDisplay *native_display = hwf_device->GetDisplay(i);
252  dc->retireSyncFd = native_display->timeline_.IncrementTimeLine();
253  hwcomposer::NativeDisplay *display = native_display->display_;
254  std::vector<HwfLayer *> &old_layers = native_display->layers_;
255  std::vector<HwfLayer *> new_layers;
256  size_t size = old_layers.size();
257  std::vector<hwcomposer::HwcLayer *> source_layers;
258  for (size_t j = 0; j < num_dc_layers; ++j) {
259  hwf_layer_t *sf_layer = &dc->hwfLayers[j];
260  if (!sf_layer || !sf_layer->target ||
261  (sf_layer->flags & HWF_LAYER_IGNORED))
262  continue;
263 
264  if (!native_display->gl_composition_ &&
265  (sf_layer->composeMode == HWF_FB_TARGET)) {
266  continue;
267  }
268 
269  HwfLayer *new_layer = new HwfLayer();
270  if (size > j) {
271  HwfLayer *old_layer = old_layers.at(j);
272  new_layer->hwc_layer_ = old_layer->hwc_layer_;
273  old_layer->hwc_layer_ = NULL;
274  }
275 
276  new_layer->InitFromHwcLayer(sf_layer);
277  source_layers.emplace_back(new_layer->hwc_layer_);
278  new_layer->index_ = j;
279  new_layers.emplace_back(new_layer);
280  sf_layer->acquireSyncFd = -1;
281  sf_layer->releaseSyncFd = -1;
282  }
283 
284  if (source_layers.empty()) {
285  return 0;
286  }
287 
288  int32_t retire_fence = -1;
289  old_layers.swap(new_layers);
290  size = new_layers.size();
291  for (size_t i = 0; i < size; i++) {
292  HwfLayer *layer = new_layers.at(i);
293  delete layer;
294  }
295 
296  std::vector<HwfLayer *>().swap(new_layers);
297 
298  LOG_I("\tWill to present.\n");
299  bool success = display->Present(source_layers, &retire_fence);
300  if (!success) {
301  LOG_E("Failed to set layers in the composition");
302  return -1;
303  }
304 
305  if (retire_fence > 0)
306  close(retire_fence);
307 
308  size = old_layers.size();
309  for (size_t i = 0; i < size; i++) {
310  hwcomposer::HwcLayer *layer = old_layers.at(i)->hwc_layer_;
311  int32_t release_fence = layer->GetReleaseFence();
312 
313  if (release_fence <= 0)
314  continue;
315 
316  hwf_layer_t *sf_layer = &dc->hwfLayers[old_layers.at(i)->index_];
317  sf_layer->releaseSyncFd = release_fence;
318  }
319 
320  std::vector<hwcomposer::HwcLayer *>().swap(source_layers);
321 
322  LOG_I("\tflip --> display[%d] -- end.\n", i);
323  }
324 
325  LOG_I("HwfDevice::flip --> exit.\n");
326 
327  return 0;
328 }

567  hwf_device->primary_display_.display_id_ = 0;

569  hwf_device->primary_display_.display_->SetExplicitSyncSupport(
570  hwf_device->disable_explicit_sync_);

572
