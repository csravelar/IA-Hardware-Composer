64 inline uint32_t GetNativeBuffer(uint32_t gpu_fd, HWCNativeHandle handle) {
65  uint32_t id = 0;
66  uint32_t prime_fd = -1;
67  if (!handle->meta_data_.fb_modifiers_[0]) {
68  prime_fd = handle->import_data.fd_data.fd;
69  } else {
70  prime_fd = handle->import_data.fd_modifier_data.fds[0];
71  }
72  if (drmPrimeFDToHandle(gpu_fd, prime_fd, &id)) {
73  ETRACE("Error generate handle from prime fd %d", prime_fd);
74  }
