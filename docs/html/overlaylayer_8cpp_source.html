127 void OverlayLayer::ValidateTransform(uint32_t transform,
128  uint32_t display_transform) {
129  std::map<int, int> tmap = {{kIdentity, 0},
130  {kTransform90, 1},
131  {kTransform180, 2},
132  {kTransform270, 3}};
133  std::vector<int> inv_tmap = {kIdentity, kTransform90, kTransform180,
134  kTransform270};
135 
136  int mtransform =
137  transform & (kIdentity | kTransform90 | kTransform180 | kTransform270);
138  if (tmap.find(mtransform) != tmap.end()) {
139  mtransform = tmap[mtransform];
140  } else {
141  // reaching here indicates that transform is
142  // is an OR of multiple values
143  // Assign Identity in this case
144  mtransform = kIdentity;
145  }
146 
147  // The elements {0, 1, 2, 3} form a circulant matrix under mod 4 arithmetic
148  mtransform = (mtransform + display_transform) % 4;
149  mtransform = inv_tmap[mtransform];
150  plane_transform_ = mtransform;
151 
152  if (plane_transform_ & kTransform90) {
153  if (transform & kReflectX)
154  plane_transform_ |= kReflectX;
155 
156  if (transform & kReflectY)
157  plane_transform_ |= kReflectY;
158  }
159 }

161 void OverlayLayer::InitializeState(HwcLayer* layer,
162  ResourceManager* resource_manager,
163  OverlayLayer* previous_layer,
164  uint32_t z_order, uint32_t layer_index,
165  uint32_t max_height, uint32_t rotation,
166  bool handle_constraints,
167  FrameBufferManager* frame_buffer_manager) {
168  transform_ = layer->GetTransform();
169  if (rotation != kRotateNone) {
170  ValidateTransform(layer->GetTransform(), rotation);
171  } else {
172  plane_transform_ = transform_;
173  }
174 
175  alpha_ = layer->GetAlpha();
176  layer_index_ = layer_index;
177  z_order_ = z_order;
178  source_crop_width_ = layer->GetSourceCropWidth();
179  source_crop_height_ = layer->GetSourceCropHeight();
180  source_crop_ = layer->GetSourceCrop();
181  blending_ = layer->GetBlending();
182  surface_damage_ = layer->GetLayerDamage();
183  if (previous_layer && layer->HasZorderChanged()) {
184  if (previous_layer->actual_composition_ == kGpu) {
185  CalculateRect(previous_layer->display_frame_, surface_damage_);
186  bool force_partial_clear = true;
187  // We can skip Clear in case display frame, transforms are same.
188  if (previous_layer->display_frame_ == display_frame_ &&
189  transform_ == previous_layer->transform_ &&
190  plane_transform_ == previous_layer->plane_transform_) {
191  force_partial_clear = false;
192  }
193 
194  if (force_partial_clear) {
195  state_ |= kForcePartialClear;
196  }
197  } else if (!layer->IsCursorLayer()) {
198  state_ |= kNeedsReValidation;
199  }
200  }
201 
202  SetBuffer(layer->GetNativeHandle(), layer->GetAcquireFence(),
203  resource_manager, true, frame_buffer_manager);
204 
205  if (!surface_damage_.empty()) {
206  if (type_ == kLayerCursor) {
207  const std::shared_ptr<OverlayBuffer>& buffer = imported_buffer_->buffer_;
208  surface_damage_.right = surface_damage_.left + buffer->GetWidth();
209  surface_damage_.bottom = surface_damage_.top + buffer->GetHeight();
210  }
211  }
212 
213  if (!handle_constraints) {
214  if (previous_layer) {
215  ValidatePreviousFrameState(previous_layer, layer);
216  }
217  return;
218  }
219 
220  int32_t left_constraint = layer->GetLeftConstraint();
221  int32_t right_constraint = layer->GetRightConstraint();
222  int32_t left_source_constraint = layer->GetLeftSourceConstraint();
223  int32_t right_source_constraint = layer->GetRightSourceConstraint();
224  int32_t display_frame_left = display_frame_.left;
225  uint32_t frame_width = display_frame_.right - display_frame_.left;
226  uint32_t source_width =
227  static_cast<int>(source_crop_.right - source_crop_.left);
228  uint32_t frame_offset_left = 0;
229  uint32_t frame_offset_right = frame_width;
230  if (left_constraint >= 0 && right_constraint >= 0) {
231  if (display_frame_.left > right_source_constraint) {
232  state_ |= kInvisible;
233  return;
234  }
235 
236  if (display_frame_.right < left_source_constraint) {
237  state_ |= kInvisible;
238  return;
239  }
240 
241  if (display_frame_.left < left_source_constraint) {
242  frame_offset_left = left_source_constraint - display_frame_left;
243  display_frame_.left = left_source_constraint;
244  }
245 
246  if (display_frame_.right > right_source_constraint) {
247  frame_offset_right = right_source_constraint - display_frame_left;
248  display_frame_.right = right_source_constraint;
249  }
250 
251  // Handle case where we might be using logical and Mosaic together.
252  display_frame_.left =
253  (display_frame_.left - left_source_constraint) + left_constraint;
254  display_frame_.right =
255  (display_frame_.right - left_source_constraint) + left_constraint;
256  IMOSAICDISPLAYTRACE(
257  "display_frame_ %d %d %d %d left_source_constraint: %d "
258  "left_constraint: %d \n",
259  display_frame_.left, display_frame_.right, display_frame_.top,
260  display_frame_.bottom, left_source_constraint, left_constraint);
261 
262  display_frame_.bottom =
263  std::min(max_height, static_cast<uint32_t>(display_frame_.bottom));
264  display_frame_width_ = display_frame_.right - display_frame_.left;
265  display_frame_height_ = display_frame_.bottom - display_frame_.top;
266 
267  if ((surface_damage_.left < display_frame_.left) &&
268  (surface_damage_.right > display_frame_.left)) {
269  surface_damage_.left = display_frame_.left;
270  }
271 
272  if (surface_damage_.right > display_frame_.right) {
273  surface_damage_.right = display_frame_.right;
274  }
275 
276  if (AnalyseOverlap(surface_damage_, display_frame_) != kOutside) {
277  surface_damage_.bottom =
278  std::min(surface_damage_.bottom, display_frame_.bottom);
279  surface_damage_.right =
280  std::min(surface_damage_.right, display_frame_.right);
281  surface_damage_.left =
282  std::max(surface_damage_.left, display_frame_.left);
283  } else {
284  surface_damage_.reset();
285  }
286  IMOSAICDISPLAYTRACE(
287  "surface_damage_ %d %d %d %d left_source_constraint: %d "
288  "left_constraint: %d \n",
289  surface_damage_.left, surface_damage_.right, surface_damage_.top,
290  surface_damage_.bottom, left_source_constraint, left_constraint);
291 
292  // split the source in proportion of frame rect offset for sub displays as:
293  // 1. the original source size may be different with the original frame
294  // rect,
295  // we need get proportional content of source.
296  // 2. the UI content may cross the sub displays of Mosaic or Logical mode
297 
298  source_crop_.left = static_cast<float>(source_width) *
299  (static_cast<float>(frame_offset_left) /
300  static_cast<float>(frame_width));
301  source_crop_.right = static_cast<float>(source_width) *
302  (static_cast<float>(frame_offset_right) /
303  static_cast<float>(frame_width));
304  source_crop_width_ = static_cast<int>(ceilf(source_crop_.right) -
305  static_cast<int>(source_crop_.left));
306  source_crop_height_ = static_cast<int>(ceilf(source_crop_.bottom) -
307  static_cast<int>(source_crop_.top));
308  }
309 
310  if (previous_layer) {
311  ValidatePreviousFrameState(previous_layer, layer);
312  }
313 }

339 void OverlayLayer::ValidatePreviousFrameState(OverlayLayer* rhs,
340  HwcLayer* layer) {
341  OverlayBuffer* buffer = imported_buffer_->buffer_.get();
342  supported_composition_ = rhs->supported_composition_;
343  actual_composition_ = rhs->actual_composition_;
344 
345  bool content_changed = false;
346  bool rect_changed = layer->HasDisplayRectChanged();
347  bool source_rect_changed = layer->HasSourceRectChanged();
348  if (source_rect_changed)
349  state_ |= kSourceRectChanged;
350 
351  // We expect cursor plane to support alpha always.
352  if ((actual_composition_ & kGpu) || (type_ == kLayerCursor)) {
353  if (actual_composition_ & kGpu) {
354  content_changed = rect_changed || source_rect_changed;
355  // This layer has replaced an existing layer, let's make sure
356  // we re-draw this and previous layer regions.
357  if (!layer->IsValidated()) {
358  content_changed = true;
359  CalculateRect(rhs->display_frame_, surface_damage_);
360  } else if (!content_changed) {
361  if ((buffer->GetFormat() !=
362  rhs->imported_buffer_->buffer_->GetFormat()) ||
363  (alpha_ != rhs->alpha_) || (blending_ != rhs->blending_) ||
364  (transform_ != rhs->transform_)) {
365  content_changed = true;
366  }
367  }
368  } else if (type_ == kLayerCursor) {
369  if (layer->HasLayerAttributesChanged()) {
370  state_ |= kNeedsReValidation;
371  }
372  }
373  } else {
374  // Ensure the buffer can be supported by display for direct
375  // scanout.
376  if (buffer->GetFormat() != rhs->imported_buffer_->buffer_->GetFormat()) {
377  state_ |= kNeedsReValidation;
378  return;
379  }
380 
381  // If previous layer was opaque and we have alpha now,
382  // let's mark this layer for re-validation. Plane
383  // supporting XRGB format might not necessarily support
384  // transparent planes. We assume plane supporting
385  // ARGB will support XRGB.
386  if ((rhs->alpha_ == 0xff) && (alpha_ != rhs->alpha_)) {
387  state_ |= kNeedsReValidation;
388  return;
389  }
390 
391  if (blending_ != rhs->blending_) {
392  state_ |= kNeedsReValidation;
393  return;
394  }
395 
396  if (rect_changed || layer->HasLayerAttributesChanged()) {
397  state_ |= kNeedsReValidation;
398  return;
399  }
400 
401  if (source_rect_changed) {
402  // If the overall width and height hasn't changed, it
403  // shouldn't impact the plane composition results.
404  if ((source_crop_width_ != rhs->source_crop_width_) ||
405  (source_crop_height_ != rhs->source_crop_height_)) {
406  state_ |= kNeedsReValidation;
407  return;
408  }
409  }
410  }
411 
412  if (!rect_changed) {
413  state_ &= ~kDimensionsChanged;
414  }
415 
416  if (!layer->HasVisibleRegionChanged() && !content_changed &&
417  surface_damage_.empty() && !layer->HasLayerContentChanged() &&
418  !(state_ & kNeedsReValidation)) {
419  state_ &= ~kLayerContentChanged;
420  }
421 }

428 void OverlayLayer::CloneLayer(const OverlayLayer* layer,
429  const HwcRect<int>& display_frame,
430  ResourceManager* resource_manager,
431  uint32_t z_order,
432  FrameBufferManager* frame_buffer_manager) {

443  surface_damage_ = layer->GetSurfaceDamage();
444  transform_ = layer->transform_;
445  plane_transform_ = layer->plane_transform_;
446  alpha_ = layer->alpha_;
447  layer_index_ = z_order;
448  z_order_ = z_order;
449  blending_ = layer->blending_;
