141 bool DisplayQueue::ForcePlaneValidation(int add_index, int remove_index,
142  int total_layers_size,
143  size_t total_planes) {
144  // This is the case where cursor has changed z_order.
145  if (remove_index == add_index)
146  return false;
147 
148  int free_planes = display_plane_manager_->GetTotalOverlays() - total_planes;
149  if (free_planes <= 0) {
150  return false;
151  }
152 
153  // Bail out in case we have new layers to fill the remaining planes.

155  return false;
156  }

161 void DisplayQueue::GetCachedLayers(const std::vector<OverlayLayer>& layers,
162  int remove_index,
163  DisplayPlaneStateList* composition,
164  bool* render_layers, bool* can_ignore_commit,
165  bool* needs_plane_validation,
166  bool* force_full_validation,
167  int* add_index) {
168  CTRACE();
169  bool needs_gpu_composition = false;
170  bool ignore_commit = true;
171  bool check_to_squash = false;
172  bool plane_validation = false;
173  // If Scanout layers DisplayFrame rect has changed, we need
174  // to re-calculate our Composition regions for planes using
175  // GPU Composition.
176  bool reset_composition_regions = false;
177  bool reset_plane = remove_index != -1 ? true : false;
178  bool removed_layers = false;
179  bool pop_plane = false;
180  size_t previous_size = 0;
181 
182  for (DisplayPlaneState& previous_plane : previous_plane_state_) {
183  previous_size++;
184  // Mark surfaces of all planes to be released once they are
185  // offline.
186  if (removed_layers) {
187  if (previous_plane.NeedsOffScreenComposition()) {
188  display_plane_manager_->MarkSurfacesForRecycling(
189  &previous_plane, surfaces_not_inuse_, true);
190  }
191 
192  previous_plane.GetDisplayPlane()->SetInUse(false);
193  continue;
194  }
195 
196  composition->emplace_back();
197  DisplayPlaneState& last_plane = composition->back();
198  last_plane.CopyState(previous_plane);
199  if (reset_plane) {
200  const std::vector<size_t>& source_layers = last_plane.GetSourceLayers();
201  size_t source_layers_size = source_layers.size();
202  size_t index = source_layers.at(source_layers_size - 1);
203  size_t threshold = static_cast<size_t>(remove_index);
204  bool needs_plane_validation = false;
205 
206  if (index >= threshold) {
207  removed_layers = true;
208 #ifdef SURFACE_TRACING
209  size_t original_size = source_layers_size;
210 #endif
211  bool has_one_layer = source_layers_size == 1 ? true : false;
212  if (!has_one_layer) {
213  last_plane.ResetLayers(layers, threshold, &needs_plane_validation);
214  }
215 
216  source_layers_size = source_layers.size();
217 #ifdef SURFACE_TRACING
218  ISURFACETRACE(
219  "Layers removed. Total old Layers: %d Total new Layers: %d "
220  "Threshold: "
221  "%d Plane Layer Index: %d Total Planes: %d previous_plane_state_ "
222  "%d \n",
223  original_size, source_layers_size, threshold, index,
224  composition->size(), previous_plane_state_.size());
225 #endif
226  // We need to force re-validation of commit to ensure we update any
227  // Scalar usage with the new combination of layers.
228  ignore_commit = false;
229 
230  if (last_plane.GetSourceLayers().empty() || has_one_layer) {
231  display_plane_manager_->MarkSurfacesForRecycling(
232  &last_plane, surfaces_not_inuse_, true);
233 
234  // On some platforms disabling primary disables
235  // the whole pipe. Let's revalidate the new layers
236  // and ensure primary has a buffer.
237  if (last_plane.GetDisplayPlane() ==
238  previous_plane_state_.begin()->GetDisplayPlane()) {
239 #ifdef SURFACE_TRACING
240  ISURFACETRACE("Primary plane is empty forcing full validation. \n");
241 #endif
242  *force_full_validation = true;
243  *can_ignore_commit = false;
244  return;
245  }
246 
247  last_plane.GetDisplayPlane()->SetInUse(false);
248  composition->pop_back();
249  pop_plane = ForcePlaneValidation(*add_index, remove_index,
250  layers.size(), composition->size());
251 #ifdef SURFACE_TRACING
252  ISURFACETRACE(
253  "Plane removed. Total old Layers: %d Total new Layers: %d "
254  "Threshold: "
255  "%d Plane Layer Index: %d Total Planes: %d previous_plane_state_ "
256  "%d \n",
257  original_size, source_layers_size, threshold, index,
258  composition->size(), previous_plane_state_.size());
259 #endif
260  continue;
261  }
262 
263  std::vector<OverlayPlane> temp;
264  while (display_plane_manager_->SquashPlanesAsNeeded(
265  layers, *composition, temp, surfaces_not_inuse_,
266  &plane_validation)) {
267  continue;
268  }
269  DisplayPlaneState& squashed_plane = composition->back();
270 
271  squashed_plane.ValidateReValidation();
272 
273  if (squashed_plane.RevalidationType() &
274  DisplayPlaneState::ReValidationType::kScanout) {
275  const std::vector<size_t>& source_layers =
276  squashed_plane.GetSourceLayers();
277  const OverlayLayer* layer = &(layers.at(source_layers.at(0)));
278  // Check if Actual & Supported Composition differ for this
279  // layer. If so than let' mark it for validation.
280  if (source_layers_size == 1) {
281  if (layer->CanScanOut() &&
282  squashed_plane.NeedsOffScreenComposition()) {
283  plane_validation = true;
284  } else {
285  check_to_squash = true;
286  squashed_plane.RevalidationDone(
287  DisplayPlaneState::ReValidationType::kScanout);
288  }
289  }
290  }
291  }
292 
293  if (needs_plane_validation)
294  plane_validation = true;
295 
296  if (removed_layers) {
297  pop_plane = ForcePlaneValidation(*add_index, remove_index,
298  layers.size(), composition->size());
299  if (pop_plane)
300  continue;
301  }
302  }
303 
304  DisplayPlaneState& target_plane = composition->back();
305  if (target_plane.NeedsOffScreenComposition()) {
306  HwcRect<int> surface_damage = HwcRect<int>(0, 0, 0, 0);
307  bool update_rect = reset_plane;
308  bool refresh_surfaces = reset_composition_regions;
309  bool force_partial_clear = false;
310 
311  const std::vector<size_t>& source_layers = target_plane.GetSourceLayers();
312  size_t layers_size = source_layers.size();
313  if (!removed_layers && !update_rect) {
314  for (size_t i = 0; i < layers_size; i++) {
315  const size_t& source_index = source_layers.at(i);
316  const OverlayLayer& layer = layers.at(source_index);
317  if (layer.HasDimensionsChanged()) {
318  update_rect = true;
319  break;
320  }
321 
322  if (refresh_surfaces) {
323  continue;
324  }
325 
326  if (layer.NeedsPartialClear()) {
327  force_partial_clear = true;
328  }
329 
330  if (layer.HasLayerContentChanged()) {
331  CalculateRect(layer.GetSurfaceDamage(), surface_damage);
332  }
333  }
334  }
335 
336  if (layers_size == 1) {
337  // If this is a video plane, ensure that buffer format
338  // is still related to Video/Camera. Handle buffer format
339  // changes in this case.
340  if (target_plane.IsVideoPlane()) {
341  const OverlayLayer& layer = layers.at(source_layers.at(0));
342  if (!layer.IsVideoLayer()) {
343  target_plane.SetVideoPlane(false);
344  display_plane_manager_->MarkSurfacesForRecycling(
345  &target_plane, surfaces_not_inuse_, true);
346  refresh_surfaces = true;
347  }
348  } else if (target_plane.CanSupportVideo() &&
349  !target_plane.IsVideoPlane()) {
350  const OverlayLayer& layer = layers.at(source_layers.at(0));
351  if (layer.IsVideoLayer()) {
352  target_plane.SetVideoPlane(true);
353  display_plane_manager_->MarkSurfacesForRecycling(
354  &target_plane, surfaces_not_inuse_, true);
355  refresh_surfaces = true;
356  }
357  }
358  }
359 
360  if (!removed_layers && update_rect) {
361  target_plane.RefreshLayerRects(layers);
362  surface_damage.reset();
363  }
364 
365  // Let's check if we need to check this plane-layer combination.
366  if (update_rect) {
367  target_plane.ValidateReValidation();
368  if (target_plane.RevalidationType() !=
369  DisplayPlaneState::ReValidationType::kNone) {
370  plane_validation = true;
371  }
372  }
373 
374  if (update_rect || refresh_surfaces || !surface_damage.empty() ||
375  force_partial_clear) {
376  needs_gpu_composition = true;
377  if (target_plane.NeedsSurfaceAllocation()) {
378  display_plane_manager_->SetOffScreenPlaneTarget(target_plane);
379  } else if (refresh_surfaces || reset_plane) {
380  target_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
381  } else if (!update_rect && !surface_damage.empty()) {
382  if (force_partial_clear) {
383  target_plane.RefreshSurfaces(NativeSurface::kPartialClear, true);
384  }
385 
386  target_plane.UpdateDamage(surface_damage);
387  }
388 
389  if (refresh_surfaces || reset_plane || update_rect) {
390  std::vector<OverlayPlane> temp;
391  bool squashed = display_plane_manager_->SquashPlanesAsNeeded(
392  layers, *composition, temp, surfaces_not_inuse_,
393  &plane_validation);
394  if (squashed) {
395  // We squashed planes and it's not the last one.
396  // We might have messed up with plane order, let's
397  // force full validation.
398  if (previous_size != previous_plane_state_.size()) {
399  *force_full_validation = true;
400  *can_ignore_commit = false;
401  return;
402  } else {
403  while (display_plane_manager_->SquashPlanesAsNeeded(
404  layers, *composition, temp, surfaces_not_inuse_,
405  &plane_validation)) {
406  continue;
407  }
408  }
409  }
410  }
411  } else {
412  target_plane.UpdateDamage(surface_damage);
413  }
414 
415  DisplayPlaneState& squashed_plane = composition->back();
416  squashed_plane.RefreshSurfaces(NativeSurface::kFullClear);
417 
418  if (!needs_gpu_composition)
419  needs_gpu_composition = !(squashed_plane.IsSurfaceRecycled());
420 
421  reset_composition_regions = false;
422  } else {
423  reset_composition_regions = false;
424  const OverlayLayer* layer =
425  &(layers.at(last_plane.GetSourceLayers().front()));
426 
427  OverlayBuffer* buffer = layer->GetBuffer();
428  if (buffer->GetFb() == 0) {
429  buffer->CreateFrameBuffer();
430 
431  // FB creation failed, we need to re-validate the
432  // whole commit.
433  if (buffer->GetFb() == 0) {
434  *force_full_validation = true;
435  *can_ignore_commit = false;
436  return;
437  }
438 
439  reset_composition_regions = true;
440  }
441 
442  last_plane.SetOverlayLayer(layer);
443  if (layer->HasLayerContentChanged()) {
444  ignore_commit = false;
445  }
446 
447  bool needs_revalidation = layer->NeedsRevalidation();
448  if (layer->HasDimensionsChanged() || needs_revalidation) {
449  ignore_commit = false;
450  reset_composition_regions = true;
451  }
452 
453  if (needs_revalidation) {
454  plane_validation = true;
455  }
456  }
457  }
458 
459  *render_layers = needs_gpu_composition;
460  if (needs_gpu_composition)
461  ignore_commit = false;
462 
463  *can_ignore_commit = ignore_commit;
464  *needs_plane_validation = plane_validation;
465  size_t total_planes = composition->size();
466  // Check if we can squash the last overlay (Before Cursor Plane).
467  // If we have freed planes, let's see if we can actually map
468  // layers to planes for direct scanout.
469  if (check_to_squash && !pop_plane) {
470  if (composition->back().IsCursorPlane()) {
471  // We cannot squash Cursor plane.
472  total_planes -= 1;
473  }
474 
475  if (total_planes > 2) {
476  DisplayPlaneState& old_plane = composition->at(total_planes - 2);
477  DisplayPlaneState& last_overlay = composition->at(total_planes - 1);
478  const std::vector<size_t>& source_layers = last_overlay.GetSourceLayers();
479 
480  if (old_plane.CanSquash() && last_overlay.CanSquash() &&
481  source_layers.size() == 1) {
482 #ifdef SURFACE_TRACING
483  ISURFACETRACE(
484  "Moving layer index %d from plane index: %d to plane idex: %d. \n",
485  source_layers.at(0), total_planes - 1, total_planes - 2);
486 #endif
487  const OverlayLayer* layer = &(layers.at(source_layers.at(0)));
488  if (display_plane_manager_->ForceSeparatePlane(layers, old_plane,
489  layer)) {
490  return;
491  }
492 
493  old_plane.AddLayer(layer);
494  old_plane.ResetCompositionRegion();
495 
496  // If overlay has offscreen surfaces, discard
497  // them.
498  if (last_overlay.GetOffScreenTarget()) {
499  display_plane_manager_->MarkSurfacesForRecycling(
500  &last_overlay, surfaces_not_inuse_, true);
501  }
502 
503  // Let's allocate an offscreen surface if needed.
504  display_plane_manager_->SetOffScreenPlaneTarget(old_plane);
505 
506  last_overlay.GetDisplayPlane()->SetInUse(false);
507  composition->erase(composition->begin() + (total_planes - 1));
508  *render_layers = true;
509  *can_ignore_commit = false;
510  }
511  }
512  } else if (pop_plane) {
513  DisplayPlaneState& last_plane = composition->back();
514  // Ignore in case we are already scanning out the layer.
515  if (!last_plane.NeedsOffScreenComposition()) {
516  return;
517  }
518 
519  const std::vector<size_t>& source_layers = last_plane.GetSourceLayers();
520  // Let's try to allocate more planes in case we have free planes and
521  // layers to map.
522  *add_index = source_layers.at(0);
523  display_plane_manager_->MarkSurfacesForRecycling(&last_plane,
524  surfaces_not_inuse_, true);
525  composition->erase(composition->begin() + (total_planes - 1));
526  }
527 }

529 bool DisplayQueue::QueueUpdate(std::vector<HwcLayer*>& source_layers,
530  int32_t* retire_fence, bool* ignore_clone_update,
531  PixelUploaderCallback* call_back,
532  bool handle_constraints) {
533  CTRACE();
534  ScopedIdleStateTracker tracker(idle_tracker_, compositor_,
535  resource_manager_.get(), this);
536  if (tracker.IgnoreUpdate()) {
537  return true;
538  }
539 
540  size_t size = source_layers.size();
541  size_t previous_size = in_flight_layers_.size();
542  std::vector<OverlayLayer> layers;
543  int remove_index = -1;
544  int add_index = -1;
545  // If last commit failed, lets force full validation as
546  // state might be all wrong in our side.
547  bool idle_frame = tracker.RenderIdleMode();
548  bool validate_layers =
549  last_commit_failed_update_ || previous_plane_state_.empty();
550  *retire_fence = -1;
551  uint32_t z_order = 0;
552  bool has_video_layer = false;
553  bool re_validate_commit = false;
554  needs_clone_validation_ = false;
555 
556  for (size_t layer_index = 0; layer_index < size; layer_index++) {
557  HwcLayer* layer = source_layers.at(layer_index);
558  layer->SetReleaseFence(-1);
559  if (!layer->IsVisible())
560  continue;
561 
562  layers.emplace_back();
563  OverlayLayer* overlay_layer = &(layers.back());
564  OverlayLayer* previous_layer = NULL;
565  if (previous_size > z_order) {
566  previous_layer = &(in_flight_layers_.at(z_order));
567  } else if (add_index == -1) {
568  add_index = z_order;
569  }
570 
571  if (scaling_tracker_.scaling_state_ == ScalingTracker::kNeedsScaling) {
572  HwcRect<int> display_frame = layer->GetDisplayFrame();
573  display_frame.left =
574  display_frame.left +
575  (display_frame.left * scaling_tracker_.scaling_width);
576  display_frame.top = display_frame.top +
577  (display_frame.top * scaling_tracker_.scaling_height);
578  display_frame.right =
579  display_frame.right +
580  (display_frame.right * scaling_tracker_.scaling_width);
581  display_frame.bottom =
582  display_frame.bottom +
583  (display_frame.bottom * scaling_tracker_.scaling_height);
584 
585  overlay_layer->InitializeFromScaledHwcLayer(
586  layer, resource_manager_.get(), previous_layer, z_order, layer_index,
587  display_frame, display_plane_manager_->GetHeight(), plane_transform_,
588  handle_constraints, fb_manager_);
589  } else {
590  overlay_layer->InitializeFromHwcLayer(
591  layer, resource_manager_.get(), previous_layer, z_order, layer_index,
592  display_plane_manager_->GetHeight(), plane_transform_,
593  handle_constraints, fb_manager_);
594  }
595 
596  if (!overlay_layer->IsVisible()) {
597  layers.pop_back();
598  continue;
599  }
600 
601  if (overlay_layer->IsVideoLayer()) {
602  has_video_layer = true;
603  }
604 
605  if (overlay_layer->NeedsRevalidation()) {
606  re_validate_commit = true;
607  } else if (overlay_layer->HasLayerContentChanged()) {
608  idle_frame = false;
609  }
610 
611  if (overlay_layer->IsCursorLayer()) {
612  tracker.FrameHasCursor();
613  }
614 
615  z_order++;
616 
617  // Handle case where Media layer has been destroyed/created or has changed
618  // z-order.
619  if (add_index != 0 && overlay_layer->IsVideoLayer()) {
620  if ((previous_layer && !previous_layer->IsVideoLayer()) ||
621  !previous_layer) {
622  add_index = 0;
623 #ifdef SURFACE_TRACING
624  ISURFACETRACE(
625  "Video layer has changed between frames: remove_index: %d "
626  "add_index: "
627  "%d \n",
628  remove_index, add_index);
629 #endif
630  continue;
631  }
632  }
633 
634  if (add_index == 0 || validate_layers ||
635  ((add_index != -1) && (remove_index != -1))) {
636  needs_clone_validation_ = true;
637  continue;
638  }
639 
640  // Handle case where Cursor layer has been destroyed/created or has changed
641  // z-order.
642  if (previous_layer &&
643  previous_layer->IsCursorLayer() != overlay_layer->IsCursorLayer()) {
644  if (remove_index == -1)
645  remove_index = previous_layer->GetZorder();
646 
647  // Treat this case as if a new layer has been created.
648  if (add_index == -1)
649  add_index = overlay_layer->GetZorder();
650 
651 #ifdef SURFACE_TRACING
652  ISURFACETRACE(
653  "Cursor layer has changed between frames: remove_index: %d "
654  "add_index: "
655  "%d \n",
656  remove_index, add_index);
657 #endif
658  }
659  }
660 
661  // We may have skipped layers which are not visible.
662  size = layers.size();
663  if ((add_index == 0) || validate_layers) {
664  // If index is zero, no point trying for incremental validation.
665  validate_layers = true;
666  } else if (previous_size > size) {
667  if (remove_index == -1) {
668  remove_index = size;
669  } else if (add_index != -1) {
670  remove_index = std::min(add_index, remove_index);
671  }
672  }
673 
674  if (idle_frame) {
675  if ((add_index != -1) || (remove_index != -1) || re_validate_commit) {
676  idle_frame = false;
677  }
678  }
679 
680  if (!validate_layers)
681  validate_layers = idle_frame;
