40 void ResetRectToRegion(const HwcRegion& hwc_region, HwcRect<int>& rect) {
41  size_t total_rects = hwc_region.size();
42  if (total_rects == 0) {
43  rect.left = 0;
44  rect.top = 0;
45  rect.right = 0;
46  rect.bottom = 0;
47  return;
48  }
49 
50  const HwcRect<int>& new_rect = hwc_region.at(0);
51  rect.left = new_rect.left;
52  rect.top = new_rect.top;
53  rect.right = new_rect.right;
54  rect.bottom = new_rect.bottom;
55 
56  for (uint32_t r = 1; r < total_rects; r++) {
57  const HwcRect<int>& temp = hwc_region.at(r);
58  rect.left = std::min(rect.left, temp.left);
59  rect.top = std::min(rect.top, temp.top);
60  rect.right = std::max(rect.right, temp.right);
61  rect.bottom = std::max(rect.bottom, temp.bottom);
62  }
63 }
