54 // POI is the point of interest while traversing through x coordinates
55 struct POI {
56  EventType type;
57  uint64_t rect_id;
58  uint64_t x;
59  uint64_t top_y;

67 // This function will take active region and right x
68 // For an active region there will be set of YPOI
69 // It will traverse through each y_poi and given out
70 // rectangle with rect_ids active at that time.
71 void GenerateOutLayers(Region *reg, uint64_t x,
72  const HwcRect<int> &damage_region,
73  std::vector<RectSet<int>> *out) {
74  Rect<int> out_rect;
75  out_rect.left = std::max(damage_region.left, static_cast<int>(reg->sx));
76  out_rect.right = std::min(damage_region.right, static_cast<int>(x));
77  RectIDs rect_ids;
78 
79  for (std::set<YPOI>::iterator y_poi_it = reg->y_points.begin();
80  y_poi_it != reg->y_points.end(); y_poi_it++) {
81  const YPOI &y_poi = *y_poi_it;
82  // No need to check for start or end event
83  // as rect_ids is empty
84  if (rect_ids.isEmpty()) {
85  out_rect.top = std::max(damage_region.top, static_cast<int>(y_poi.y));
86  rect_ids.add(y_poi.rect_id);
87  } else {
88  if (out_rect.top == static_cast<int>(y_poi.y)) {
89  if (y_poi.type == START) {
90  rect_ids.add(y_poi.rect_id);
91  } else {
92  rect_ids.subtract(y_poi.rect_id);
93  }
94  continue;
95  }
96  out_rect.bottom = y_poi.y;
97  if (AnalyseOverlap(damage_region, out_rect) == kOutside)
98  continue;
99 
100  out->emplace_back(RectSet<int>(rect_ids, out_rect));
101  out_rect.top = std::max(damage_region.top, static_cast<int>(y_poi.y));
102  if (y_poi.type == START) {
103  rect_ids.add(y_poi.rect_id);
104  } else {
105  rect_ids.subtract(y_poi.rect_id);
106  }
107  }
108  }
109 }

111 // This function will remove y coordinates corresponding to given rect_id
112 void RemoveYpois(Region *reg, uint64_t rect_id) {
113  std::set<YPOI>::iterator top_it = reg->y_points.begin();
114  while (top_it != reg->y_points.end()) {
115  if ((*top_it).rect_id == rect_id) {
116  reg->y_points.erase(top_it++);
117  } else {
118  top_it++;
119  }
120  }
121 }

123 bool compare_region(const Region *first, const Region *second) {
124  uint64_t first_min_y = (*(first->y_points.begin())).y;
125  uint64_t second_min_y = (*(second->y_points.begin())).y;
126  return (first_min_y < second_min_y);

129 void get_draw_regions(const std::vector<Rect<int>> &in,
130  const HwcRect<int> &damage_region,
131  std::vector<RectSet<int>> *out) {
132  if (in.size() > RectIDs::max_elements) {
133  return;
134  }
135 
136  // Set of all point of interests from input rectangles.
137  std::set<POI> pois;
138  std::list<Region *> imp_reg;
139  std::list<Region> active_regions;
140 
141  // This loop will add all point of interests into pois.
142  for (uint64_t i = 0; i < in.size(); i++) {
143  const Rect<int> &rect = in[i];
144 
145  // Filter out empty or invalid rects.
146  if (rect.left >= rect.right || rect.top >= rect.bottom)
147  continue;
148 
149  if (AnalyseOverlap(damage_region, rect) == kOutside)
150  continue;
151 
152  POI poi;
153  poi.rect_id = i;
154  poi.x = std::max(damage_region.left, rect.left);
155  poi.top_y = std::max(damage_region.top, rect.top);
156  poi.bot_y = std::min(damage_region.bottom, rect.bottom);
157  poi.type = START;
158  pois.insert(poi);
159 
160  poi.type = END;
161  poi.x = std::min(damage_region.right, rect.right);
162  pois.insert(poi);
163  }
164 
165  for (std::set<POI>::iterator it = pois.begin(); it != pois.end(); ++it) {
166  const POI &poi = *it;
167  // First rectangle has to be inserted into active region
168  // This condition will be true if existing all active
169  // regions are already copied to out.
170  // If current poi is of type END there are no active regions,
171  // then this poi might already covered in previous pass
172  if (active_regions.size() == 0 && poi.type == START) {
173  Region reg;
174  reg.sx = poi.x;
175  YPOI y_poi;
176 
177  y_poi.rect_id = poi.rect_id;
178  y_poi.type = START;
179  y_poi.y = poi.top_y;
180  reg.y_points.insert(y_poi);
181 
182  y_poi.type = END;
183  y_poi.y = poi.bot_y;
184  reg.y_points.insert(y_poi);
185 
186  RectIDs rectIds;
187  rectIds.add(poi.rect_id);
188  reg.rect_ids = rectIds;
189  active_regions.push_back(reg);
190  continue;
191  }
192 
193  // If active_regions in not empty, Check if current
194  // poi y points fall in range of any existing
195  // active_regions.
196  // If yes, get that active region and do further processing
197  // If No, create a new region and insert into active regions
198  // If it is start event then there is possibility that multiple
199  // active_regions get impacted.
200  // If it is end event then one or none active_regions will get
201  // impacted.
202  bool found = false;
203  imp_reg.clear();
204  std::list<Region>::iterator it_reg = active_regions.begin();
205  while (it_reg != active_regions.end()) {
206  Region &cur_reg = *it_reg;
207  uint64_t min_y = (*(cur_reg.y_points.begin())).y;
208  uint64_t max_y = (*(cur_reg.y_points.rbegin())).y;
209  // If bottom y is less than minimum y in region or top y is greater than
210  // max y in region, then this region is not impacted by this rect
211  if (poi.bot_y <= min_y || poi.top_y >= max_y) {
212  it_reg++;
213  continue;
214  } else {
215  found = true;
216  // Found atleast one affected active region. If it is start event,
217  // add rect_id to cur_reg.rect_ids, also top_y and bot_y to
218  // cur_reg.y_points. if it is end event, remove rect_id from
219  // cur_reg.rect_ids and also top_y and bot_y from cur_reg.y_points.
220  // Also, if it is end event, check cur_reg.rect_ids is non empty,
221  // if it is empty remove region from active_regions.
222  // If it is start or end event, check next poi.x and see if it is same
223  // and
224  // those y coordinates fall in this region and it is END event, if yes
225  // 1) remove that rect_id and y coordinates as well
226  // 2)contine to check next poi.x until you find mismatch x.
227  if (poi.x == cur_reg.sx) {
228  if (poi.type == START) {
229  cur_reg.rect_ids.add(poi.rect_id);
230  imp_reg.push_back(&cur_reg);
231  }
232 
233  it_reg++;
234  continue;
235  }
236  if (poi.type == START) {
237  GenerateOutLayers(&cur_reg, poi.x, damage_region, out);
238  cur_reg.sx = poi.x;
239  cur_reg.rect_ids.add(poi.rect_id);
240  imp_reg.push_back(&cur_reg);
241  std::set<POI>::iterator next_poi_it = it;
242  next_poi_it++;
243  for (; next_poi_it != pois.end(); next_poi_it++) {
244  const POI &next_poi = *next_poi_it;
245  if (next_poi.x != poi.x) {
246  break;
247  } else {
248  if (next_poi.bot_y <= min_y || next_poi.top_y >= max_y ||
249  next_poi.type == START) {
250  continue;
251  }
252  cur_reg.rect_ids.subtract(next_poi.rect_id);
253  RemoveYpois(&cur_reg, next_poi.rect_id);
254  }
255  }
256  it_reg++;
257  } else {
258  GenerateOutLayers(&cur_reg, poi.x, damage_region, out);
259  RemoveYpois(&cur_reg, poi.rect_id);
260  cur_reg.sx = poi.x;
261  cur_reg.rect_ids.subtract(poi.rect_id);
262 
263  std::set<POI>::iterator next_poi_it = it;
264  next_poi_it++;
265  for (; next_poi_it != pois.end(); next_poi_it++) {
266  const POI &next_poi = *next_poi_it;
267  if (next_poi.x != poi.x) {
268  break;
269  } else {
270  if (next_poi.bot_y <= min_y || next_poi.top_y >= max_y ||
271  next_poi.type == START) {
272  continue;
273  }
274  cur_reg.rect_ids.subtract(next_poi.rect_id);
275  RemoveYpois(&cur_reg, next_poi.rect_id);
276  }
277  }
278  if (cur_reg.rect_ids.isEmpty()) {
279  active_regions.erase(it_reg++);
280  } else {
281  it_reg++;
282  }
283  }
284  }
285  }
286  // If no affected active region found, add new active region
287  if (!found && poi.type == START) {
288  Region reg;
289  reg.sx = poi.x;
290  YPOI y_poi;
291 
292  y_poi.rect_id = poi.rect_id;
293  y_poi.type = START;
294  y_poi.y = poi.top_y;
295  reg.y_points.insert(y_poi);
296 
297  y_poi.type = END;
298  y_poi.y = poi.bot_y;
299  reg.y_points.insert(y_poi);
300 
301  RectIDs rectIds;
302  rectIds.add(poi.rect_id);
303  reg.rect_ids = rectIds;
304  active_regions.push_back(reg);
305  } else {
306  if (imp_reg.size() > 1 && poi.type == START) {
307  imp_reg.sort(compare_region);
308  uint64_t cur_y = 0;
309  for (std::list<Region *>::iterator cur_imp_reg_it = imp_reg.begin();
310  cur_imp_reg_it != imp_reg.end(); cur_imp_reg_it++) {
311  Region &cur_imp_reg = *(*cur_imp_reg_it);
312  YPOI y_poi;
313  y_poi.rect_id = poi.rect_id;
314  y_poi.type = START;
315 
316  if (cur_y == 0) {
317  y_poi.y = poi.top_y;
318  } else {
319  y_poi.y = cur_y;
320  }
321  // This is to split vertical
322  // line into all impacted
323  // regions.
324  cur_imp_reg.y_points.insert(y_poi);
325  // Take bottom of current region as start of next impacted region
326  cur_y = (*(cur_imp_reg.y_points.rbegin())).y;
327  std::list<Region *>::iterator next_imp_reg_it = cur_imp_reg_it;
328  next_imp_reg_it++;
329  if (next_imp_reg_it == imp_reg.end()) {
330  // If there is an another
331  // region which is impacted, no
332  // need to add anything.
333  // if there is no other active region left,
334  // take bottom y and push into this active region
335  y_poi.y = poi.bot_y;
336  } else {
337  y_poi.y = cur_y;
338  }
339  y_poi.type = END;
340  cur_imp_reg.y_points.insert(y_poi);
341  }
342  } else if (imp_reg.size() == 1 && poi.type == START) {
343  // Only one region got impacted add y coordinated to that region
344  std::list<Region *>::iterator cur_imp_reg_it = imp_reg.begin();
345  YPOI y_poi;
346  y_poi.rect_id = poi.rect_id;
347  y_poi.type = START;
348  y_poi.y = poi.top_y;
349  (*cur_imp_reg_it)->y_points.insert(y_poi);
350  y_poi.type = END;
351  y_poi.y = poi.bot_y;
352  (*cur_imp_reg_it)->y_points.insert(y_poi);
353  }
354  }
355  }
356 }
