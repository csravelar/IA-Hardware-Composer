52 std::shared_ptr<OverlayBuffer>& ResourceManager::FindCachedBuffer(
53  const uint32_t& native_buffer) {
54  BUFFER_MAP& first_map = cached_buffers_[0];
55  static std::shared_ptr<OverlayBuffer> pBufNull = nullptr;
56  for (auto& map : cached_buffers_) {
57  if (map.empty()) {
58  continue;
59  }
60 
61  BUFFER_MAP::iterator it = map.find(native_buffer);
62 
63  if (it != map.end()) {
64  std::shared_ptr<OverlayBuffer>& pBuf = it->second;
65  if (&map != &first_map) {
66  first_map.emplace(std::make_pair(native_buffer, pBuf));
67  }
68 #ifdef RESOURCE_CACHE_TRACING
69  hit_count_++;
70 #endif
71  return pBuf;
72  }
73  }
74 
75 #ifdef RESOURCE_CACHE_TRACING
76  miss_count_++;
77  if (miss_count_ % 100 == 0)
78  ICACHETRACE("cache miss count is %llu, while hit count is %llu",
79  miss_count_, hit_count_);
80 #endif
81 
82  return pBufNull;
83 }

146 bool ResourceManager::PreparePurgedResources() {
147  if (cached_buffers_.size() > 4)
148  cached_buffers_.pop_back();
149 
150  if (purged_resources_.empty() && purged_media_resources_.empty())
151  return false;
152 
153  lock_.lock();
154  if (!purged_resources_.empty()) {
155  size_t purged_size = purged_resources_.size();
156  for (size_t i = 0; i < purged_size; i++) {
157  const ResourceHandle& handle = purged_resources_.at(i);
158  destroy_gl_resources_.emplace_back();
159  ResourceHandle& temp = destroy_gl_resources_.back();
160  std::memcpy(&temp, &handle, sizeof temp);
161  }
162  std::vector<ResourceHandle>().swap(purged_resources_);
163  }
164 
165  if (!purged_media_resources_.empty()) {
166  size_t purged_size = purged_media_resources_.size();
167  for (size_t i = 0; i < purged_size; i++) {
168  const MediaResourceHandle& handle = purged_media_resources_.at(i);
169  destroy_media_resources_.emplace_back();
170  MediaResourceHandle& temp = destroy_media_resources_.back();
171  std::memcpy(&temp, &handle, sizeof temp);
172  }
173  std::vector<MediaResourceHandle>().swap(purged_media_resources_);
174  }
175 
176  destroy_gpu_resources_ = has_purged_gpu_resources_;
177  has_purged_gpu_resources_ = false;
178  lock_.unlock();
179 
180  return true;
181 }
