37 RingBuffer::Allocation RingBuffer::Allocate(size_t size, size_t alignment) {
38  if (size > buffer_size_)
39  return Allocation();
40 
41  size_t base = write_offset_;
42  size_t padding = ((alignment - (base & (alignment - 1))) & (alignment - 1));
43  size_t jump = padding + size;
44  if (base + padding + size > buffer_size_) {
45  base = 0;
46  padding = 0;
47  jump = buffer_size_ - write_offset_ + size;
48  }
49 
50  if (IsSpanInUse(base, size)) {
51  return Allocation();
52  }
53 
54  jump_queue_.emplace_back(base + padding, jump);
55 
56  write_offset_ += jump;
57  write_offset_ %= buffer_size_;
58

60 }

62 void RingBuffer::Free(uint8_t *ptr) {
63  size_t base = ptr - buffer_;
64  bool found = false;
65  for (jump_entry_t &entry : jump_queue_) {
66  if (entry.base == base) {
67  entry.free = true;
68  found = true;
69  break;
70  }
71  }
72 
73  auto it = jump_queue_.begin();

75  if (!it->free)

95 bool RingBuffer::IsSpanInUse(size_t first, size_t size) {
96  if (jump_queue_.empty())
97  return false;
98 
99  size_t last = first + size - 1;
100 
101  size_t use_first = read_offset_;
102  size_t use_last = (write_offset_ == 0 ? buffer_size_ : write_offset_) - 1;
103  if (use_last < use_first)
104  use_last += buffer_size_;
105 
106  if (first <= use_first && last >= use_first)
107  return true;
108 
109  if (first <= use_last && last >= use_last)
110  return true;
111 
112  if (use_first <= first && use_last >= first)
113  return true;
114 
115  if (use_first <= last && use_last >= last)
116  return true;
117 
118  return false;
119 }
