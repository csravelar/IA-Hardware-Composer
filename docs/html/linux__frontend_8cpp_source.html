73  void Callback(uint32_t display, bool connected) {
74  auto hook = reinterpret_cast<IAHWC_PFN_HOTPLUG>(hook_);
75  uint32_t status;
76  if (connected) {
77  status = static_cast<uint32_t>(IAHWC_DISPLAY_STATUS_CONNECTED);
78  if (display_)
79  display_->RunPixelUploader(true);
80  } else {
81  status = static_cast<uint32_t>(IAHWC_DISPLAY_STATUS_DISCONNECTED);
82  if (display_)
83  display_->RunPixelUploader(false);
84  }
85 
86  if (hook)
87  hook(data_, display, status);
88  }
89 
90  private:
91  iahwc_callback_data_t data_;
92  iahwc_function_ptr_t hook_;
93  IAHWC::IAHWCDisplay* display_;
94 };

253 int IAHWC::RegisterCallback(int32_t description, uint32_t display_id,
254  iahwc_callback_data_t data,
255  iahwc_function_ptr_t hook) {
256  switch (description) {
257  case IAHWC_CALLBACK_VSYNC: {
258  if (display_id >= displays_.size())
259  return IAHWC_ERROR_BAD_DISPLAY;
260  IAHWCDisplay* display = displays_.at(display_id);
261  return display->RegisterVsyncCallback(data, hook);
262  }
263  case IAHWC_CALLBACK_PIXEL_UPLOADER: {
264  if (display_id >= displays_.size())
265  return IAHWC_ERROR_BAD_DISPLAY;
266 
267  IAHWCDisplay* display = displays_.at(display_id);
268  display->RegisterPixelUploaderCallback(data, hook);
269  return IAHWC_ERROR_NONE;
270  }
271  case IAHWC_CALLBACK_HOTPLUG: {
272  if (display_id >= displays_.size())
273  return IAHWC_ERROR_BAD_DISPLAY;
274  for (auto display : displays_)
275  display->RegisterHotPlugCallback(data, hook);
276  return IAHWC_ERROR_NONE;
277  }
278 
279  default:
280  return IAHWC_ERROR_BAD_PARAMETER;
281  }
282 }

367 int IAHWC::IAHWCDisplay::PresentDisplay(int32_t* release_fd) {
368  std::vector<hwcomposer::HwcLayer*> layers;
369  /*
370  * Here the assumption is that the layer index set by the compositor
371  * is numbered from bottom -> top, i.e. the bottom most layer has the
372  * index of 0 and increases upwards.
373  */
374  uint32_t total_layers = layers_.size();
375  layers.resize(total_layers);
376  total_layers -= 1;
377 
378  for (std::pair<const iahwc_layer_t, IAHWCLayer>& l : layers_) {
379  IAHWCLayer& temp = l.second;
380  uint32_t layer_index = total_layers - temp.GetLayerIndex();
381  layers[layer_index] = temp.GetLayer();
382  }
383 
384  native_display_->Present(layers, release_fd, this);
385 
386  return IAHWC_ERROR_NONE;
387 }

519 int IAHWC::IAHWCLayer::SetRawPixelData(iahwc_raw_pixel_data bo) {
520  const NativeBufferHandler* buffer_handler =
521  raw_data_uploader_->GetNativeBufferHandler();
522  ClosePrimeHandles();
523  if (pixel_buffer_ &&
524  ((orig_height_ != bo.height) || (orig_stride_ != bo.stride))) {
525  if (upload_in_progress_) {
526  raw_data_uploader_->Synchronize();
527  }
528 
529  buffer_handler->ReleaseBuffer(pixel_buffer_);
530  buffer_handler->DestroyHandle(pixel_buffer_);
531  pixel_buffer_ = NULL;
532  }
533 
534  if (!pixel_buffer_) {
535  int layer_type =
536  layer_usage_ == IAHWC_LAYER_USAGE_CURSOR ? kLayerCursor : kLayerNormal;
537  bool modifier_used = false;
538  if (!buffer_handler->CreateBuffer(bo.width, bo.height, bo.format,
539  &pixel_buffer_, layer_type,
540  &modifier_used, 0, true)) {
541  ETRACE("PixelBuffer: CreateBuffer failed");
542  return -1;
543  }
544 
545  if (!buffer_handler->ImportBuffer(pixel_buffer_)) {
546  ETRACE("PixelBuffer: ImportBuffer failed");
547  return -1;
548  }
549 
550  if (pixel_buffer_->meta_data_.prime_fds_[0] <= 0) {
551  ETRACE("PixelBuffer: prime_fd_ is invalid.");
552  return -1;
553  }
554 
555  orig_width_ = bo.width;
556  orig_height_ = bo.height;
557  orig_stride_ = bo.stride;
558  iahwc_layer_.SetNativeHandle(pixel_buffer_);
559  }
560 
561  upload_in_progress_ = true;
562  raw_data_uploader_->UpdateLayerPixelData(
563  pixel_buffer_, orig_width_, orig_height_, orig_stride_, bo.callback_data,
564  (uint8_t*)bo.buffer, this, iahwc_layer_.GetSurfaceDamage());
565 
566  return IAHWC_ERROR_NONE;
567 }

598 int IAHWC::IAHWCLayer::SetLayerTransform(int32_t layer_transform) {
599  // 270* and 180* cannot be combined with flips. More specifically, they
600  // already contain both horizontal and vertical flips, so those fields are
601  // redundant in this case. 90* rotation can be combined with either horizontal
602  // flip or vertical flip, so treat it differently
603  int32_t temp = 0;
604  if (layer_transform == IAHWC_TRANSFORM_ROT_270) {
605  temp = hwcomposer::HWCTransform::kTransform270;
606  } else if (layer_transform == IAHWC_TRANSFORM_ROT_180) {
607  temp = hwcomposer::HWCTransform::kTransform180;
608  } else {
609  if (layer_transform & IAHWC_TRANSFORM_FLIP_H)
610  temp |= hwcomposer::HWCTransform::kReflectX;
611  if (layer_transform & IAHWC_TRANSFORM_FLIP_V)
612  temp |= hwcomposer::HWCTransform::kReflectY;
613  if (layer_transform & IAHWC_TRANSFORM_ROT_90)
614  temp |= hwcomposer::HWCTransform::kTransform90;
615  }
616  iahwc_layer_.SetTransform(temp);
617 
618  return IAHWC_ERROR_NONE;
619 }
