29 void RenderState::ConstructState(std::vector<OverlayLayer> &layers,
30  const CompositionRegion &region,
31  uint32_t downscaling_factor,
32  bool uses_display_up_scaling,
33  bool use_plane_transform) {
34  float bounds[4];
35  std::copy_n(region.frame.bounds, 4, bounds);
36  x_ = bounds[0];
37  y_ = bounds[1];
38  width_ = bounds[2] - bounds[0];
39  height_ = bounds[3] - bounds[1];
40  scissor_x_ = x_;
41  scissor_y_ = y_;
42  scissor_width_ = width_;
43  scissor_height_ = height_;
44  const std::vector<size_t> &source = region.source_layers;
45  for (size_t texture_index : source) {
46  OverlayLayer &layer = layers.at(texture_index);
47  layer_state_.emplace_back();
48  RenderState::LayerState &src = layer_state_.back();
49  src.layer_index_ = texture_index;
50  bool swap_xy = false;
51  bool flip_xy[2] = {false, false};
52  uint32_t transform = layer.GetTransform();
53  if (use_plane_transform) {
54  transform = layer.GetPlaneTransform();
55  }
56 
57  switch (transform) {
58  case HWCTransform::kTransform180: {
59  swap_xy = false;
60  flip_xy[0] = true;
61  flip_xy[1] = true;
62  break;
63  }
64  case HWCTransform::kTransform270: {
65  swap_xy = true;
66  flip_xy[0] = true;
67  flip_xy[1] = false;
68  break;
69  }
70  case HWCTransform::kTransform90: {
71  swap_xy = true;
72  flip_xy[0] = false;
73  flip_xy[1] = true;
74  break;
75  }
76  case HWCTransform::kTransform90 | HWCTransform::kReflectX: {
77  swap_xy = true;
78  flip_xy[0] = true;
79  flip_xy[1] = true;
80  break;
81  }
82  case HWCTransform::kTransform90 | HWCTransform::kReflectY: {
83  swap_xy = true;
84  flip_xy[0] = false;
85  flip_xy[1] = false;
86  break;
87  }
88  default: {
89  if (layer.GetTransform() & HWCTransform::kReflectX)
90  flip_xy[0] = true;
91  if (layer.GetTransform() & HWCTransform::kReflectY)
92  flip_xy[1] = true;
93  }
94  }
95 
96  if (swap_xy)
97  std::copy_n(&TransformMatrices[4], 4, src.texture_matrix_);
98  else
99  std::copy_n(&TransformMatrices[0], 4, src.texture_matrix_);
100 
101  HwcRect<float> display_rect;
102  float display_size[2];
103 
104  if (uses_display_up_scaling) {
105  display_rect = layer.GetSourceCrop();
106  display_size[0] = static_cast<float>(layer.GetSourceCropWidth());
107  display_size[1] = static_cast<float>(layer.GetSourceCropHeight());
108  } else {
109  const HwcRect<int> &display_Rect = layer.GetDisplayFrame();
110  display_rect.left = static_cast<float>(display_Rect.left);
111  display_rect.right = static_cast<float>(display_Rect.right);
112  display_rect.top = static_cast<float>(display_Rect.top);
113  display_rect.bottom = static_cast<float>(display_Rect.bottom);
114  if (downscaling_factor > 1) {
115  display_rect.right =
116  display_rect.right -
117  ((display_rect.right - display_rect.left) / downscaling_factor);
118 
119  display_size[0] = display_rect.right - display_rect.left;
120  display_size[1] = display_rect.bottom - display_rect.top;
121  } else {
122  display_size[0] = static_cast<float>(layer.GetDisplayFrameWidth());
123  display_size[1] = static_cast<float>(layer.GetDisplayFrameHeight());
124  }
125  }
126 
127  float tex_width = static_cast<float>(layer.GetBuffer()->GetWidth());
128  float tex_height = static_cast<float>(layer.GetBuffer()->GetHeight());
129  const HwcRect<float> &source_crop = layer.GetSourceCrop();
130 
131  HwcRect<float> crop_rect(
132  source_crop.left / tex_width, source_crop.top / tex_height,
133  source_crop.right / tex_width, source_crop.bottom / tex_height);
134 
135  float crop_size[2] = {crop_rect.bounds[2] - crop_rect.bounds[0],
136  crop_rect.bounds[3] - crop_rect.bounds[1]};
137 
138  for (int j = 0; j < 4; j++) {
139  int b = j ^ (swap_xy ? 1 : 0);
140  float bound_percent =
141  (bounds[b] - display_rect.bounds[b % 2]) / display_size[b % 2];
142  if (flip_xy[j % 2]) {
143  src.crop_bounds_[j] =
144  crop_rect.bounds[j % 2 + 2] - bound_percent * crop_size[j % 2];
145  } else {
146  src.crop_bounds_[j] =
147  crop_rect.bounds[j % 2] + bound_percent * crop_size[j % 2];
148  }
149  }
150 
151  if (layer.GetBlending() == HWCBlending::kBlendingNone) {
152  src.alpha_ = src.premult_ = 1.0f;
153  break;
154  }
155 
156  src.alpha_ = layer.GetAlpha() / 255.0f;
157  src.premult_ =
158  (layer.GetBlending() == HWCBlending::kBlendingPremult) ? 1.0f : 0.0f;
159  }
160 }
