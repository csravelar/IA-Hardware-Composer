128 bool Gralloc1BufferHandler::CreateBuffer(uint32_t w, uint32_t h, int format,
129  HWCNativeHandle *handle,
130  uint32_t layer_type,
131  bool *modifier_used,
132  int64_t preferred_modifier,
133  bool /*raw_pixel_buffer*/) const {
134  struct gralloc_handle *temp = new struct gralloc_handle();
135  gralloc1_device_t *gralloc1_dvc =
136  reinterpret_cast<gralloc1_device_t *>(device_);
137 
138  create_descriptor_(gralloc1_dvc, &temp->gralloc1_buffer_descriptor_t_);
139  uint32_t usage = 0;
140  uint32_t pixel_format = 0;
141  bool force_normal_usage = false;
142  if (format != 0) {
143  pixel_format = DrmFormatToHALFormat(format);
144  }
145 
146  if (pixel_format == 0) {
147  pixel_format = HAL_PIXEL_FORMAT_RGBA_8888;
148  }
149 
150  set_format_(gralloc1_dvc, temp->gralloc1_buffer_descriptor_t_, pixel_format);
151 #ifdef ENABLE_RBC
152  if (set_modifier_) {
153  uint64_t modifier = 0;
154  if (preferred_modifier != -1) {
155  modifier = preferred_modifier;
156  } else {
157  modifier = choose_drm_modifier(format);
158  }
159 
160  set_modifier_(gralloc1_dvc, temp->gralloc1_buffer_descriptor_t_, modifier);
161  }
162 
163  if (modifier_used) {
164  *modifier_used = true;
165  }
166 #else
167  if (modifier_used) {
168  *modifier_used = false;
169  }
170 #endif
171 
172  if ((layer_type == hwcomposer::kLayerVideo) &&
173  !IsSupportedMediaFormat(format)) {
174  ETRACE("Forcing normal usage for Video Layer. \n");
175  force_normal_usage = true;
176  }
177 
178  if ((layer_type == hwcomposer::kLayerNormal) || force_normal_usage) {
179  usage |= GRALLOC1_CONSUMER_USAGE_HWCOMPOSER |
180  GRALLOC1_PRODUCER_USAGE_GPU_RENDER_TARGET |
181  GRALLOC1_CONSUMER_USAGE_GPU_TEXTURE;
182  } else if (layer_type == hwcomposer::kLayerVideo) {
183  switch (pixel_format) {
184  case HAL_PIXEL_FORMAT_YCbCr_422_I:
185  case HAL_PIXEL_FORMAT_Y8:
186  usage |= GRALLOC1_CONSUMER_USAGE_GPU_TEXTURE |
187  GRALLOC1_PRODUCER_USAGE_VIDEO_DECODER;
188  break;
189  default:
190  usage |= GRALLOC1_PRODUCER_USAGE_CAMERA |
191  GRALLOC1_CONSUMER_USAGE_CAMERA |
192  GRALLOC1_PRODUCER_USAGE_VIDEO_DECODER |
193  GRALLOC1_CONSUMER_USAGE_GPU_TEXTURE;
194  }
195  } else if (layer_type == hwcomposer::kLayerCursor) {
196  usage |= GRALLOC1_CONSUMER_USAGE_CURSOR;
197  }
198 
199  set_consumer_usage_(gralloc1_dvc, temp->gralloc1_buffer_descriptor_t_, usage);
200  set_producer_usage_(gralloc1_dvc, temp->gralloc1_buffer_descriptor_t_, usage);
201  set_dimensions_(gralloc1_dvc, temp->gralloc1_buffer_descriptor_t_, w, h);
202  allocate_(gralloc1_dvc, 1, &temp->gralloc1_buffer_descriptor_t_,
203  &temp->handle_);
204 
205  if (!temp->handle_) {
206  ETRACE("Failed to allocate buffer \n");
207  }
208 
209  temp->hwc_buffer_ = true;
210  *handle = temp;
211 
212  return true;
213 }

280  GRALLOC1_PRODUCER_USAGE_CPU_WRITE_OFTEN,
