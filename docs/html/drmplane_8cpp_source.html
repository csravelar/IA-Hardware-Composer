33 bool DrmPlane::Property::Initialize(
34  uint32_t fd, const char* name,
35  const ScopedDrmObjectPropertyPtr& plane_props, uint32_t* rotation,
36  uint64_t* in_formats_prop_value) {
37  uint32_t count_props = plane_props->count_props;
38  for (uint32_t i = 0; i < count_props; i++) {
39  ScopedDrmPropertyPtr property(
40  drmModeGetProperty(fd, plane_props->props[i]));
41  if (property && !strcmp(property->name, name)) {
42  id = property->prop_id;
43  if (rotation) {
44  uint32_t temp = 0;
45  for (int enum_index = 0; enum_index < property->count_enums;
46  enum_index++) {
47  struct drm_mode_property_enum* penum = &(property->enums[enum_index]);
48  if (!strcmp(penum->name, "rotate-90")) {
49  temp |= DRM_MODE_ROTATE_90;
50  }
51  if (!strcmp(penum->name, "rotate-180"))
52  temp |= DRM_MODE_ROTATE_180;
53  else if (!strcmp(penum->name, "rotate-270"))
54  temp |= DRM_MODE_ROTATE_270;
55  else if (!strcmp(penum->name, "rotate-0"))
56  temp |= DRM_MODE_ROTATE_0;
57  }
58 
59  *rotation = temp;
60  }
61  if (!strcmp(property->name, "IN_FORMATS")) {
62  if (in_formats_prop_value) {
63  *in_formats_prop_value = plane_props->prop_values[i];
64  }
65  }
66  break;
67  }
68  }
69  if (!id) {
70  ETRACE("Could not find property %s", name);
71  return false;
72  }
73  return true;
74 }

88 bool DrmPlane::Initialize(uint32_t gpu_fd,
89  const std::vector<uint32_t>& formats) {
90  supported_formats_ = formats;
91  uint32_t total_size = supported_formats_.size();
92  for (uint32_t j = 0; j < total_size; j++) {
93  uint32_t format = supported_formats_.at(j);
94  if (IsSupportedMediaFormat(format)) {
95  prefered_video_format_ = format;
96  break;
97  }
98  }
99 
100  for (uint32_t j = 0; j < total_size; j++) {
101  uint32_t format = supported_formats_.at(j);
102  switch (format) {
103  case DRM_FORMAT_BGRA8888:
104  case DRM_FORMAT_RGBA8888:
105  case DRM_FORMAT_ABGR8888:
106  case DRM_FORMAT_ARGB8888:
107  case DRM_FORMAT_RGB888:
108  case DRM_FORMAT_XRGB8888:
109  case DRM_FORMAT_XBGR8888:
110  case DRM_FORMAT_RGBX8888:
111  prefered_format_ = format;
112  break;
113  }
114  }
115 
116  if (type_ == DRM_PLANE_TYPE_PRIMARY) {
117  if (prefered_format_ != DRM_FORMAT_XBGR8888 &&
118  IsSupportedFormat(DRM_FORMAT_XBGR8888)) {
119  prefered_format_ = DRM_FORMAT_XBGR8888;
120  }
121  }
122 
123  if (prefered_video_format_ == 0) {
124  prefered_video_format_ = prefered_format_;
125  }
126 
127  ScopedDrmObjectPropertyPtr plane_props(
128  drmModeObjectGetProperties(gpu_fd, id_, DRM_MODE_OBJECT_PLANE));
129  if (!plane_props) {
130  ETRACE("Unable to get plane properties.");
131  return false;
132  }
133  uint32_t count_props = plane_props->count_props;
134  for (uint32_t i = 0; i < count_props; i++) {
135  ScopedDrmPropertyPtr property(
136  drmModeGetProperty(gpu_fd, plane_props->props[i]));
137  if (property && !strcmp(property->name, "type")) {
138  type_ = plane_props->prop_values[i];
139  break;
140  }
141  }
142 
143  bool ret = crtc_prop_.Initialize(gpu_fd, "CRTC_ID", plane_props);
144  if (!ret)
145  return false;
146 
147  ret = fb_prop_.Initialize(gpu_fd, "FB_ID", plane_props);
148  if (!ret)
149  return false;
150 
151  ret = crtc_x_prop_.Initialize(gpu_fd, "CRTC_X", plane_props);
152  if (!ret)
153  return false;
154 
155  ret = crtc_y_prop_.Initialize(gpu_fd, "CRTC_Y", plane_props);
156  if (!ret)
157  return false;
158 
159  ret = crtc_w_prop_.Initialize(gpu_fd, "CRTC_W", plane_props);
160  if (!ret)
161  return false;
162 
163  ret = crtc_h_prop_.Initialize(gpu_fd, "CRTC_H", plane_props);
164  if (!ret)
165  return false;
166 
167  ret = src_x_prop_.Initialize(gpu_fd, "SRC_X", plane_props);
168  if (!ret)
169  return false;
170 
171  ret = src_y_prop_.Initialize(gpu_fd, "SRC_Y", plane_props);
172  if (!ret)
173  return false;
174 
175  ret = src_w_prop_.Initialize(gpu_fd, "SRC_W", plane_props);
176  if (!ret)
177  return false;
178 
179  ret = src_h_prop_.Initialize(gpu_fd, "SRC_H", plane_props);
180  if (!ret)
181  return false;
182 
183  ret = rotation_prop_.Initialize(gpu_fd, "rotation", plane_props, &rotation_);
184  if (!ret)
185  ETRACE("Could not get rotation property");
186 
187  ret = alpha_prop_.Initialize(gpu_fd, "alpha", plane_props);
188  if (!ret)
189  ETRACE("Could not get alpha property");
190 
191  ret = in_fence_fd_prop_.Initialize(gpu_fd, "IN_FENCE_FD", plane_props);
192  if (!ret) {
193  ETRACE("Could not get IN_FENCE_FD property");
194  in_fence_fd_prop_.id = 0;
195  }
196 
197  // query and store supported modifiers for format, from in_formats
198  // property
199  uint64_t in_formats_prop_value = 0;
200  ret = in_formats_prop_.Initialize(gpu_fd, "IN_FORMATS", plane_props, NULL,
201  &in_formats_prop_value);
202  if (!ret) {
203  ETRACE("Could not get IN_FORMATS property");
204  }
205 
206  if (in_formats_prop_value != 0) {
207  drmModePropertyBlobPtr blob =
208  drmModeGetPropertyBlob(gpu_fd, in_formats_prop_value);
209  if (blob == nullptr || blob->data == nullptr) {
210  ETRACE("Unable to get property data\n");
211  return false;
212  }
213 
214  struct drm_format_modifier_blob* m =
215  (struct drm_format_modifier_blob*)(blob->data);
216  struct drm_format_modifier* mod_o =
217  (struct drm_format_modifier*)(void*)(((char*)m) + m->modifiers_offset);
218 
219  bool y_tiled_ccs_supported = false;
220  bool y_tiled_yf_ccs_supported = false;
221 
222  for (uint32_t j = 0; j < total_size; j++) {
223  uint32_t format = supported_formats_.at(j);
224  format_mods modifiers_obj;
225  modifiers_obj.format = format;
226  uint32_t format_index = j;
227 
228  struct drm_format_modifier* mod = mod_o;
229  for (int i = 0; i < (int)m->count_modifiers; i++, mod++) {
230  if (mod->formats & (1ULL << format_index)) {
231  modifiers_obj.mods.emplace_back(mod->modifier);
232  if (mod->modifier == I915_FORMAT_MOD_Y_TILED_CCS) {
233  y_tiled_ccs_supported = true;
234  } else if (mod->modifier == I915_FORMAT_MOD_Yf_TILED_CCS) {
235  y_tiled_yf_ccs_supported = true;
236  }
237  }
238  }
239 
240  if (modifiers_obj.mods.size() == 0) {
241  modifiers_obj.mods.emplace_back(DRM_FORMAT_MOD_NONE);
242  prefered_modifier_ = DRM_FORMAT_MOD_NONE;
243  } else {
244  if (y_tiled_ccs_supported) {
245  prefered_modifier_ = I915_FORMAT_MOD_Y_TILED_CCS;
246  } else if (y_tiled_yf_ccs_supported) {
247  prefered_modifier_ = I915_FORMAT_MOD_Yf_TILED_CCS;
248  } else {
249  prefered_modifier_ = modifiers_obj.mods.at(0);
250  }
251  }
252 
253  formats_modifiers_.emplace_back(modifiers_obj);
254  }
255 
256  drmModeFreePropertyBlob(blob);
257  }
258  return true;
259 }

261 bool DrmPlane::UpdateProperties(drmModeAtomicReqPtr property_set,
262  uint32_t crtc_id, const OverlayLayer* layer,
263  bool test_commit) const {
264  uint64_t alpha = 0xFF;
265  OverlayBuffer* buffer = layer->GetBuffer();
266  const HwcRect<int>& display_frame = layer->GetDisplayFrame();
267  const HwcRect<float>& source_crop = layer->GetSourceCrop();
268  int fence = kms_fence_;
269  if (test_commit) {
270  fence = layer->GetAcquireFence();
271  }
272 
273  if (layer->GetBlending() == HWCBlending::kBlendingPremult)
274  alpha = layer->GetAlpha();
275 
276  IDISPLAYMANAGERTRACE("buffer->GetFb() ---------------------- STARTS %d",
277  buffer->GetFb());
278  int success =
279  drmModeAtomicAddProperty(property_set, id_, crtc_prop_.id, crtc_id) < 0;
280  success |= drmModeAtomicAddProperty(property_set, id_, fb_prop_.id,
281  buffer->GetFb()) < 0;
282  success |= drmModeAtomicAddProperty(property_set, id_, crtc_x_prop_.id,
283  display_frame.left) < 0;
284  success |= drmModeAtomicAddProperty(property_set, id_, crtc_y_prop_.id,
285  display_frame.top) < 0;
286 
287  if (layer->IsCursorLayer()) {
288  success |= drmModeAtomicAddProperty(property_set, id_, crtc_w_prop_.id,
289  buffer->GetWidth()) < 0;
290  success |= drmModeAtomicAddProperty(property_set, id_, crtc_h_prop_.id,
291  buffer->GetHeight()) < 0;
292  success |=
293  drmModeAtomicAddProperty(property_set, id_, src_x_prop_.id, 0) < 0;
294  success |=
295  drmModeAtomicAddProperty(property_set, id_, src_y_prop_.id, 0) < 0;
296 
297  success |= drmModeAtomicAddProperty(property_set, id_, src_w_prop_.id,
298  buffer->GetWidth() << 16) < 0;
299  success |= drmModeAtomicAddProperty(property_set, id_, src_h_prop_.id,
300  buffer->GetHeight() << 16) < 0;
301  } else {
302  success |= drmModeAtomicAddProperty(property_set, id_, crtc_w_prop_.id,
303  layer->GetDisplayFrameWidth()) < 0;
304  success |= drmModeAtomicAddProperty(property_set, id_, crtc_h_prop_.id,
305  layer->GetDisplayFrameHeight()) < 0;
306  success |= drmModeAtomicAddProperty(
307  property_set, id_, src_x_prop_.id,
308  static_cast<int>(ceilf(source_crop.left)) << 16) < 0;
309  success |= drmModeAtomicAddProperty(
310  property_set, id_, src_y_prop_.id,
311  static_cast<int>(ceilf((source_crop.top))) << 16) < 0;
312  success |= drmModeAtomicAddProperty(property_set, id_, src_w_prop_.id,
313  layer->GetSourceCropWidth() << 16) < 0;
314  success |= drmModeAtomicAddProperty(property_set, id_, src_h_prop_.id,
315  layer->GetSourceCropHeight() << 16) < 0;
316  }
317 
318  if (rotation_prop_.id) {
319  uint32_t rotation = 0;
320  uint32_t transform = layer->GetPlaneTransform();
321  if (transform & kTransform90) {
322  rotation |= DRM_MODE_ROTATE_90;
323  if (transform & kReflectX)
324  rotation |= DRM_MODE_REFLECT_X;
325  if (transform & kReflectY)
326  rotation |= DRM_MODE_REFLECT_Y;
327  } else if (transform & kTransform180)
328  rotation |= DRM_MODE_ROTATE_180;
329  else if (transform & kTransform270)
330  rotation |= DRM_MODE_ROTATE_270;
331  else
332  rotation |= DRM_MODE_ROTATE_0;
333 
334  success = drmModeAtomicAddProperty(property_set, id_, rotation_prop_.id,
335  rotation) < 0;
336  }
337 
338  if (alpha_prop_.id) {
339  success =
340  drmModeAtomicAddProperty(property_set, id_, alpha_prop_.id, alpha) < 0;
341  }
342 
343  if (fence > 0 && in_fence_fd_prop_.id) {
344  success = drmModeAtomicAddProperty(property_set, id_, in_fence_fd_prop_.id,
345  fence) < 0;
346  }
347 
348  if (success) {
349  ETRACE("Could not update properties for plane with id: %d", id_);
350  return false;
351  }
352  IDISPLAYMANAGERTRACE("buffer->GetFb() ---------------------- ENDS%d",
353  buffer->GetFb());
354  return true;
355 }

420 bool DrmPlane::ValidateLayer(const OverlayLayer* layer) {
421  uint64_t alpha = 0xFF;
422 
423  if (layer->GetBlending() == HWCBlending::kBlendingPremult)
424  alpha = layer->GetAlpha();
425 
426  if (type_ == DRM_PLANE_TYPE_OVERLAY && (alpha != 0 && alpha != 0xFF) &&
427  alpha_prop_.id == 0) {
428  IDISPLAYMANAGERTRACE(
429  "Alpha property not supported, Cannot composite layer using Overlay.");
430  return false;
431  }
432 
433  bool zero_rotation = false;
434  uint32_t transform = layer->GetPlaneTransform();
435  if (transform == kIdentity) {
436  zero_rotation = true;
437  }
438 
439  if (!zero_rotation && rotation_prop_.id == 0) {
440  IDISPLAYMANAGERTRACE(
441  "Rotation property not supported, Cannot composite layer using "
442  "Overlay.");
443  return false;
444  }
445 
446  if (!IsSupportedFormat(layer->GetBuffer()->GetFormat())) {
447  IDISPLAYMANAGERTRACE(
448  "Layer cannot be supported as format is not supported.");
449  return false;
450  }
451 
452  return IsSupportedTransform(transform);
453 }

507 bool DrmPlane::IsSupportedModifier(uint64_t modifier, uint32_t format) {
508  uint32_t count = formats_modifiers_.size();
509  for (uint32_t i = 0; i < count; i++) {
510  const format_mods& obj = formats_modifiers_.at(i);
511  if (obj.format == format) {
512  std::vector<uint64_t>::const_iterator it;
513  it = std::find(obj.mods.begin(), obj.mods.end(), modifier);
514  if (it != obj.mods.end()) {
515  return true;
516  }
517  }
518  }
519  return false;
