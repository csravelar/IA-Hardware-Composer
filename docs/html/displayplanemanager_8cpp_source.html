49 bool DisplayPlaneManager::Initialize(uint32_t width, uint32_t height,
50  FrameBufferManager *frame_buffer_manager) {
51  fb_manager_ = frame_buffer_manager;
52  width_ = width;
53  height_ = height;
54  bool status = plane_handler_->PopulatePlanes(overlay_planes_);
55  if (!overlay_planes_.empty()) {
56  total_overlays_ = overlay_planes_.size();
57  if (total_overlays_ > 1) {
58  cursor_plane_ = overlay_planes_.back().get();
59  // If this is a universal plane, let's not restrict it to
60  // cursor usage only.

62  cursor_plane_ = NULL;
63  } else {
64  total_overlays_--;
65  }
66  }
67  }
68 
69  return status;

72 bool DisplayPlaneManager::ValidateLayers(
73  std::vector<OverlayLayer> &layers, int add_index, bool disable_overlay,
74  bool *commit_checked, bool *re_validation_needed,
75  DisplayPlaneStateList &composition,
76  DisplayPlaneStateList &previous_composition,
77  std::vector<NativeSurface *> &mark_later) {
78  CTRACE();
79 
80  if (add_index <= 0) {
81  if (!previous_composition.empty()) {
82  for (DisplayPlaneState &plane : previous_composition) {
83  MarkSurfacesForRecycling(&plane, mark_later, true);
84  }
85  }
86 
87  if (!composition.empty()) {
88  for (DisplayPlaneState &plane : previous_composition) {
89  MarkSurfacesForRecycling(&plane, mark_later, true);
90  }
91 
92  DisplayPlaneStateList().swap(composition);
93  }
94 
95 #ifdef SURFACE_TRACING
96  if (add_index <= 0) {
97  ISURFACETRACE("Full validation being performed. \n");
98  }
99 #endif
100  }
101 
102  std::vector<OverlayPlane> commit_planes;
103 
104  for (DisplayPlaneState &temp : composition) {
105  commit_planes.emplace_back(
106  OverlayPlane(temp.GetDisplayPlane(), temp.GetOverlayLayer()));
107  }
108 
109  // In case we are forcing GPU composition for all layers and using a single
110  // plane.
111  if (disable_overlay) {
112 #ifdef SURFACE_TRACING
113  ISURFACETRACE("Forcing GPU For all layers %d %d %d %d \n", disable_overlay,
114  composition.empty(), add_index <= 0, layers.size());
115 #endif
116  ForceGpuForAllLayers(commit_planes, composition, layers, mark_later, false);
117 
118  *re_validation_needed = false;
119  *commit_checked = true;
120  return true;
121  }
122 
123  auto overlay_begin = overlay_planes_.begin();
124  if (!composition.empty()) {
125  overlay_begin = overlay_planes_.begin() + composition.size();
126  }
127 
128  // Let's mark all planes as free to be used.
129  for (auto j = overlay_begin; j != overlay_planes_.end(); ++j) {
130  j->get()->SetInUse(false);
131  }
132 
133  std::vector<OverlayLayer *> cursor_layers;
134  auto layer_begin = layers.begin();
135  auto layer_end = layers.end();
136  bool validate_final_layers = false;
137  bool test_commit_done = false;
138  OverlayLayer *previous_layer = NULL;
139 
140  if (add_index > 0) {
141  layer_begin = layers.begin() + add_index;
142  }
143 
144  if (layer_begin != layer_end) {
145  auto overlay_end = overlay_planes_.end();
146 #ifdef DISABLE_CURSOR_PLANE
147  if (!enable_last_plane_ || cursor_plane_) {
148  overlay_end = overlay_planes_.end() - 1;
149  }
150 #else
151  if (cursor_plane_) {
152  overlay_end = overlay_planes_.end() - 1;
153  }
154 #endif
155 
156  // Handle layers for overlays.
157  for (auto j = overlay_begin; j < overlay_end; ++j) {
158  DisplayPlane *plane = j->get();
159  if (previous_layer && !composition.empty()) {
160  DisplayPlaneState &last_plane = composition.back();
161  if (last_plane.NeedsOffScreenComposition()) {
162  ValidateForDisplayScaling(composition.back(), commit_planes);
163  }
164  }
165 
166  // Let's break in case we have already mapped all our
167  // layers.
168  if (layer_begin == layer_end)
169  break;
170 
171  // Handle remaining overlay planes.
172  for (auto i = layer_begin; i != layer_end; ++i) {
173  OverlayLayer *layer = &(*(i));
174  ++layer_begin;
175  // Ignore cursor layer as it will handled separately.
176  if (layer->IsCursorLayer()) {
177  cursor_layers.emplace_back(layer);
178  continue;
179  }
180 
181  bool prefer_seperate_plane = layer->PreferSeparatePlane();
182  if (!prefer_seperate_plane && previous_layer) {
183  prefer_seperate_plane = previous_layer->PreferSeparatePlane();
184  }
185 
186  // Previous layer should not be used anywhere below, so can be
187  // safely reset to current layer.
188  previous_layer = layer;
189 
190  commit_planes.emplace_back(OverlayPlane(plane, layer));
191  // If we are able to composite buffer with the given plane, lets use
192  // it.
193  bool fall_back = FallbacktoGPU(plane, layer, commit_planes);
194  test_commit_done = true;
195  bool force_separate = false;
196  if (fall_back && !prefer_seperate_plane && !composition.empty()) {
197  force_separate =
198  ForceSeparatePlane(layers, composition.back(), layer);
199  }
200 
201  if (!fall_back || prefer_seperate_plane || force_separate) {
202  if (validate_final_layers)
203  validate_final_layers = fall_back;
204 
205  composition.emplace_back(plane, layer, this, layer->GetZorder(),
206  display_transform_);
207 #ifdef SURFACE_TRACING
208  ISURFACETRACE(
209  "Added Layer for direct Scanout: layer index: %d "
210  "validate_final_layers: %d force_separate: %d fall_back: %d \n",
211  layer->GetZorder(), validate_final_layers, force_separate,
212  fall_back);
213 #endif
214  plane->SetInUse(true);
215  DisplayPlaneState &last_plane = composition.back();
216  if (layer->IsVideoLayer()) {
217  last_plane.SetVideoPlane(true);
218  }
219  if (fall_back) {
220  if (!validate_final_layers)
221  validate_final_layers = !(last_plane.GetOffScreenTarget());
222 
223  ResetPlaneTarget(last_plane, commit_planes.back());
224  }
225 
226  break;
227  } else {
228  if (composition.empty()) {
229  composition.emplace_back(plane, layer, this, layer->GetZorder(),
230  display_transform_);
231 #ifdef SURFACE_TRACING
232  ISURFACETRACE("Added Layer: %d %d validate_final_layers: %d \n",
233  layer->GetZorder(), composition.size(),
234  validate_final_layers);
235 #endif
236  DisplayPlaneState &last_plane = composition.back();
237  ResetPlaneTarget(last_plane, commit_planes.back());
238  validate_final_layers = true;
239  if (display_transform_ != kIdentity) {
240  // If DisplayTransform is not supported, let's check if
241  // we can fallback to GPU rotation for this plane.
242  if (last_plane.GetRotationType() ==
243  DisplayPlaneState::RotationType::kDisplayRotation) {
244  last_plane.SetRotationType(
245  DisplayPlaneState::RotationType::kGPURotation, false);
246 
247  // Check if we can rotate using Display plane.
248  if (FallbacktoGPU(last_plane.GetDisplayPlane(),
249  last_plane.GetOffScreenTarget()->GetLayer(),
250  commit_planes)) {
251  last_plane.SetRotationType(
252  DisplayPlaneState::RotationType::kGPURotation, true);
253  } else {
254  validate_final_layers = false;
255  }
256  }
257  }
258 
259  break;
260  } else {
261  commit_planes.pop_back();
262 #ifdef SURFACE_TRACING
263  ISURFACETRACE("Added Layer: %d %d validate_final_layers: %d \n",
264  layer->GetZorder(), composition.size(),
265  validate_final_layers);
266 #endif
267  composition.back().AddLayer(layer);
268  while (SquashPlanesAsNeeded(layers, composition, commit_planes,
269  mark_later, &validate_final_layers)) {
270  j--;
271  plane = j->get();
272  }
273 
274  DisplayPlaneState &last_plane = composition.back();
275  if (!validate_final_layers)
276  validate_final_layers = !(last_plane.GetOffScreenTarget());
277  ResetPlaneTarget(last_plane, commit_planes.back());
278  }
279  }
280  }
281  }
282 
283  if ((layer_begin != layer_end) && (!composition.empty())) {
284  bool is_video = composition.back().IsVideoPlane();
285  previous_layer = NULL;
286  DisplayPlaneState &last_plane = composition.back();
287  // We dont have any additional planes. Pre composite remaining layers
288  // to the last overlay plane.
289  for (auto i = layer_begin; i != layer_end; ++i) {
290  previous_layer = &(*(i));
291  // Ignore cursor layer as it will handled separately.
292  if (previous_layer->IsCursorLayer()) {
293  cursor_layers.emplace_back(previous_layer);
294  previous_layer = NULL;
295  continue;
296  }
297 #ifdef SURFACE_TRACING
298  ISURFACETRACE("Added Layer: %d \n", previous_layer->GetZorder());
299 #endif
300  last_plane.AddLayer(previous_layer);
301  }
302 
303  if (composition.back().NeedsOffScreenComposition()) {
304  while (SquashPlanesAsNeeded(layers, composition, commit_planes,
305  mark_later, &validate_final_layers)) {
306  continue;
307  }
308  DisplayPlaneState &squashed_plane = composition.back();
309  // In this case we need to fallback to 3Dcomposition till Media
310  // backend adds support for multiple layers.
311  bool force_buffer = false;
312  if (is_video && squashed_plane.GetSourceLayers().size() > 1 &&
313  squashed_plane.GetOffScreenTarget()) {
314  MarkSurfacesForRecycling(&squashed_plane, mark_later, true);
315  force_buffer = true;
316  }
317 
318  if (force_buffer || squashed_plane.NeedsSurfaceAllocation()) {
319  ResetPlaneTarget(squashed_plane, commit_planes.back());
320  validate_final_layers = true;
321  }
322 
323  if (previous_layer) {
324  squashed_plane.UsePlaneScalar(false);
325  }
326 
327  commit_planes.back().layer = squashed_plane.GetOverlayLayer();
328  }
329  }
330  }
331 
332  if (!cursor_layers.empty()) {
333  ValidateCursorLayer(layers, commit_planes, cursor_layers, mark_later,
334  composition, &validate_final_layers, &test_commit_done,
335  false);
336 
337  if (validate_final_layers && add_index > 0 &&
338  (composition.size() == (overlay_planes_.size() - 1))) {
339  // If commit failed here and we are doing incremental validation,
340  // something might be wrong with other layer+plane combinations.
341  // Let's ensure DisplayQueue, checks final combination again and
342  // request full validation if needed.
343  *commit_checked = false;
344  return true;
345  }
346  }
347 
348  if (composition.empty()) {
349  *re_validation_needed = false;
350  *commit_checked = true;
351  return true;
352  }
353 
354  if (validate_final_layers) {
355  ValidateFinalLayers(commit_planes, composition, layers, mark_later, false);
356  test_commit_done = true;
357  }
358 
359  bool render_layers = false;
360  FinalizeValidation(composition, commit_planes, &render_layers,
361  re_validation_needed);
362 
363  *commit_checked = test_commit_done;
364  return render_layers;
365 }

402 void DisplayPlaneManager::ValidateCursorLayer(
403  std::vector<OverlayLayer> &all_layers,
404  std::vector<OverlayPlane> &commit_planes,
405  std::vector<OverlayLayer *> &cursor_layers,
406  std::vector<NativeSurface *> &mark_later,
407  DisplayPlaneStateList &composition, bool *validate_final_layers,
408  bool *test_commit_done, bool recycle_resources) {
409  CTRACE();
410  if (cursor_layers.empty()) {
411  return;
412  }
413 
414  DisplayPlaneState *last_plane = GetLastUsedOverlay(composition);
415  bool is_video = false;
416  if (last_plane)
417  is_video = last_plane->IsVideoPlane();
418 
419  uint32_t total_size = cursor_layers.size();
420  uint32_t cursor_index = 0;
421  auto overlay_end = overlay_planes_.end();
422  auto overlay_begin = overlay_end - 1;
423  if (total_size > 1) {
424  overlay_begin = overlay_planes_.begin() + composition.size();
425  }
426 
427 #ifdef DISABLE_CURSOR_PLANE
428  if (!enable_last_plane_) {
429  overlay_end = overlay_planes_.end() - 1;
430  if (total_size == 1)
431  overlay_begin = overlay_planes_.begin() + composition.size();
432  }
433 #endif
434  for (auto j = overlay_begin; j < overlay_end; ++j) {
435  if (cursor_index == total_size)
436  break;
437 
438  DisplayPlane *plane = j->get();
439  if (plane->InUse()) {
440  ITRACE("Trying to use a plane for cursor which is already in use. \n");
441  last_plane = NULL;
442  break;
443  }
444 
445  OverlayLayer *cursor_layer = cursor_layers.at(cursor_index);
446  commit_planes.emplace_back(OverlayPlane(plane, cursor_layer));
447  bool fall_back = FallbacktoGPU(plane, cursor_layer, commit_planes);
448  *test_commit_done = true;
449 
450  // Lets ensure we fall back to GPU composition in case
451  // cursor layer cannot be scanned out directly.
452  if (fall_back && !is_video && last_plane) {
453  commit_planes.pop_back();
454  cursor_layer->SetLayerComposition(OverlayLayer::kGpu);
455 #ifdef SURFACE_TRACING
456  ISURFACETRACE("Added CursorLayer: %d \n", cursor_layer->GetZorder());
457 #endif
458  last_plane->AddLayer(cursor_layer);
459  while (SquashPlanesAsNeeded(all_layers, composition, commit_planes,
460  mark_later, validate_final_layers)) {
461  continue;
462  }
463 
464  last_plane = GetLastUsedOverlay(composition);
465  bool reset_overlay = false;
466  if (!last_plane->GetOffScreenTarget())
467  reset_overlay = true;
468 
469  PreparePlaneForCursor(last_plane, mark_later, validate_final_layers,
470  last_plane->IsVideoPlane(), recycle_resources);
471 
472  if (reset_overlay) {
473  // Layer for the plane should have changed, reset commit planes.
474  std::vector<OverlayPlane>().swap(commit_planes);
475  for (DisplayPlaneState &temp : composition) {
476  commit_planes.emplace_back(
477  OverlayPlane(temp.GetDisplayPlane(), temp.GetOverlayLayer()));
478  }
479  }
480 
481  last_plane->UsePlaneScalar(false);
482  } else {
483  composition.emplace_back(plane, cursor_layer, this,
484  cursor_layer->GetZorder(), display_transform_);
485 #ifdef SURFACE_TRACING
486  ISURFACETRACE("Added CursorLayer for direct scanout: %d \n",
487  cursor_layer->GetZorder());
488 #endif
489  plane->SetInUse(true);
490  if (fall_back) {
491  DisplayPlaneState &temp = composition.back();
492  SetOffScreenPlaneTarget(temp);
493  cursor_layer->SetLayerComposition(OverlayLayer::kGpu);
494  *validate_final_layers = true;
495  } else {
496  cursor_layer->SetLayerComposition(OverlayLayer::kDisplay);
497  *validate_final_layers = false;
498  }
499 
500  last_plane = GetLastUsedOverlay(composition);
501  is_video = last_plane->IsVideoPlane();
502  }
503 
504  cursor_index++;
505  }
506 
507  // We dont have any additional planes. Pre composite remaining cursor layers
508  // to the last overlay plane.
509  OverlayLayer *last_layer = NULL;
510  if (!last_plane && (cursor_index < total_size))
511  last_plane = GetLastUsedOverlay(composition);
512 
513  for (uint32_t i = cursor_index; i < total_size; i++) {
514  OverlayLayer *cursor_layer = cursor_layers.at(i);
515 #ifdef SURFACE_TRACING
516  ISURFACETRACE("Added CursorLayer: %d \n", cursor_layer->GetZorder());
517 #endif
518  last_plane->AddLayer(cursor_layer);
519  cursor_layer->SetLayerComposition(OverlayLayer::kGpu);
520  last_layer = cursor_layer;
521  while (SquashPlanesAsNeeded(all_layers, composition, commit_planes,
522  mark_later, validate_final_layers)) {
523  continue;
524  }
525 
526  last_plane = GetLastUsedOverlay(composition);
527  }
528 
529  if (last_layer) {
530  PreparePlaneForCursor(last_plane, mark_later, validate_final_layers,
531  last_plane->IsVideoPlane(), recycle_resources);
532  last_plane->UsePlaneScalar(false);
533  }
534 }

536 void DisplayPlaneManager::ValidateForDisplayTransform(
537  DisplayPlaneState &last_plane,
538  const std::vector<OverlayPlane> &commit_planes) {
539  if (display_transform_ != kIdentity) {
540  // No need for any check if we are relying on rotation during
541  // 3D Composition pass.
542  DisplayPlaneState::RotationType original_rotation =
543  last_plane.GetRotationType();
544  if (last_plane.RevalidationType() &
545  DisplayPlaneState::ReValidationType::kRotation) {
546  uint32_t validation_done = DisplayPlaneState::ReValidationType::kRotation;
547  last_plane.SetRotationType(
548  DisplayPlaneState::RotationType::kDisplayRotation, false);
549  // Ensure Rotation doesn't impact the results.
550  if (FallbacktoGPU(last_plane.GetDisplayPlane(),
551  last_plane.GetOffScreenTarget()->GetLayer(),
552  commit_planes)) {
553  last_plane.SetRotationType(
554  DisplayPlaneState::RotationType::kGPURotation, false);
555  }
556 
557  last_plane.RevalidationDone(validation_done);
558  }
559 
560  if (original_rotation != last_plane.GetRotationType()) {
561  last_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
562  }
563  }
564 }

566 void DisplayPlaneManager::ValidateForDownScaling(
567  DisplayPlaneState &last_plane,
568  const std::vector<OverlayPlane> &commit_planes) {
569 #ifdef ENABLE_DOWNSCALING
570  uint32_t original_downscaling_factor = last_plane.GetDownScalingFactor();
571  if (last_plane.RevalidationType() &
572  DisplayPlaneState::ReValidationType::kDownScaling) {
573  last_plane.SetDisplayDownScalingFactor(1, false);
574  if (!last_plane.IsUsingPlaneScalar() && last_plane.CanUseGPUDownScaling()) {
575  last_plane.SetDisplayDownScalingFactor(4, false);
576  if (!plane_handler_->TestCommit(commit_planes)) {
577  last_plane.SetDisplayDownScalingFactor(1, false);
578  }
579  }
580 
581  uint32_t validation_done =
582  DisplayPlaneState::ReValidationType::kDownScaling;
583  last_plane.RevalidationDone(validation_done);
584  }
585 
586  if (original_downscaling_factor != last_plane.GetDownScalingFactor()) {
587  last_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
588  }
589 #else
590  HWC_UNUSED(commit_planes);
591  HWC_UNUSED(last_plane);
592 #endif

595 void DisplayPlaneManager::ValidateForDisplayScaling(
596  DisplayPlaneState &last_plane, std::vector<OverlayPlane> &commit_planes) {
597  last_plane.ValidateReValidation();
598  if (!(last_plane.RevalidationType() &
599  DisplayPlaneState::ReValidationType::kUpScalar)) {
600  return;
601  }
602 
603  last_plane.RevalidationDone(DisplayPlaneState::ReValidationType::kUpScalar);
604 
605  bool old_state = last_plane.IsUsingPlaneScalar();
606  if (old_state) {
607  last_plane.UsePlaneScalar(false, false);
608  }
609 
610  if (!last_plane.CanUseDisplayUpScaling()) {
611  // If we used plane scalar, clear surfaces.
612  if (old_state) {
613  last_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
614  }
615 
616  return;
617  }
618 
619  // TODO: Scalars are limited in HW. Determine scaling ratio
620  // which would really benefit vs doing it in GPU side.
621 
622  // Display frame and Source rect are different, let's check if
623  // we can take advantage of scalars attached to this plane.
624  last_plane.UsePlaneScalar(true, false);
625 
626  OverlayPlane &last_overlay_plane = commit_planes.back();
627  last_overlay_plane.layer = last_plane.GetOverlayLayer();
628 
629  bool fall_back =
630  FallbacktoGPU(last_plane.GetDisplayPlane(),
631  last_plane.GetOffScreenTarget()->GetLayer(), commit_planes);
632  if (fall_back) {
633  last_plane.UsePlaneScalar(false, false);
634  }
635 
636  if (old_state != last_plane.IsUsingPlaneScalar()) {
637  last_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
638  }
639 }

817 void DisplayPlaneManager::ForceGpuForAllLayers(
818  std::vector<OverlayPlane> &commit_planes,
819  DisplayPlaneStateList &composition, std::vector<OverlayLayer> &layers,
820  std::vector<NativeSurface *> &mark_later, bool recycle_resources) {
821  // Let's mark all planes as free to be used.
822  for (auto j = overlay_planes_.begin(); j != overlay_planes_.end(); ++j) {
823  j->get()->SetInUse(false);
824  }
825 
826  if (!composition.empty()) {
827  for (DisplayPlaneState &plane : composition) {
828  MarkSurfacesForRecycling(&plane, mark_later, recycle_resources);
829  }
830  }
831 
832  auto layer_begin = layers.begin();
833  auto layer_end = layers.end();
834  DisplayPlaneStateList().swap(composition);
835  std::vector<OverlayPlane>().swap(commit_planes);
836  OverlayLayer *primary_layer = &(*(layers.begin()));
837  DisplayPlane *current_plane = overlay_planes_.at(0).get();
838 
839  composition.emplace_back(current_plane, primary_layer, this,
840  primary_layer->GetZorder(), display_transform_);
841  DisplayPlaneState &last_plane = composition.back();
842  last_plane.ForceGPURendering();
843  layer_begin++;
844 #ifdef SURFACE_TRACING
845  ISURFACETRACE("Added layer in ForceGpuForAllLayers: %d \n",
846  primary_layer->GetZorder());
847 #endif
848 
849  for (auto i = layer_begin; i != layer_end; ++i) {
850 #ifdef SURFACE_TRACING
851  ISURFACETRACE("Added layer in ForceGpuForAllLayers: %d \n", i->GetZorder());
852 #endif
853  last_plane.AddLayer(&(*(i)));
854  i->SetLayerComposition(OverlayLayer::kGpu);
855  }
856 
857  EnsureOffScreenTarget(last_plane);
858  current_plane->SetInUse(true);
859  commit_planes.emplace_back(
860  OverlayPlane(last_plane.GetDisplayPlane(), last_plane.GetOverlayLayer()));
861  // Check for Any display transform to be applied.
862  ValidateForDisplayTransform(last_plane, commit_planes);
863  // Check for any change to scalar usage.
864  ValidateForDisplayScaling(last_plane, commit_planes);
865  // Check for Downscaling.
866  ValidateForDownScaling(last_plane, commit_planes);
867  // Reset andy Scanout validation state.
868  uint32_t validation_done = DisplayPlaneState::ReValidationType::kScanout;
869  last_plane.RevalidationDone(validation_done);
870 }

897  NativeSurface *surface = surfaces.at(i);
898  surface->SetSurfaceAge(-1);
899  }

907 bool DisplayPlaneManager::ReValidatePlanes(
908  DisplayPlaneStateList &composition, std::vector<OverlayLayer> &layers,
909  std::vector<NativeSurface *> &mark_later, bool *request_full_validation,
910  bool needs_revalidation_checks, bool re_validate_commit) {
911 #ifdef SURFACE_TRACING
912  ISURFACETRACE(
913  "ReValidatePlanes called needs_revalidation_checks %d re_validate_commit "
914  "%d \n",
915  needs_revalidation_checks, re_validate_commit);
916 #endif
917  // Let's first check the current combination works.
918  *request_full_validation = false;
919  bool render = false;
920  bool reset_composition_region = false;
921  std::vector<OverlayPlane> commit_planes;
922  for (DisplayPlaneState &temp : composition) {
923  commit_planes.emplace_back(
924  OverlayPlane(temp.GetDisplayPlane(), temp.GetOverlayLayer()));
925 
926  if (temp.Scanout()) {
927  continue;
928  }
929 
930  render = true;
931  }
932 
933  if (re_validate_commit) {
934  // If this combination fails just fall back to full validation.
935  if (!plane_handler_->TestCommit(commit_planes)) {
936 #ifdef SURFACE_TRACING
937  ISURFACETRACE(
938  "ReValidatePlanes Test commit failed. Forcing full validation. \n");
939 #endif
940  *request_full_validation = true;
941  return render;
942  }
943  }
944 
945  if (!needs_revalidation_checks) {
946  return render;
947  }
948 
949  uint32_t index = 0;
950 
951  for (DisplayPlaneState &last_plane : composition) {
952  if (!last_plane.NeedsOffScreenComposition()) {
953  index++;
954  reset_composition_region = false;
955  continue;
956  }
957 
958  if (reset_composition_region) {
959  last_plane.RefreshSurfaces(NativeSurface::kFullClear, true);
960  }
961 
962  reset_composition_region = false;
963  uint32_t revalidation_type = last_plane.RevalidationType();
964 
965  if (!revalidation_type) {
966  render = true;
967  index++;
968  continue;
969  }
970 
971  uint32_t validation_done = DisplayPlaneState::ReValidationType::kScanout;
972  if (revalidation_type & DisplayPlaneState::ReValidationType::kScanout) {
973  const std::vector<size_t> &source_layers = last_plane.GetSourceLayers();
974  bool uses_scalar = last_plane.IsUsingPlaneScalar();
975  // Store current layer to re-set in case commit fails.
976  const OverlayLayer *current_layer = last_plane.GetOverlayLayer();
977  OverlayLayer *layer = &(layers.at(source_layers.at(0)));
978  last_plane.SetOverlayLayer(layer);
979  // Disable GPU Rendering.
980  last_plane.DisableGPURendering();
981  if (uses_scalar)
982  last_plane.UsePlaneScalar(false, false);
983 
984  layer->SetLayerComposition(OverlayLayer::kDisplay);
985 
986  commit_planes.at(index).layer = last_plane.GetOverlayLayer();
987 
988  // If this combination fails just fall back to original state.
989  if (FallbacktoGPU(last_plane.GetDisplayPlane(), layer, commit_planes)) {
990  // Reset to old state.
991  last_plane.ForceGPURendering();
992  layer->SetLayerComposition(OverlayLayer::kGpu);
993  last_plane.SetOverlayLayer(current_layer);
994  commit_planes.at(index).layer = last_plane.GetOverlayLayer();
995  if (uses_scalar)
996  last_plane.UsePlaneScalar(true, false);
997  } else {
998 #ifdef SURFACE_TRACING
999  ISURFACETRACE("ReValidatePlanes called: moving to scan \n");
1000 #endif
1001  MarkSurfacesForRecycling(&last_plane, mark_later, true);
1002  last_plane.SetOverlayLayer(layer);
1003  reset_composition_region = true;
1004  }
1005  }
1006 
1007  render = true;
1008  index++;
1009  if (revalidation_type & DisplayPlaneState::ReValidationType::kUpScalar) {
1010  ValidateForDisplayScaling(last_plane, commit_planes);
1011  validation_done |= DisplayPlaneState::ReValidationType::kUpScalar;
1012  }
1013 
1014  if (revalidation_type & DisplayPlaneState::ReValidationType::kRotation) {
1015  validation_done |= DisplayPlaneState::ReValidationType::kRotation;
1016  // Save old rotation type.
1017  DisplayPlaneState::RotationType old_type = last_plane.GetRotationType();
1018  DisplayPlaneState::RotationType new_type = old_type;
1019  if (old_type == DisplayPlaneState::RotationType::kGPURotation) {
1020  last_plane.SetRotationType(
1021  DisplayPlaneState::RotationType::kDisplayRotation, false);
1022  } else if (re_validate_commit) {
1023  // We should have already done a full commit check above.
1024  // As their is no state change we can avoid another test
1025  // commit here.
1026  last_plane.RevalidationDone(validation_done);
1027  continue;
1028  }
1029 
1030  // Check if we can rotate using Display plane.
1031  if (FallbacktoGPU(last_plane.GetDisplayPlane(),
1032  last_plane.GetOffScreenTarget()->GetLayer(),
1033  commit_planes)) {
1034  new_type = DisplayPlaneState::RotationType::kGPURotation;
1035  }
1036 
1037  if (old_type != new_type) {
1038  // Set new rotation type. Clear surfaces in case type has changed.
1039  last_plane.SetRotationType(new_type, true);
1040  }
1041  }
1042 
1043  if (revalidation_type & DisplayPlaneState::ReValidationType::kDownScaling) {
1044  validation_done |= DisplayPlaneState::ReValidationType::kDownScaling;
1045  // Make sure we are not handling upscaling.
1046  if (last_plane.IsUsingPlaneScalar()) {
1047  ETRACE(
1048  "We are using upscaling and also trying to validate for "
1049  "downscaling \n");
1050  if (last_plane.GetDownScalingFactor() > 1)
1051  last_plane.SetDisplayDownScalingFactor(1, true);
1052  } else {
1053  // Check for Downscaling.
1054  ValidateForDownScaling(last_plane, commit_planes);
1055  }
1056  }
1057 
1058  last_plane.RevalidationDone(validation_done);
1059  }
1060 
1061  return render;
1062 }
