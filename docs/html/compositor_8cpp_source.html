60 bool Compositor::Draw(DisplayPlaneStateList &comp_planes,
61  std::vector<OverlayLayer> &layers,
62  const std::vector<HwcRect<int>> &display_frame) {
63  CTRACE();
64  const DisplayPlaneState *comp = NULL;
65  std::vector<size_t> dedicated_layers;
66  std::vector<DrawState> draw_state;
67  std::vector<DrawState> media_state;
68 
69  for (DisplayPlaneState &plane : comp_planes) {
70  if (plane.Scanout()) {
71  if (!plane.IsSurfaceRecycled()) {
72  dedicated_layers.insert(dedicated_layers.end(),
73  plane.GetSourceLayers().begin(),
74  plane.GetSourceLayers().end());
75  }
76  } else if (plane.IsVideoPlane()) {
77  dedicated_layers.insert(dedicated_layers.end(),
78  plane.GetSourceLayers().begin(),
79  plane.GetSourceLayers().end());
80  media_state.emplace_back();
81  plane.SwapSurfaceIfNeeded();
82  DrawState &state = media_state.back();
83  state.surface_ = plane.GetOffScreenTarget();
84  MediaState &media_state = state.media_state_;
85  lock_.lock();
86  media_state.colors_ = colors_;
87  media_state.scaling_mode_ = scaling_mode_;
88  media_state.deinterlace_ = deinterlace_;
89  lock_.unlock();
90  const OverlayLayer &layer = layers[plane.GetSourceLayers().at(0)];
91  media_state.layer_ = &layer;
92  } else if (plane.NeedsOffScreenComposition()) {
93  comp = &plane;
94  plane.SwapSurfaceIfNeeded();
95  std::vector<CompositionRegion> &comp_regions =
96  plane.GetCompositionRegion();
97  bool regions_empty = comp_regions.empty();
98  NativeSurface *surface = plane.GetOffScreenTarget();
99  if (surface == NULL) {
100  ETRACE("GetOffScreenTarget() returned NULL pointer 'surface'.");
101  return false;
102  }
103  if (!regions_empty &&
104  (surface->ClearSurface() || surface->IsPartialClear() ||
105  surface->IsSurfaceDamageChanged())) {
106  plane.ResetCompositionRegion();
107  regions_empty = true;
108  }
109 
110  if (surface->ClearSurface()) {
111  plane.UpdateDamage(plane.GetDisplayFrame());
112  }
113 
114  if (regions_empty) {
115  SeparateLayers(dedicated_layers, comp->GetSourceLayers(), display_frame,
116  surface->GetSurfaceDamage(), comp_regions);
117  }
118 
119  std::vector<size_t>().swap(dedicated_layers);
120  if (comp_regions.empty())
121  continue;
122 
123  draw_state.emplace_back();
124  DrawState &state = draw_state.back();
125  state.surface_ = surface;
126  size_t num_regions = comp_regions.size();
127  state.states_.reserve(num_regions);
128  bool use_plane_transform = false;
129  if (plane.GetRotationType() ==
130  DisplayPlaneState::RotationType::kGPURotation) {
131  use_plane_transform = true;
132  }
133 
134  CalculateRenderState(layers, comp_regions, state,
135  plane.GetDownScalingFactor(),
136  plane.IsUsingPlaneScalar(), use_plane_transform);
137 
138  if (state.states_.empty()) {
139  draw_state.pop_back();
140  }
141  }
142  }
143 
144  bool status = true;
145  if (!draw_state.empty() || !media_state.empty())
146  status = thread_->Draw(draw_state, media_state, layers);
147 
148  return status;
149 }

151 bool Compositor::DrawOffscreen(std::vector<OverlayLayer> &layers,
152  const std::vector<HwcRect<int>> &display_frame,
153  const std::vector<size_t> &source_layers,
154  ResourceManager *resource_manager,
155  uint32_t width, uint32_t height,
156  HWCNativeHandle output_handle,
157  int32_t acquire_fence, int32_t *retire_fence) {
158  std::vector<CompositionRegion> comp_regions;
159  SeparateLayers(std::vector<size_t>(), source_layers, display_frame,
160  HwcRect<int>(0, 0, width, height), comp_regions);
161  if (comp_regions.empty()) {
162  ETRACE(
163  "Failed to prepare offscreen buffer. "
164  "error: %s",
165  PRINTERROR());
166  return false;
167  }
168 
169  NativeSurface *surface = Create3DBuffer(width, height);
170  surface->InitializeForOffScreenRendering(output_handle, resource_manager);
171  std::vector<DrawState> draw;
172  std::vector<DrawState> media;
173  draw.emplace_back();
174  DrawState &draw_state = draw.back();
175  draw_state.destroy_surface_ = true;
176  draw_state.surface_ = surface;
177  size_t num_regions = comp_regions.size();
178  draw_state.states_.reserve(num_regions);
179  CalculateRenderState(layers, comp_regions, draw_state, 1, false);
180 
181  if (draw_state.states_.empty()) {
182  return true;
183  }
184 
185  if (acquire_fence > 0) {
186  draw_state.acquire_fences_.emplace_back(acquire_fence);
187  }
188 
189  bool status = thread_->Draw(draw, media, layers);
190  if (status) {
191  *retire_fence = draw_state.retire_fence_;
192  } else {
193  *retire_fence = -1;
194  }
195 
196  return status;
197 }

203 void Compositor::CalculateRenderState(
204  std::vector<OverlayLayer> &layers,
205  const std::vector<CompositionRegion> &comp_regions, DrawState &draw_state,
206  uint32_t downscaling_factor, bool uses_display_up_scaling,
207  bool use_plane_transform) {
208  CTRACE();
209  size_t num_regions = comp_regions.size();
210  for (size_t region_index = 0; region_index < num_regions; region_index++) {
211  const CompositionRegion &region = comp_regions.at(region_index);
212  RenderState state;
213  state.ConstructState(layers, region, downscaling_factor,
214  uses_display_up_scaling, use_plane_transform);
215  if (state.layer_state_.empty()) {
216  continue;
217  }
218 
219  draw_state.states_.emplace(draw_state.states_.begin(), state);
220  const std::vector<size_t> &source = region.source_layers;
221  for (size_t texture_index : source) {
222  OverlayLayer &layer = layers.at(texture_index);
223  int32_t fence = layer.ReleaseAcquireFence();
224  if (fence > 0) {
225  draw_state.acquire_fences_.emplace_back(fence);
226  }
227  }
228  }
229 }

270 // Below code is taken from drm_hwcomposer adopted to our needs.
271 static std::vector<size_t> SetBitsToVector(
272  uint64_t in, const std::vector<size_t> &index_map) {
273  std::vector<size_t> out;
274  size_t msb = sizeof(in) * 8 - 1;
275  uint64_t mask = (uint64_t)1 << msb;
276  for (size_t i = msb; mask != (uint64_t)0; i--, mask >>= 1)
277  if (in & mask)
278  out.emplace_back(index_map[i]);
279  return out;
280 }

282 void Compositor::SeparateLayers(const std::vector<size_t> &dedicated_layers,
283  const std::vector<size_t> &source_layers,
284  const std::vector<HwcRect<int>> &display_frame,
285  const HwcRect<int> &damage_region,
286  std::vector<CompositionRegion> &comp_regions) {
287  CTRACE();
288  if (source_layers.size() > 64) {
289  ETRACE("Failed to separate layers because there are more than 64");
290  return;
291  }
292 
293  size_t num_exclude_rects = 0;
294  // Index at which the actual layers begin
295  size_t layer_offset = dedicated_layers.size();
296  if (source_layers.size() + layer_offset > 64) {
297  WTRACE(
298  "Exclusion rectangles are being truncated to make the rectangle count "
299  "fit into 64");
300  num_exclude_rects = 64 - source_layers.size() - dedicated_layers.size();
301  }
302 
303  // We inject all the exclude rects into the rects list. Any resulting rect
304  // that includes ANY of the first num_exclude_rects is rejected. After the
305  // exclude rects, we add the lower layers. The rects that intersect with
306  // these layers will be inspected and only those which are to be composited
307  // above the layer will be included in the composition regions.
308  std::vector<HwcRect<int>> layer_rects(source_layers.size() + layer_offset);
309  std::transform(
310  dedicated_layers.begin(), dedicated_layers.end(),
311  layer_rects.begin() + num_exclude_rects,
312  [=](size_t layer_index) { return display_frame[layer_index]; });
313  std::transform(source_layers.begin(), source_layers.end(),
314  layer_rects.begin() + layer_offset, [=](size_t layer_index) {
315  return display_frame[layer_index];
316  });
317 
318  std::vector<RectSet<int>> separate_regions;
319  get_draw_regions(layer_rects, damage_region, &separate_regions);
320  uint64_t exclude_mask = ((uint64_t)1 << num_exclude_rects) - 1;
321  uint64_t dedicated_mask = (((uint64_t)1 << dedicated_layers.size()) - 1)
322  << num_exclude_rects;
323 
324  for (RectSet<int> &region : separate_regions) {
325  if (region.id_set.getBits() & exclude_mask)
326  continue;
327 
328  // If a rect intersects one of the dedicated layers, we need to remove the
329  // layers from the composition region which appear *below* the dedicated
330  // layer. This effectively punches a hole through the composition layer such
331  // that the dedicated layer can be placed below the composition and not
332  // be occluded.
333  uint64_t dedicated_intersect = region.id_set.getBits() & dedicated_mask;
334  for (size_t i = 0; dedicated_intersect && i < dedicated_layers.size();
335  ++i) {
336  // Only exclude layers if they intersect this particular dedicated layer
337  if (!(dedicated_intersect & (1 << (i + num_exclude_rects))))
338  continue;
339 
340  for (size_t j = 0; j < source_layers.size(); ++j) {
341  if (source_layers[j] < dedicated_layers[i])
342  region.id_set.subtract(j + layer_offset);
343  }
344  }
345 
346  if (!(region.id_set.getBits() >> layer_offset))
347  continue;
348 
349  comp_regions.emplace_back(CompositionRegion{
350  region.rect, SetBitsToVector(region.id_set.getBits() >> layer_offset,
351  source_layers)});
352  }
353 }
