26 // Generic ring buffer that always allocates contiguous regions. Chunks can be
27 // freed in any order, but the in flight pointer (read_offset_) only advances
28 // when the oldest in flight chunk is free. It's optimized. for the case where
29 // the oldest chunks are the first to be freed, but this is not strictly
30 // required. Returned Allocation objects automatically free themselves when they
31 // go out of scope.
32 class RingBuffer {
33  public:
34  class Allocation {
35  public:
36  Allocation() = default;
37  Allocation(RingBuffer *parent, uint8_t *ptr) {
38  parent_ = parent;
39  ptr_ = ptr;
40  }
41  Allocation(Allocation &&rhs) noexcept {
42  parent_ = rhs.parent_;
43  ptr_ = rhs.ptr_;
44  rhs.parent_ = nullptr;
45  rhs.ptr_ = nullptr;
46  }
47 
48  ~Allocation() {
49  CheckedFree();
50  }
51 
52  Allocation &operator=(Allocation &&rhs) noexcept {
53  if (&rhs == this)
54  return *this;
55  CheckedFree();
56 
57  parent_ = rhs.parent_;
58  ptr_ = rhs.ptr_;
59  rhs.parent_ = nullptr;
60  rhs.ptr_ = nullptr;
61 
62  return *this;
63  }
64 
65  explicit operator bool() const {
66  return ptr_ != nullptr && parent_ != nullptr;
67  }
68 
69  uint8_t *ptr() {
70  return ptr_;
71  }
72 
73  template <typename T>
74  T *get() {
75  return (T *)ptr_;
76  }
77 
78  size_t offset() const {
79  return ptr_ - parent_->buffer_;
80  }
81 
82  private:
83  RingBuffer *parent_ = nullptr;
84  uint8_t *ptr_ = nullptr;
85 
86  void CheckedFree() {
87  if (parent_ && ptr_)
88  parent_->Free(ptr_);
89  }
90  };
91 
92  RingBuffer() = default;
93  RingBuffer(uint8_t *buffer, size_t buffer_size)
94  : buffer_(buffer), buffer_size_(buffer_size) {
95  }
96 
97  explicit operator bool() const {
98  return buffer_ != nullptr;
99  }
100 
101  uint8_t *get() {
102  return buffer_;
103  }
104 
105  // If succesful, the returned allocation's offset within the ring buffer will
106  // be aligned, but the allocation's pointer may not be. alignment must be a
107  // power of two.
108  Allocation Allocate(size_t size, size_t alignment);
109 
110  private:

120  }
121  };

126 };
