86 void DisplayPlaneState::AddLayer(const OverlayLayer *layer) {
87  const HwcRect<int> &display_frame = layer->GetDisplayFrame();
88  HwcRect<int> target_display_frame = private_data_->display_frame_;
89  CalculateRect(display_frame, target_display_frame);
90  HwcRect<float> target_source_crop = private_data_->source_crop_;
91  CalculateSourceRect(layer->GetSourceCrop(), target_source_crop);
92  private_data_->source_layers_.emplace_back(layer->GetZorder());
93 
94  private_data_->state_ = DisplayPlanePrivateState::State::kRender;
95 
96  // If layers are less than 2, we need to enforce rect checks as
97  // we shouldn't have done them yet (i.e. Previous state could have
98  // been direct scanout.)
99  bool rect_updated = true;
100  for (NativeSurface *surface : private_data_->surfaces_) {
101  // Damage whole old rect.
102  surface->UpdateSurfaceDamage(private_data_->display_frame_, true);
103  }
104 
105  if (private_data_->source_layers_.size() > 2 &&
106  (private_data_->display_frame_ == target_display_frame) &&
107  ((private_data_->source_crop_ == target_source_crop))) {
108  rect_updated = false;
109  } else {
110  private_data_->display_frame_ = target_display_frame;
111  private_data_->source_crop_ = target_source_crop;
112  for (NativeSurface *surface : private_data_->surfaces_) {
113  surface->UpdateSurfaceDamage(private_data_->display_frame_, true);
114  }
115  }
116 
117  if (!private_data_->rect_updated_)
118  private_data_->rect_updated_ = rect_updated;
119 
120  if (!private_data_->has_cursor_layer_)
121  private_data_->has_cursor_layer_ = layer->IsCursorLayer();
122 
123  // TODO: Add checks for Video type once our
124  // Media backend can support compositing more
125  // than one layer together.
126  private_data_->type_ = DisplayPlanePrivateState::PlaneType::kNormal;
127  private_data_->apply_effects_ = false;
128 
129  // Reset Validation state.
130  if (re_validate_layer_ & ReValidationType::kScanout)
131  re_validate_layer_ &= ~ReValidationType::kScanout;
132 
133  private_data_->refresh_surface_ = true;
134  RefreshSurfaces(NativeSurface::kPartialClear);
135 
136  recycled_surface_ = false;
137 }

139 void DisplayPlaneState::ResetLayers(const std::vector<OverlayLayer> &layers,
140  size_t remove_index, bool *rects_updated) {
141  std::vector<size_t> current_layers = private_data_->source_layers_;
142  std::vector<size_t>().swap(private_data_->source_layers_);
143  std::vector<size_t> &new_layers = private_data_->source_layers_;
144 
145  private_data_->has_cursor_layer_ = false;
146  HwcRect<int> target_display_frame;
147  HwcRect<float> target_source_crop;
148  bool has_video = false;
149  bool layer_removed = false;
150  for (const size_t &index : current_layers) {
151  if (index >= remove_index) {
152 #ifdef SURFACE_TRACING
153  ISURFACETRACE("Reset breaks index: %d remove_index %d \n", index,
154  remove_index);
155 #endif
156  layer_removed = true;
157  break;
158  }
159 
160  const OverlayLayer &layer = layers.at(index);
161  bool is_cursor = layer.IsCursorLayer();
162 
163  if (is_cursor) {
164  private_data_->has_cursor_layer_ = true;
165  } else if (!has_video) {
166  has_video = layer.IsVideoLayer();
167  }
168 
169  const HwcRect<int> &df = layer.GetDisplayFrame();
170  const HwcRect<float> &source_crop = layer.GetSourceCrop();
171  CalculateRect(df, target_display_frame);
172  CalculateSourceRect(source_crop, target_source_crop);
173 #ifdef SURFACE_TRACING
174  ISURFACETRACE("Reset adds index: %d \n", layer.GetZorder());
175 #endif
176  new_layers.emplace_back(layer.GetZorder());
177  }
178 
179 #ifdef SURFACE_TRACING
180  ISURFACETRACE(
181  "Reset called has_video: %d Source Layers Size: %d Previous Source "
182  "Layers Size: %d Has Cursor: %d Total Layers Size: %d \n",
183  has_video, private_data_->source_layers_.size(), current_layers.size(),
184  private_data_->has_cursor_layer_, layers.size());
185 #endif
186 
187  if (private_data_->source_layers_.empty()) {
188  return;
189  }
190 
191  for (NativeSurface *surface : private_data_->surfaces_) {
192  // Damage whole old rect.
193  surface->UpdateSurfaceDamage(private_data_->display_frame_, true);
194  }
195 
196  bool rect_updated = true;
197  if ((private_data_->display_frame_ == target_display_frame) &&
198  ((private_data_->source_crop_ == target_source_crop))) {
199  rect_updated = false;
200  } else {
201  private_data_->display_frame_ = target_display_frame;
202  private_data_->source_crop_ = target_source_crop;
203  for (NativeSurface *surface : private_data_->surfaces_) {
204  surface->UpdateSurfaceDamage(private_data_->display_frame_, true);
205  }
206  }
207 
208  if (!private_data_->rect_updated_)
209  private_data_->rect_updated_ = rect_updated;
210 
211  *rects_updated = rect_updated;
212 
213  if (private_data_->source_layers_.size() == 1) {
214  if (private_data_->has_cursor_layer_) {
215  private_data_->type_ = DisplayPlanePrivateState::PlaneType::kCursor;
216  } else if (has_video) {
217  private_data_->type_ = DisplayPlanePrivateState::PlaneType::kVideo;
218  } else {
219  private_data_->type_ = DisplayPlanePrivateState::PlaneType::kNormal;
220  }
221 
222  if (!has_video) {
223  re_validate_layer_ |= ReValidationType::kScanout;
224  } else {
225  // Reset Validation state.
226  re_validate_layer_ &= ~ReValidationType::kScanout;
227  }
228  } else {
229  private_data_->type_ = DisplayPlanePrivateState::PlaneType::kNormal;
230  // Reset Validation state.
231  re_validate_layer_ &= ~ReValidationType::kScanout;
232  }
233 
234  private_data_->refresh_surface_ = true;
235  recycled_surface_ = false;
236  RefreshSurfaces(NativeSurface::kPartialClear);
237 }

239 void DisplayPlaneState::RefreshLayerRects(
240  const std::vector<OverlayLayer> &layers) {
241  const std::vector<size_t> &current_layers = private_data_->source_layers_;
242  HwcRect<int> target_display_frame;
243  HwcRect<float> target_source_crop;
244  HwcRect<int> surface_damage = HwcRect<int>(0, 0, 0, 0);
245  bool only_cursor_layer = true;
246  for (const size_t &index : current_layers) {
247  const OverlayLayer &layer = layers.at(index);
248  const HwcRect<int> &df = layer.GetDisplayFrame();
249  const HwcRect<float> &source_crop = layer.GetSourceCrop();
250  CalculateRect(df, target_display_frame);
251  CalculateSourceRect(source_crop, target_source_crop);
252  if (!layer.IsCursorLayer() && (layer.HasDimensionsChanged())) {
253  only_cursor_layer = false;
254  }
255 
256  if (layer.HasLayerContentChanged()) {
257  CalculateRect(layer.GetSurfaceDamage(), surface_damage);
258  }
259  }
260 
261  if (!only_cursor_layer) {
262  CalculateRect(private_data_->display_frame_, surface_damage);
263  }
264 
265  bool rect_updated = true;
266  if ((private_data_->display_frame_ == target_display_frame) &&
267  ((private_data_->source_crop_ == target_source_crop))) {
268  rect_updated = false;
269  } else {
270  private_data_->display_frame_ = target_display_frame;
271  private_data_->source_crop_ = target_source_crop;
272  if (!only_cursor_layer) {
273  CalculateRect(private_data_->display_frame_, surface_damage);
274  }
275  }
276 
277  if (!private_data_->rect_updated_)
278  private_data_->rect_updated_ = rect_updated;
279 
280  private_data_->refresh_surface_ = true;
281  recycled_surface_ = false;
282  if (!surface_damage.empty()) {
283  for (NativeSurface *surface : private_data_->surfaces_) {
284  surface->UpdateSurfaceDamage(surface_damage, true);
285  }
286 
287  RefreshSurfaces(NativeSurface::kPartialClear);
288  }
289 }

333 void DisplayPlaneState::SwapSurfaceIfNeeded() {
334  if (surface_swapped_) {
335  return;
336  }
337 
338  size_t size = private_data_->surfaces_.size();
339  if (size == 0)
340  return;
341 
342  if (size == 3) {
343  std::vector<NativeSurface *> temp;
344  temp.reserve(size);
345  // Lets make sure front buffer is now back in the list.
346  temp.emplace_back(private_data_->surfaces_.at(2));
347  temp.emplace_back(private_data_->surfaces_.at(0));
348  temp.emplace_back(private_data_->surfaces_.at(1));
349  private_data_->surfaces_.swap(temp);
350  }
351 
352  surface_swapped_ = true;
353  recycled_surface_ = false;
354  NativeSurface *surface = private_data_->surfaces_.at(0);
355  private_data_->layer_ = surface->GetLayer();
356 }

358 void DisplayPlaneState::HandleCommitFailure() {
359  size_t size = private_data_->surfaces_.size();
360  if (size == 0)
361  return;
362 
363  if (surface_swapped_) {
364  if (size == 3) {
365  std::vector<NativeSurface *> temp;
366  temp.reserve(size);
367  // Lets make sure we restore the buffer queue.
368  temp.emplace_back(private_data_->surfaces_.at(1));
369  temp.emplace_back(private_data_->surfaces_.at(2));
370  temp.emplace_back(private_data_->surfaces_.at(0));
371  private_data_->surfaces_.swap(temp);
372  }
373 
374  NativeSurface *surface = private_data_->surfaces_.at(0);
375  private_data_->layer_ = surface->GetLayer();
376  }
377 
378  for (uint32_t i = 0; i < size; i++) {
379  NativeSurface *surface = private_data_->surfaces_.at(i);
380  surface->SetSurfaceAge(2 - i);
381  surface->SetClearSurface(NativeSurface::kFullClear);
382  }

399 void DisplayPlaneState::RefreshSurfaces(NativeSurface::ClearType clear_surface,
400  bool force) {
401  if (!private_data_->refresh_surface_ && !force) {
402  return;
403  }
404 
405  const HwcRect<int> &target_display_frame = private_data_->display_frame_;
406  HwcRect<float> scaled_rect;
407  CalculateSourceCrop(scaled_rect);
408 
409  for (NativeSurface *surface : private_data_->surfaces_) {
410  surface->ResetDisplayFrame(target_display_frame);
411  surface->ResetSourceCrop(scaled_rect);
412 
413  bool clear = surface->ClearSurface();
414  bool partial_clear = surface->IsPartialClear();
415 
416  if (clear_surface == NativeSurface::kFullClear) {
417  surface->SetClearSurface(NativeSurface::kFullClear);
418  } else if (!clear && !partial_clear) {
419  surface->SetClearSurface(clear_surface);
420  }
421  }
422 
423  if (private_data_->rect_updated_) {
424  ValidateReValidation();
425  }
426 
427  recycled_surface_ = false;
428  private_data_->refresh_surface_ = false;
429 }

591 void DisplayPlaneState::ValidateReValidation() {
592  if (!private_data_->rect_updated_)
593  return;
594 
595  if (private_data_->plane_transform_ != kIdentity &&
596  !private_data_->unsupported_display_rotation_) {
597  re_validate_layer_ |= ReValidationType::kRotation;
598  }
599 
600  if (private_data_->source_layers_.size() == 1 &&
601  !(private_data_->type_ == DisplayPlanePrivateState::PlaneType::kVideo)) {
602  re_validate_layer_ |= ReValidationType::kScanout;
603  } else {
604  bool use_scalar = CanUseDisplayUpScaling();
605  if (private_data_->use_plane_scalar_ != use_scalar) {
606  re_validate_layer_ |= ReValidationType::kUpScalar;
607 #ifdef ENABLE_DOWNSCALING
608  } else {
609  bool down_scale = CanUseGPUDownScaling();
610  if ((private_data_->down_scaling_factor_ > 0) != down_scale) {
611  re_validate_layer_ = ReValidationType::kDownScaling;
612  }
613 #endif
614  }
615  }
616 
617  private_data_->rect_updated_ = false;
618 }

620 bool DisplayPlaneState::CanUseDisplayUpScaling() const {
621  if (!private_data_->rect_updated_) {
622  return private_data_->can_use_display_scalar_;
623  }
624 
625  bool value = true;
626 
627  // We cannot use plane scaling for Layers with different scaling ratio.
628  if (private_data_->source_layers_.size() > 1) {
629  value = false;
630  } else if (private_data_->use_plane_scalar_ &&
631  !private_data_->can_use_downscaling_) {
632  value = false;
633  }
634 
635  if (value) {
636  const HwcRect<int> &target_display_frame = private_data_->display_frame_;
637  const HwcRect<float> &target_src_rect = private_data_->source_crop_;
638  uint32_t display_frame_width =
639  target_display_frame.right - target_display_frame.left;
640  uint32_t display_frame_height =
641  target_display_frame.bottom - target_display_frame.top;
642  uint32_t source_crop_width = static_cast<uint32_t>(
643  ceilf(target_src_rect.right - target_src_rect.left));
644  uint32_t source_crop_height = static_cast<uint32_t>(
645  ceilf(target_src_rect.bottom - target_src_rect.top));
646 
647  if (value) {
648  // Source and Display frame width, height are same and scaling is not
649  // needed.
650  if ((display_frame_width == source_crop_width) &&
651  (display_frame_height == source_crop_height)) {
652  value = false;
653  }
654 
655  if (value) {
656  // Display frame width, height is lesser than Source. Let's downscale
657  // it with our compositor backend.
658  if ((display_frame_width < source_crop_width) &&
659  (display_frame_height < source_crop_height)) {
660  value = false;
661  }
662  }
663 
664  if (value) {
665  // Display frame height is less. If the cost of upscaling width is less
666  // than downscaling height, than return.
667  if ((display_frame_width > source_crop_width) &&
668  (display_frame_height < source_crop_height)) {
669  uint32_t width_cost =
670  (display_frame_width - source_crop_width) * display_frame_height;
671  uint32_t height_cost =
672  (source_crop_height - display_frame_height) * display_frame_width;
673  if (height_cost > width_cost) {
674  value = false;
675  }
676  }
677  }
678 
679  if (value) {
680  // Display frame width is less. If the cost of upscaling height is less
681  // than downscaling width, than return.
682  if ((display_frame_width < source_crop_width) &&
683  (display_frame_height > source_crop_height)) {
684  uint32_t width_cost =
685  (source_crop_width - display_frame_width) * display_frame_height;
686  uint32_t height_cost =
687  (display_frame_height - source_crop_height) * display_frame_width;
688  if (width_cost > height_cost) {
689  value = false;
690  }
691  }
692  }
693  }
694  }
695 
696  private_data_->can_use_display_scalar_ = value;
697

701 bool DisplayPlaneState::CanUseGPUDownScaling() const {
702 #ifndef ENABLE_DOWNSCALING
703  private_data_->can_use_downscaling_ = false;
704  return false;
705 #endif
706  if (!private_data_->rect_updated_) {
707  return private_data_->can_use_downscaling_;
708  }
709 
710  bool value = false;
711  private_data_->can_use_downscaling_ = false;
712  if (!NeedsOffScreenComposition()) {
713  value = false;
714  } else if (private_data_->use_plane_scalar_ &&
715  private_data_->can_use_display_scalar_) {
716  value = false;
717  } else {
718  const HwcRect<int> &target_display_frame = private_data_->display_frame_;
719  const HwcRect<float> &target_src_rect = private_data_->source_crop_;
720 
721  uint32_t display_frame_width =
722  target_display_frame.right - target_display_frame.left;
723  uint32_t display_frame_height =
724  target_display_frame.bottom - target_display_frame.top;
725  uint32_t source_crop_width = static_cast<uint32_t>(
726  ceilf(target_src_rect.right - target_src_rect.left));
727  uint32_t source_crop_height = static_cast<uint32_t>(
728  ceilf(target_src_rect.bottom - target_src_rect.top));
729  if (display_frame_width < 500) {
730  // Ignore < 500 pixels.
731  value = false;
732  } else if ((display_frame_width == source_crop_width) &&
733  (display_frame_height == source_crop_height)) {
734  value = true;
735  } else {
736  // If we are already downscaling content by less than 25%, no need
737  // for any further downscaling.
738  if (display_frame_width >
739  (source_crop_width -
740  (source_crop_width / private_data_->down_scaling_factor_))) {
741  value = true;
742  }
743  }
744  }
745 
746  private_data_->can_use_downscaling_ = value;
747 
748  return private_data_->can_use_display_scalar_;
