55 bool DrmDisplayManager::Initialize() {
56  CTRACE();
57  fd_ = drmOpen("i915", NULL);
58  if (fd_ < 0) {
59  ETRACE("Failed to open dri %s", PRINTERROR());
60  return -ENODEV;
61  }
62 
63  struct drm_set_client_cap cap = {DRM_CLIENT_CAP_UNIVERSAL_PLANES, 1};
64  drmIoctl(fd_, DRM_IOCTL_SET_CLIENT_CAP, &cap);
65  int ret = drmSetClientCap(fd_, DRM_CLIENT_CAP_ATOMIC, 1);
66  if (ret) {
67  ETRACE("Failed to set atomic cap %s", PRINTERROR());
68  return false;
69  }
70 
71  ret = drmSetClientCap(fd_, DRM_CLIENT_CAP_ATOMIC, 1);
72  if (ret) {
73  ETRACE("Failed to set atomic cap %d", ret);
74  return false;
75  }
76 
77  ScopedDrmResourcesPtr res(drmModeGetResources(fd_));
78 
79  for (int32_t i = 0; i < res->count_crtcs; ++i) {
80  ScopedDrmCrtcPtr c(drmModeGetCrtc(fd_, res->crtcs[i]));
81  if (!c) {
82  ETRACE("Failed to get crtc %d", res->crtcs[i]);
83  return false;
84  }
85 
86  std::unique_ptr<DrmDisplay> display(
87  new DrmDisplay(fd_, i, c->crtc_id, this));
88 
89  displays_.emplace_back(std::move(display));
90  }
91 
92 #ifndef DISABLE_HOTPLUG_NOTIFICATION
93  hotplug_fd_ = socket(PF_NETLINK, SOCK_DGRAM, NETLINK_KOBJECT_UEVENT);
94  if (hotplug_fd_ < 0) {
95  ETRACE("Failed to create socket for hot plug monitor. %s", PRINTERROR());
96  return true;
97  }
98 
99  struct sockaddr_nl addr;
100  memset(&addr, 0, sizeof(addr));
101  addr.nl_family = AF_NETLINK;
102  addr.nl_pid = getpid();
103  addr.nl_groups = 0xffffffff;
104 
105  ret = bind(hotplug_fd_, (struct sockaddr *)&addr, sizeof(addr));
106  if (ret) {
107  ETRACE("Failed to bind sockaddr_nl and hot plug monitor fd. %s",
108  PRINTERROR());
109  return true;
110  }
111 
112  fd_handler_.AddFd(hotplug_fd_);
113 #endif
114  IHOTPLUGEVENTTRACE("DisplayManager Initialization succeeded.");
115  return true;
116 }

118 void DrmDisplayManager::HotPlugEventHandler() {
119  CTRACE();
120  int fd = hotplug_fd_;
121  char buffer[DRM_HOTPLUG_EVENT_SIZE];
122  int ret;
123 
124  memset(&buffer, 0, sizeof(buffer));
125  while (true) {
126  bool drm_event = false, hotplug_event = false;
127  size_t srclen = DRM_HOTPLUG_EVENT_SIZE - 1;
128  ret = read(fd, &buffer, srclen);
129  if (ret <= 0) {
130  if (ret < 0)
131  ETRACE("Failed to read uevent. %s", PRINTERROR());
132 
133  return;
134  }
135 
136  buffer[ret] = '\0';
137 
138  for (int32_t i = 0; i < ret;) {
139  char *event = buffer + i;
140  if (!strcmp(event, "DEVTYPE=drm_minor"))
141  drm_event = true;
142  else if (!strcmp(event, "HOTPLUG=1") || // Common hotplug request
143  !strcmp(event,
144  "HDMI-Change")) { // Hotplug happened during suspend
145  hotplug_event = true;
146  }
147 
148  if (hotplug_event && drm_event)
149  break;
150 
151  i += strlen(event) + 1;
152  }
153 
154  if (drm_event && hotplug_event) {
155  IHOTPLUGEVENTTRACE(
156  "Recieved Hot Plug event related to display calling "
157  "UpdateDisplayState.");
158  UpdateDisplayState();
159  }
160  }
161 }

209 bool DrmDisplayManager::UpdateDisplayState() {
210  CTRACE();
211  ScopedDrmResourcesPtr res(drmModeGetResources(fd_));
212  if (!res) {
213  ETRACE("Failed to get DrmResources resources");
214  return false;
215  }
216 
217  spin_lock_.lock();
218  // Start of assuming no displays are connected
219  for (auto &display : displays_) {
220  display->MarkForDisconnect();
221  }
222 
223  std::vector<NativeDisplay *> connected_displays;
224  std::vector<uint32_t> no_encoder;
225  uint32_t total_connectors = res->count_connectors;
226  for (uint32_t i = 0; i < total_connectors; ++i) {
227  ScopedDrmConnectorPtr connector(
228  drmModeGetConnector(fd_, res->connectors[i]));
229  if (!connector) {
230  ETRACE("Failed to get connector %d", res->connectors[i]);
231  break;
232  }
233  // check if a monitor is connected.
234  if (connector->connection != DRM_MODE_CONNECTED)
235  continue;
236 
237  // Ensure we have atleast one valid mode.
238  if (connector->count_modes == 0)
239  continue;
240 
241  if (connector->encoder_id == 0) {
242  no_encoder.emplace_back(i);
243  continue;
244  }
245 
246  std::vector<drmModeModeInfo> mode;
247  uint32_t preferred_mode = 0;
248  uint32_t size = connector->count_modes;
249  mode.resize(size);
250  for (uint32_t i = 0; i < size; ++i) {
251  mode[i] = connector->modes[i];
252  // There is only one preferred mode per connector.
253  if (mode[i].type & DRM_MODE_TYPE_PREFERRED) {
254  preferred_mode = i;
255  }
256  }
257 
258  // Lets try to find crts for any connected encoder.
259  ScopedDrmEncoderPtr encoder(drmModeGetEncoder(fd_, connector->encoder_id));
260  if (encoder && encoder->crtc_id) {
261  for (auto &display : displays_) {
262  IHOTPLUGEVENTTRACE(
263  "Trying to connect %d with crtc: %d is display connected: %d \n",
264  encoder->crtc_id, display->CrtcId(), display->IsConnected());
265  // At initilaization preferred mode is set!
266  if (!display->IsConnected() && encoder->crtc_id == display->CrtcId() &&
267  display->ConnectDisplay(mode.at(preferred_mode), connector.get(),
268  preferred_mode)) {
269  IHOTPLUGEVENTTRACE("Connected %d with crtc: %d pipe:%d \n",
270  encoder->crtc_id, display->CrtcId(),
271  display->GetDisplayPipe());
272  // Set the modes supported for each display
273  display->SetDrmModeInfo(mode);
274  break;
275  }
276  }
277  }
278  }
279 
280  // Deal with connectors with encoder_id == 0.
281  uint32_t size = no_encoder.size();
282  for (uint32_t i = 0; i < size; ++i) {
283  ScopedDrmConnectorPtr connector(
284  drmModeGetConnector(fd_, res->connectors[no_encoder.at(i)]));
285  if (!connector) {
286  ETRACE("Failed to get connector %d", res->connectors[i]);
287  break;
288  }
289 
290  std::vector<drmModeModeInfo> mode;
291  uint32_t preferred_mode = 0;
292  uint32_t size = connector->count_modes;
293  mode.resize(size);
294  for (uint32_t i = 0; i < size; ++i) {
295  mode[i] = connector->modes[i];
296  // There is only one preferred mode per connector.
297  if (mode[i].type & DRM_MODE_TYPE_PREFERRED) {
298  preferred_mode = i;
299  }
300  }
301 
302  // Try to find an encoder for the connector.
303  size = connector->count_encoders;
304  for (uint32_t j = 0; j < size; ++j) {
305  ScopedDrmEncoderPtr encoder(
306  drmModeGetEncoder(fd_, connector->encoders[j]));
307  if (!encoder)
308  continue;
309 
310  for (auto &display : displays_) {
311  if (!display->IsConnected() &&
312  (encoder->possible_crtcs & (1 << display->GetDisplayPipe())) &&
313  display->ConnectDisplay(mode.at(preferred_mode), connector.get(),
314  preferred_mode)) {
315  IHOTPLUGEVENTTRACE("Connected with crtc: %d pipe:%d \n",
316  display->CrtcId(), display->GetDisplayPipe());
317  // Set the modes supported for each display
318  display->SetDrmModeInfo(mode);
319  break;
320  }
321  }
322  }
323  }
324 
325  for (auto &display : displays_) {
326  if (!display->IsConnected()) {
327  display->DisConnect();
328  } else if (callback_) {
329  connected_displays.emplace_back(display.get());
330  }
331  }
332 
333  if (callback_) {
334  callback_->Callback(connected_displays);
335  }
336 
337  spin_lock_.unlock();
338 #ifndef ENABLE_ANDROID_WA
339  notify_client_ = true;
340 #endif
341 
342  if (notify_client_ || (!(displays_.at(0)->IsConnected()))) {
343  IHOTPLUGEVENTTRACE("NotifyClientsOfDisplayChangeStatus Called %d %d \n",
344  notify_client_, displays_.at(0)->IsConnected());
345  NotifyClientsOfDisplayChangeStatus();
346  }
347 
348  return true;
349 }

351 void DrmDisplayManager::NotifyClientsOfDisplayChangeStatus() {
352  spin_lock_.lock();
353  bool disable_last_plane_usage = false;
354  uint32_t total_connected_displays = 0;
355  for (auto &display : displays_) {
356  if (display->IsConnected()) {
357  display->NotifyClientOfDisConnectedState();
358  total_connected_displays++;
359  }
360 
361  if (total_connected_displays > 1) {
362  disable_last_plane_usage = true;
363  break;
364  }
365  }
366 
367  for (auto &display : displays_) {
368  display->NotifyDisplayWA(disable_last_plane_usage);
369  display->ForceRefresh();
370  }
371 
372  for (auto &display : displays_) {
373  if (!display->IsConnected()) {
374  display->NotifyClientOfDisConnectedState();
375  } else {
376  display->NotifyClientOfConnectedState();
377  }
378  }
379 
380  static bool nested_display_registered = false;
381  if (!nested_display_registered) {
382  nested_display_->HotPlugUpdate(true);
383  nested_display_registered = true;
384  }
385 
386 #ifdef ENABLE_ANDROID_WA
387  notify_client_ = true;
388 #endif
389 
390  spin_lock_.unlock();
391 }
