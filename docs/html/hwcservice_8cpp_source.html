293 HwcService::Controls::DisplayModeGetAvailableModes(uint32_t display) {
294  std::vector<HwcsDisplayModeInfo> modes;
295  hwcomposer::NativeDisplay *phyDisplay;
296  if (!display) {
297  phyDisplay = mHwc.GetPrimaryDisplay();
298  } else {
299  phyDisplay = mHwc.GetExtendedDisplay(display - 1);
300  }
301  uint32_t numConfigs;
302  int32_t tempValue;
303  phyDisplay->GetDisplayConfigs(&numConfigs, NULL);
304  for (uint32_t i = 0; i < numConfigs; i++) {
305  HwcsDisplayModeInfo mode;
306  phyDisplay->GetDisplayAttribute(i, hwcomposer::HWCDisplayAttribute::kWidth,
307  &tempValue);
308  mode.width = tempValue;
309  phyDisplay->GetDisplayAttribute(i, hwcomposer::HWCDisplayAttribute::kHeight,
310  &tempValue);
311  mode.height = tempValue;
312  phyDisplay->GetDisplayAttribute(
313  i, hwcomposer::HWCDisplayAttribute::kRefreshRate, &tempValue);
314  mode.refresh = tempValue;
315  phyDisplay->GetDisplayAttribute(i, hwcomposer::HWCDisplayAttribute::kDpiX,
316  &tempValue);
317  mode.xdpi = tempValue;
318  phyDisplay->GetDisplayAttribute(i, hwcomposer::HWCDisplayAttribute::kDpiY,
319  &tempValue);
320  mode.ydpi = tempValue;
321  modes.push_back(mode);
322  }
323  return modes;
324 }
