116 status_t BnService::onTransact(uint32_t code, const Parcel& data, Parcel* reply,
117  uint32_t flags) {
118  switch (code) {
119  case BpService::GET_HWC_VERSION: {
120  CHECK_INTERFACE(IService, data, reply);
121  reply->writeString8(GetHwcVersion());
122  return NO_ERROR;
123  }
124 
125  case BpService::SET_OPTION: {
126  CHECK_INTERFACE(IService, data, reply);
127  String16 option = data.readString16();
128  String16 optionValue = data.readString16();
129  status_t ret = SetOption(String8(option), String8(optionValue));
130  reply->writeInt32(ret);
131  return NO_ERROR;
132  }
133 
134  case BpService::DUMP_OPTIONS: {
135  CHECK_INTERFACE(IService, data, reply);
136  DumpOptions();
137  return NO_ERROR;
138  }
139 
140  case BpService::DISABLE_LOG_TO_LOGCAT: {
141  CHECK_INTERFACE(IService, data, reply);
142  status_t ret = EnableLogviewToLogcat(false);
143  reply->writeInt32(ret);
144  return NO_ERROR;
145  }
146 
147  case BpService::ENABLE_LOG_TO_LOGCAT: {
148  CHECK_INTERFACE(IService, data, reply);
149  status_t ret = EnableLogviewToLogcat();
150  reply->writeInt32(ret);
151  return NO_ERROR;
152  }
153 
154  case BpService::TRANSACT_GET_DIAGNOSTIC: {
155  CHECK_INTERFACE(IService, data, reply);
156  sp<IBinder> b = IInterface::asBinder(this->GetDiagnostic());
157  reply->writeStrongBinder(b);
158  return NO_ERROR;
159  }
160 
161  case BpService::TRANSACT_GET_CONTROLS: {
162  CHECK_INTERFACE(IService, data, reply);
163  sp<IBinder> b = IInterface::asBinder(this->GetControls());
164  reply->writeStrongBinder(b);
165  return NO_ERROR;
166  }
167 
168  default:
169  return BBinder::onTransact(code, data, reply, flags);
170  }
171 }
