113 bool MosaicDisplay::SetActiveConfig(uint32_t config) {
114  config_ = config;
115  width_ = 0;
116  height_ = 0;
117  uint32_t size = physical_displays_.size();
118  for (uint32_t i = 0; i < size; i++) {
119  physical_displays_.at(i)->SetActiveConfig(config_);
120  }
121 
122  uint32_t avg = 0;
123  int32_t previous_refresh = 0;
124  for (uint32_t i = 0; i < size; i++) {
125  int32_t dpix = 0;
126  int32_t dpiy = 0;
127  int32_t refresh = 0;
128  height_ = std::max(height_, physical_displays_.at(i)->Height());
129  width_ += physical_displays_.at(i)->Width();
130  physical_displays_.at(i)->GetDisplayAttribute(
131  config_, HWCDisplayAttribute::kDpiX, &dpix);
132  physical_displays_.at(i)->GetDisplayAttribute(
133  config_, HWCDisplayAttribute::kDpiY, &dpiy);
134  physical_displays_.at(i)->GetDisplayAttribute(
135  config_, HWCDisplayAttribute::kRefreshRate, &refresh);
136  dpix_ += dpix;
137  dpiy_ += dpiy;
138  refresh_ += refresh;
139  if (previous_refresh < refresh)
140  preferred_display_index_ = i;
141 
142  avg++;
143  }
144 
145  if (avg > 0) {
146  refresh_ /= avg;
147  dpix_ /= avg;
148  dpiy_ /= avg;
149  }
150 
151  return true;
152 }

172 bool MosaicDisplay::Present(std::vector<HwcLayer *> &source_layers,
173  int32_t *retire_fence,
174  PixelUploaderCallback *call_back,
175  bool /*handle_constraints*/) {
176  if (power_mode_ != kOn)
177  return true;
178 
179  lock_.lock();
180  if (update_connected_displays_) {
181  std::vector<NativeDisplay *>().swap(connected_displays_);
182  uint32_t size = physical_displays_.size();
183  int32_t previous_refresh = 0;
184  for (uint32_t i = 0; i < size; i++) {
185  if (physical_displays_.at(i)->IsConnected()) {
186  connected_displays_.emplace_back(physical_displays_.at(i));
187  for (uint32_t i = 0; i < size; i++) {
188  int32_t refresh = 0;
189  physical_displays_.at(i)->GetDisplayAttribute(
190  config_, HWCDisplayAttribute::kRefreshRate, &refresh);
191  if (previous_refresh < refresh)
192  preferred_display_index_ = i;
193  }
194  }
195  }
196  update_connected_displays_ = false;
197  }
198  lock_.unlock();
199  uint32_t size = connected_displays_.size();
200  int32_t left_constraint = 0;
201  size_t total_layers = source_layers.size();
202  int32_t fence = -1;
203  for (uint32_t i = 0; i < size; i++) {
204  NativeDisplay *display = connected_displays_.at(i);
205  int32_t right_constraint = left_constraint + display->Width();
206  std::vector<HwcLayer *> layers;
207  uint32_t dlconstraint = display->GetLogicalIndex() * display->Width();
208  uint32_t drconstraint = dlconstraint + display->Width();
209  IMOSAICDISPLAYTRACE("Display index %d \n", i);
210  IMOSAICDISPLAYTRACE("dlconstraint %d \n", dlconstraint);
211  IMOSAICDISPLAYTRACE("drconstraint %d \n", drconstraint);
212  IMOSAICDISPLAYTRACE("right_constraint %d \n", right_constraint);
213  IMOSAICDISPLAYTRACE("left_constraint %d \n", left_constraint);
214  for (size_t j = 0; j < total_layers; j++) {
215  HwcLayer *layer = source_layers.at(j);
216  const HwcRect<int> &frame_Rect = layer->GetDisplayFrame();
217  if ((frame_Rect.right < left_constraint) ||
218  (frame_Rect.left > right_constraint)) {
219  continue;
220  }
221 
222  layer->SetLeftConstraint(dlconstraint);
223  layer->SetRightConstraint(drconstraint);
224  layer->SetLeftSourceConstraint(left_constraint);
225  layer->SetRightSourceConstraint(right_constraint);
226  layer->SetTotalDisplays(size - i);
227 
228  layers.emplace_back(layer);
229  }
230 
231  if (layers.empty()) {
232  continue;
233  }
234 
235  display->Present(layers, &fence, call_back, true);
236  IMOSAICDISPLAYTRACE("Present called for Display index %d \n", i);
237  if (fence > 0 && (i != preferred_display_index_)) {
238  close(fence);
239  } else {
240  *retire_fence = fence;
241  }
242 
243  left_constraint = right_constraint;
244  }
245 
246  return true;
247 }

308 void MosaicDisplay::VSyncUpdate(int64_t timestamp) {
309  lock_.lock();
310  if (vsync_callback_ && enable_vsync_ && vsync_divisor_ > 0) {
311  vsync_counter_--;
312  vsync_timestamp_ += timestamp;
313  if (vsync_counter_ == 0) {
314  vsync_timestamp_ /= vsync_divisor_;
315  vsync_callback_->Callback(display_id_, vsync_timestamp_);
316  vsync_counter_ = vsync_divisor_;
317  vsync_timestamp_ = 0;
318  pending_vsync_ = false;
319  } else {
320  pending_vsync_ = true;
321  }
322  }
323 
324  lock_.unlock();
325 }

333 void MosaicDisplay::HotPlugUpdate(bool connected) {
334  lock_.lock();
335  update_connected_displays_ = true;
336  uint32_t total_connected_displays = 0;
337  uint32_t size = physical_displays_.size();
338  for (uint32_t i = 0; i < size; i++) {
339  if (physical_displays_.at(i)->IsConnected()) {
340  total_connected_displays++;
341  }
342  }
343 
344  if (vsync_callback_ && enable_vsync_ && pending_vsync_ &&
345  (total_connected_displays > 0)) {
346  if (vsync_counter_ == total_connected_displays) {
347  vsync_timestamp_ /= total_connected_displays;
348  vsync_callback_->Callback(display_id_, vsync_timestamp_);
349  pending_vsync_ = false;
350  }
351  }
352 
353  vsync_counter_ = total_connected_displays;
354  vsync_divisor_ = vsync_counter_;
355 
356  if (connected_ == connected) {
357  lock_.unlock();
358  return;
359  }
360 
361  if (hotplug_callback_) {
362  if (!connected && connected_ && total_connected_displays) {
363  lock_.unlock();
364  return;
365  }
366 
367  connected_ = connected;
368  hotplug_callback_->Callback(display_id_, connected);
369  }
370  lock_.unlock();
371 }
