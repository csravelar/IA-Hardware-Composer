45 bool NativeSurface::Init(ResourceManager *resource_manager, uint32_t format,
46  uint32_t usage, uint64_t modifier,
47  bool *modifier_succeeded,
48  FrameBufferManager *frame_buffer_manager) {
49  fb_manager_ = frame_buffer_manager;
50  const NativeBufferHandler *handler =
51  resource_manager->GetNativeBufferHandler();
52  resource_manager_ = resource_manager;
53  HWCNativeHandle native_handle = 0;
54  *modifier_succeeded = false;
55  bool modifier_used = false;
56 
57  if (usage == hwcomposer::kLayerVideo) {
58  modifier = 0;
59  }
60 
61  handler->CreateBuffer(width_, height_, format, &native_handle, usage,
62  &modifier_used, modifier);
63  if (!native_handle) {
64  ETRACE("Failed to create buffer\n");
65  return false;
66  }
67 
68  InitializeLayer(native_handle);
69 
70  if (modifier_used && modifier > 0) {
71  if (!layer_.GetBuffer()->CreateFrameBufferWithModifier(modifier)) {
72  WTRACE("FB creation failed with modifier, removing modifier usage\n");
73  ResourceHandle temp;
74  temp.handle_ = native_handle;
75  resource_manager_->MarkResourceForDeletion(temp, false);
76  native_handle = 0;
77 
78  handler->CreateBuffer(width_, height_, format, &native_handle, usage, 0);
79  if (!native_handle) {
80  ETRACE("Failed to create buffer\n");
81  return false;
82  }
83 
84  InitializeLayer(native_handle);
85  } else {
86  *modifier_succeeded = true;
87  }
88  }
89 
90  modifier_ = modifier;
91  native_handle_ = native_handle;
92 
93  return true;
94 }

159 void NativeSurface::UpdateSurfaceDamage(
160  const HwcRect<int> &currentsurface_damage, bool force) {
161  HwcRect<int> current_damage = currentsurface_damage;
162  if (current_damage.right > width_) {
163  current_damage.right = width_;
164  }
165 
166  if (current_damage.bottom > height_) {
167  current_damage.bottom = height_;
168  }
169 
170  HwcRect<int> &surface_damage = layer_.GetSurfaceDamage();
171  if (reset_damage_) {
172  reset_damage_ = false;
173  surface_damage.reset();
174  }
175 
176  if (surface_damage.empty()) {
177  surface_damage = current_damage;
178  damage_changed_ = true;
179 
180  if (!surface_damage.empty()) {
181  CalculateRect(previous_nc_damage_, surface_damage);
182 
183  previous_nc_damage_ = current_damage;
184  }
185 
186  if (!force && (previous_damage_ == surface_damage))
187  damage_changed_ = false;
188 
189  return;
190  }
191 
192  CalculateRect(current_damage, previous_nc_damage_);
193 
194  if (current_damage == surface_damage) {
195  return;
196  }
197 
198  CalculateRect(current_damage, surface_damage);
199 
200  if (!damage_changed_) {
201  damage_changed_ = true;
202  if (!force && (previous_damage_ == surface_damage))
203  damage_changed_ = false;
204  }
205 }
