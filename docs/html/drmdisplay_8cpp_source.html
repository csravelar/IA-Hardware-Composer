76 bool DrmDisplay::ConnectDisplay(const drmModeModeInfo &mode_info,
77  const drmModeConnector *connector,
78  uint32_t config) {
79  IHOTPLUGEVENTTRACE("DrmDisplay::Connect recieved.");
80  // TODO(kalyan): Add support for multi monitor case.
81  if (connector_ && connector->connector_id == connector_) {
82  IHOTPLUGEVENTTRACE(
83  "Display is already connected to this connector. %d %d %p \n",
84  connector->connector_id, connector_, this);
85  PhysicalDisplay::Connect();
86  return true;
87  }
88 
89  IHOTPLUGEVENTTRACE(
90  "Display is being connected to a new connector.%d %d %p \n",
91  connector->connector_id, connector_, this);
92  connector_ = connector->connector_id;
93  mmWidth_ = connector->mmWidth;
94  mmHeight_ = connector->mmHeight;
95  SetDisplayAttribute(mode_info);
96  config_ = config;
97 
98  ScopedDrmObjectPropertyPtr connector_props(drmModeObjectGetProperties(
99  gpu_fd_, connector_, DRM_MODE_OBJECT_CONNECTOR));
100  if (!connector_props) {
101  ETRACE("Unable to get connector properties.");
102  return false;
103  }
104 
105  int value = -1;
106  GetDrmHDCPObjectProperty("Content Protection", connector, connector_props,
107  &hdcp_id_prop_, &value);
108 
109  if (value >= 0) {
110  switch (value) {
111  case 0:
112  current_protection_support_ = HWCContentProtection::kUnDesired;
113  break;
114  case 1:
115  current_protection_support_ = HWCContentProtection::kDesired;
116  break;
117  default:
118  break;
119  }
120 
121  if (desired_protection_support_ == HWCContentProtection::kUnSupported) {
122  desired_protection_support_ = current_protection_support_;
123  }
124  }
125 
126  GetDrmObjectProperty("CRTC_ID", connector_props, &crtc_prop_);
127  GetDrmObjectProperty("Broadcast RGB", connector_props, &broadcastrgb_id_);
128  GetDrmObjectProperty("DPMS", connector_props, &dpms_prop_);
129 
130  PhysicalDisplay::Connect();
131  SetHDCPState(desired_protection_support_, content_type_);
132 
133  drmModePropertyPtr broadcastrgb_props =
134  drmModeGetProperty(gpu_fd_, broadcastrgb_id_);
135 
136  SetPowerMode(power_mode_);
137 
138  // This is a valid case on DSI panels.
139  if (broadcastrgb_props == NULL) {
140  WTRACE("Unable to get Broadcast RGB properties\n");
141  return true;
142  }
143 
144  if (!(broadcastrgb_props->flags & DRM_MODE_PROP_ENUM)) {
145  drmModeFreeProperty(broadcastrgb_props);
146  return false;
147  }
148 
149  if (broadcastrgb_props->enums != NULL) {
150  for (int i = 0; i < broadcastrgb_props->count_enums; i++) {
151  if (!strcmp(broadcastrgb_props->enums[i].name, "Full")) {
152  broadcastrgb_full_ = broadcastrgb_props->enums[i].value;
153  } else if (!strcmp(broadcastrgb_props->enums[i].name, "Automatic")) {
154  broadcastrgb_automatic_ = broadcastrgb_props->enums[i].value;
155  }
156  }
157  }
158 
159  drmModeFreeProperty(broadcastrgb_props);
160 
161  return true;
162 }

164 bool DrmDisplay::GetDisplayAttribute(uint32_t config /*config*/,
165  HWCDisplayAttribute attribute,
166  int32_t *value) {
167  SPIN_LOCK(display_lock_);
168  if (modes_.empty()) {
169  SPIN_UNLOCK(display_lock_);
170  return PhysicalDisplay::GetDisplayAttribute(config, attribute, value);
171  }
172 
173  float refresh;
174  bool status = true;
175 
176  switch (attribute) {
177  case HWCDisplayAttribute::kWidth:
178  if (!custom_resolution_) {
179  *value = modes_[config].hdisplay;
180  } else {
181  *value = rect_.right - rect_.left;
182  }
183  IHOTPLUGEVENTTRACE("GetDisplayAttribute: width %d set", *value);
184  break;
185  case HWCDisplayAttribute::kHeight:
186  if (!custom_resolution_) {
187  *value = modes_[config].vdisplay;
188  } else {
189  *value = rect_.bottom - rect_.top;
190  }
191  IHOTPLUGEVENTTRACE("GetDisplayAttribute: height %d set", *value);
192  break;
193  case HWCDisplayAttribute::kRefreshRate:
194  if (!custom_resolution_) {
195  refresh = (modes_[config].clock * 1000.0f) /
196  (modes_[config].htotal * modes_[config].vtotal);
197  } else {
198  refresh = (modes_[config].clock * 1000.0f) /
199  ((rect_.right - rect_.left) * (rect_.bottom - rect_.top));
200  }
201 
202  if (modes_[config].flags & DRM_MODE_FLAG_INTERLACE)
203  refresh *= 2;
204 
205  if (modes_[config].flags & DRM_MODE_FLAG_DBLSCAN)
206  refresh /= 2;
207 
208  if (modes_[config].vscan > 1)
209  refresh /= modes_[config].vscan;
210 
211  // in nanoseconds
212  *value = 1e9 / refresh;
213  break;
214  case HWCDisplayAttribute::kDpiX:
215  // Dots per 1000 inches
216  if (!custom_resolution_) {
217  *value =
218  mmWidth_ ? (modes_[config].hdisplay * kUmPerInch) / mmWidth_ : -1;
219  } else {
220  *value =
221  mmWidth_ ? ((rect_.right - rect_.left) * kUmPerInch) / mmWidth_ : -1;
222  }
223  break;
224  case HWCDisplayAttribute::kDpiY:
225  // Dots per 1000 inches
226  if (!custom_resolution_) {
227  *value =
228  mmHeight_ ? (modes_[config].vdisplay * kUmPerInch) / mmHeight_ : -1;
229  } else {
230  *value =
231  mmHeight_ ? ((rect_.bottom - rect_.top) * kUmPerInch) /mmHeight_: -1;
232  }
233  break;
234  default:
235  *value = -1;
236  status = false;
237  }
238 
239  SPIN_UNLOCK(display_lock_);
240  return status;
241 }

360 bool DrmDisplay::Commit(
361  const DisplayPlaneStateList &composition_planes,
362  const DisplayPlaneStateList &previous_composition_planes,
363  bool disable_explicit_fence, int32_t previous_fence, int32_t *commit_fence,
364  bool *previous_fence_released) {
365  // Do the actual commit.
366  ScopedDrmAtomicReqPtr pset(drmModeAtomicAlloc());
367  *previous_fence_released = false;
368 
369  if (!pset) {
370  ETRACE("Failed to allocate property set %d", -ENOMEM);
371  return false;
372  }
373 
374  if (display_state_ & kNeedsModeset) {
375  if (!ApplyPendingModeset(pset.get())) {
376  ETRACE("Failed to Modeset.");
377  return false;
378  }
379  } else if (!disable_explicit_fence && out_fence_ptr_prop_) {
380  GetFence(pset.get(), commit_fence);
381  }
382 
383  if (!CommitFrame(composition_planes, previous_composition_planes, pset.get(),
384  flags_, previous_fence, previous_fence_released)) {
385  ETRACE("Failed to Commit layers.");
386  return false;
387  }
388 
389  if (display_state_ & kNeedsModeset) {
390  display_state_ &= ~kNeedsModeset;
391  if (!disable_explicit_fence) {
392  flags_ = 0;
393  flags_ |= DRM_MODE_ATOMIC_NONBLOCK;
394  }
395  }
396 
397 #ifdef ENABLE_DOUBLE_BUFFERING
398  int32_t fence = *commit_fence;
399  if (fence > 0) {
400  HWCPoll(fence, -1);
401  close(fence);
402  *commit_fence = 0;
403  }
404 #endif
405 
406  return true;
407 }

409 bool DrmDisplay::CommitFrame(
410  const DisplayPlaneStateList &comp_planes,
411  const DisplayPlaneStateList &previous_composition_planes,
412  drmModeAtomicReqPtr pset, uint32_t flags, int32_t previous_fence,
413  bool *previous_fence_released) {
414  CTRACE();
415  if (!pset) {
416  ETRACE("Failed to allocate property set %d", -ENOMEM);
417  return false;
418  }
419 
420  for (const DisplayPlaneState &comp_plane : comp_planes) {
421  DrmPlane *plane = static_cast<DrmPlane *>(comp_plane.GetDisplayPlane());
422  const OverlayLayer *layer = comp_plane.GetOverlayLayer();
423  int32_t fence = layer->GetAcquireFence();
424  if (fence > 0) {
425  plane->SetNativeFence(dup(fence));
426  } else {
427  plane->SetNativeFence(-1);
428  }
429 
430  if (comp_plane.Scanout() && !comp_plane.IsSurfaceRecycled())
431  plane->SetBuffer(layer->GetSharedBuffer());
432 
433  if (!plane->UpdateProperties(pset, crtc_id_, layer))
434  return false;
435  }
436 
437  for (const DisplayPlaneState &comp_plane : previous_composition_planes) {
438  DrmPlane *plane = static_cast<DrmPlane *>(comp_plane.GetDisplayPlane());
439  if (plane->InUse())
440  continue;
441 
442  plane->Disable(pset);
443  }
444 
445 #ifndef ENABLE_DOUBLE_BUFFERING
446  if (previous_fence > 0) {
447  HWCPoll(previous_fence, -1);
448  close(previous_fence);
449  *previous_fence_released = true;
450  }
451 #endif
452 
453  int ret = drmModeAtomicCommit(gpu_fd_, pset, flags, NULL);
454  if (ret) {
455  ETRACE("Failed to commit pset ret=%s\n", PRINTERROR());
456  return false;
457  }
458 
459  return true;
460 }

615 uint64_t DrmDisplay::DrmRGBA(uint16_t bpc, uint16_t red, uint16_t green,
616  uint16_t blue, uint16_t alpha) const {
617  if (bpc > 16)
618  bpc = 16;
619 
620  /*
621  * If we were provided with fewer than 16 bpc, shift the value we
622  * received into the most significant bits.
623  */
624  int shift = 16 - bpc;
625 
626  uint64_t val = 0;
627  val = red << shift;
628  val <<= 16;
629  val |= green << shift;
630  val <<= 16;
631  val |= blue << shift;
632  val <<= 16;
633  val |= alpha << shift;
634 
635  return val;
636 }

665 float DrmDisplay::TransformGamma(float value, float gamma) const {
666  float result;
667 
668  result = pow(value, gamma);

677 void DrmDisplay::SetColorTransformMatrix(
678  const float *color_transform_matrix,
679  HWCColorTransform color_transform_hint) const {
680  struct drm_color_ctm *ctm =
681  (struct drm_color_ctm *)malloc(sizeof(struct drm_color_ctm));
682  if (!ctm) {
683  ETRACE("Cannot allocate CTM memory");
684  return;
685  }
686 
687  struct drm_color_ctm_post_offset *ctm_post_offset =
688  (struct drm_color_ctm_post_offset *)malloc(
689  sizeof(struct drm_color_ctm_post_offset));
690  if (!ctm_post_offset) {
691  free(ctm);
692  ETRACE("Cannot allocate ctm_post_offset memory");
693  return;
694  }
695 
696  switch (color_transform_hint) {
697  case HWCColorTransform::kIdentical: {
698  memset(ctm->matrix, 0, sizeof(ctm->matrix));
699  for (int i = 0; i < 3; i++) {
700  ctm->matrix[i * 3 + i] = (1ll << 32);
701  }
702  ctm_post_offset->red = 0;
703  ctm_post_offset->green = 0;
704  ctm_post_offset->blue = 0;
705  ApplyPendingCTM(ctm, ctm_post_offset);
706  break;
707  }
708  case HWCColorTransform::kArbitraryMatrix: {
709  for (int i = 0; i < 3; i++) {
710  for (int j = 0; j < 3; j++) {
711  ctm->matrix[i * 3 + j] =
712  FloatToFixedPoint(color_transform_matrix[j * 4 + i]);
713  }
714  }
715  ctm_post_offset->red = color_transform_matrix[12] * 0xffff;
716  ctm_post_offset->green = color_transform_matrix[13] * 0xffff;
717  ctm_post_offset->blue = color_transform_matrix[14] * 0xffff;
718  ApplyPendingCTM(ctm, ctm_post_offset);
719  break;
720  }
721  }
722  free(ctm);

724 void DrmDisplay::SetColorCorrection(struct gamma_colors gamma,
725  uint32_t contrast_c,
726  uint32_t brightness_c) const {
727  struct drm_color_lut *lut;
728  float brightness[3];
729  float contrast[3];
730  uint8_t temp[3];
731 
732  /* reset lut when contrast and brightness are all 0 */
733  if (contrast_c == 0 && brightness_c == 0) {
734  lut = NULL;
735  ApplyPendingLUT(lut);
736  free(lut);
737  return;
738  }
739 
740  lut =
741  (struct drm_color_lut *)malloc(sizeof(struct drm_color_lut) * lut_size_);
742  if (!lut) {
743  ETRACE("Cannot allocate LUT memory");
744  return;
745  }
746 
747  /* Unpack brightness values for each channel */
748  temp[0] = (brightness_c >> 16) & 0xFF;
749  temp[1] = (brightness_c >> 8) & 0xFF;
750  temp[2] = (brightness_c)&0xFF;
751 
752  /* Map brightness from -128 - 127 range into -0.5 - 0.5 range */
753  brightness[0] = (float)(temp[0]) / 255 - 0.5;
754  brightness[1] = (float)(temp[1]) / 255 - 0.5;
755  brightness[2] = (float)(temp[2]) / 255 - 0.5;
756 
757  /* Unpack contrast values for each channel */
758  temp[0] = (contrast_c >> 16) & 0xFF;
759  temp[1] = (contrast_c >> 8) & 0xFF;
760  temp[2] = (contrast_c)&0xFF;
761 
762  /* Map contrast from 0 - 255 range into 0.0 - 2.0 range */
763  contrast[0] = (float)(temp[0]) / 128;
764  contrast[1] = (float)(temp[1]) / 128;
765  contrast[2] = (float)(temp[2]) / 128;
766 
767  for (uint64_t i = 0; i < lut_size_; i++) {
768  /* Set lut[0] as 0 always as the darkest color should has brightness 0 */
769  if (i == 0) {
770  lut[i].red = 0;
771  lut[i].green = 0;
772  lut[i].blue = 0;
773  continue;
774  }
775 
776  lut[i].red = 0xFFFF * TransformGamma(TransformContrastBrightness(
777  (float)(i) / lut_size_,
778  brightness[0], contrast[0]),
779  gamma.red);
780  lut[i].green = 0xFFFF * TransformGamma(TransformContrastBrightness(
781  (float)(i) / lut_size_,
782  brightness[1], contrast[1]),
783  gamma.green);
784  lut[i].blue = 0xFFFF * TransformGamma(TransformContrastBrightness(
785  (float)(i) / lut_size_,
786  brightness[2], contrast[2]),
787  gamma.blue);
788  }
789 
790  ApplyPendingLUT(lut);
791  free(lut);
792 }

849 bool DrmDisplay::PopulatePlanes(
850  std::vector<std::unique_ptr<DisplayPlane>> &overlay_planes) {
851  ScopedDrmPlaneResPtr plane_resources(drmModeGetPlaneResources(gpu_fd_));
852  if (!plane_resources) {
853  ETRACE("Failed to get plane resources");
854  return false;
855  }
856 
857  uint32_t num_planes = plane_resources->count_planes;
858  uint32_t pipe_bit = 1 << pipe_;
859  std::set<uint32_t> plane_ids;
860  std::unique_ptr<DisplayPlane> cursor_plane;
861  for (uint32_t i = 0; i < num_planes; ++i) {
862  ScopedDrmPlanePtr drm_plane(
863  drmModeGetPlane(gpu_fd_, plane_resources->planes[i]));
864  if (!drm_plane) {
865  ETRACE("Failed to get plane ");
866  return false;
867  }
868 
869  if (!(pipe_bit & drm_plane->possible_crtcs))
870  continue;
871 
872  uint32_t formats_size = drm_plane->count_formats;
873  plane_ids.insert(drm_plane->plane_id);
874  std::unique_ptr<DrmPlane> plane(
875  CreatePlane(drm_plane->plane_id, drm_plane->possible_crtcs));
876  std::vector<uint32_t> supported_formats(formats_size);
877  for (uint32_t j = 0; j < formats_size; j++)
878  supported_formats[j] = drm_plane->formats[j];
879 
880  if (plane->Initialize(gpu_fd_, supported_formats)) {
881  if (plane->type() == DRM_PLANE_TYPE_CURSOR) {
882  cursor_plane.reset(plane.release());
883  } else {
884  overlay_planes.emplace_back(plane.release());
885  }
886  }
887  }
888 
889  if (overlay_planes.empty()) {
890  ETRACE("Failed to get primary plane for display %d", crtc_id_);
891  return false;
892  }
893 
894  // We expect layers to be in ascending order.
895  std::sort(
896  overlay_planes.begin(), overlay_planes.end(),
897  [](const std::unique_ptr<DisplayPlane> &l,
898  const std::unique_ptr<DisplayPlane> &r) { return l->id() < r->id(); });
899 
900  if (cursor_plane) {
901  overlay_planes.emplace_back(cursor_plane.release());
902  }
903 
904  return true;
905 }
