85 void VblankEventHandler::HandlePageFlipEvent(unsigned int sec,
86  unsigned int usec) {
87  int64_t timestamp = ((int64_t)sec * kOneSecondNs) + ((int64_t)usec * 1000);
88  IPAGEFLIPEVENTTRACE("HandleVblankCallBack Frame Time %f",
89  static_cast<float>(timestamp - last_timestamp_) / (1000));
90  last_timestamp_ = timestamp;

92  IPAGEFLIPEVENTTRACE("Callback called from HandlePageFlipEvent. %lu",
93  timestamp);
94  spin_lock_.lock();
95  if (enabled_ && callback_) {
96  callback_->Callback(display_, timestamp);
