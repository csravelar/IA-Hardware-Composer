109  hook(data_, display, status);

111  // FIXME: SurfaceFlinger doesn't seem to reset layers correctly
112  // when display is connected/disconnected. We force it here.
113  // Remove this workaround once fixed correctly in SurfaceFlinger.
114  if (!connected && display > 0) {
115  display_->FreeAllLayers();

134 HWC2::Error IAHWC2::Init() {
135  char value[PROPERTY_VALUE_MAX];
136  property_get("board.disable.explicit.sync", value, "0");
137  disable_explicit_sync_ = atoi(value);
138  if (disable_explicit_sync_)
139  ALOGI("EXPLICIT SYNC support is disabled");
140  else
141  ALOGI("EXPLICIT SYNC support is enabled");
142 
143  property_get("board.hwc.scaling.mode", value, "2");
144  scaling_mode_ = atoi(value);
145  switch (scaling_mode_) {
146  case 1:
147  ALOGI("HWC Scaling Mode Fast");
148  break;
149  case 2:
150  ALOGI("HWC Scaling Mode High Quality");
151  break;
152  default:
153  ALOGI("Unsupport HWC Scaling Mode");
154  break;
155  }
156 
157  if (!device_.Initialize()) {
158  ALOGE("Can't initialize drm object.");
159  return HWC2::Error::NoResources;
160  }
161 
162  const std::vector<NativeDisplay *> &displays = device_.GetAllDisplays();
163  NativeDisplay *primary_display = displays.at(0);
164  uint32_t external_display_id = 1;
165  primary_display_.Init(primary_display, 0, disable_explicit_sync_,
166  scaling_mode_);
167  size_t size = displays.size();
168 
169  // Add nested display which is expected to use the output of this display and
170  // show it himself. Nested display can be implemented with real HW display.
171  // It's different from virtual display.
172  nested_display_.InitNestedDisplay(device_.GetNestedDisplay(),
173  disable_explicit_sync_);
174 
175  for (size_t i = 0; i < size; ++i) {
176  hwcomposer::NativeDisplay *display = displays.at(i);
177  if (display == primary_display)
178  continue;
179 
180  std::unique_ptr<HwcDisplay> temp(new HwcDisplay());
181  temp->Init(display, external_display_id, disable_explicit_sync_,
182  scaling_mode_);
183  extended_displays_.emplace_back(std::move(temp));
184  external_display_id++;
185 
186  // Let's not confuse things with Virtual Display.
187  if (external_display_id == HWC_DISPLAY_VIRTUAL)
188  external_display_id = HWC_DISPLAY_VIRTUAL + 1;
189  }
190 
191  // Start the hwc service
192  hwcService_.Start(*this);
193 
194  return HWC2::Error::None;
195 }

239 HWC2::Error IAHWC2::RegisterCallback(int32_t descriptor,
240  hwc2_callback_data_t data,
241  hwc2_function_pointer_t function) {
242  supported(__func__);
243  auto callback = static_cast<HWC2::Callback>(descriptor);
244  size_t size = extended_displays_.size();
245  HWC2::Error error = HWC2::Error::None;
246 
247  switch (callback) {
248  case HWC2::Callback::Hotplug: {
249  primary_display_.RegisterHotPlugCallback(data, function);
250 #ifdef NESTED_DISPLAY_SUPPORT
251  nested_display_.RegisterHotPlugCallback(data, function);
252 #endif
253  for (size_t i = 0; i < size; ++i) {
254  IAHWC2::HwcDisplay *display = extended_displays_.at(i).get();
255  display->RegisterHotPlugCallback(data, function);
256  }
257 
258  break;
259  }
260  case HWC2::Callback::Vsync: {
261  primary_display_.RegisterVsyncCallback(data, function);
262  for (size_t i = 0; i < size; ++i) {
263  IAHWC2::HwcDisplay *display = extended_displays_.at(i).get();
264  display->RegisterVsyncCallback(data, function);
265  }
266 
267  break;
268  }
269  case HWC2::Callback::Refresh: {
270  primary_display_.RegisterRefreshCallback(data, function);
271  for (size_t i = 0; i < size; ++i) {
272  IAHWC2::HwcDisplay *display = extended_displays_.at(i).get();
273  display->RegisterRefreshCallback(data, function);
274  }
275 
276  break;
277  }
278  default:
279  error = HWC2::Error::BadParameter;
280  break;
281  }
282  return error;
283 }

622 HWC2::Error IAHWC2::HwcDisplay::PresentDisplay(int32_t *retire_fence) {
623  supported(__func__);
624  // order the layers by z-order
625  bool use_client_layer = false;
626  uint32_t client_z_order = 0;
627  bool use_cursor_layer = false;
628  uint32_t cursor_z_order = 0;
629  IAHWC2::Hwc2Layer *cursor_layer;
630  *retire_fence = -1;
631  std::map<uint32_t, IAHWC2::Hwc2Layer *> z_map;
632 
633  // if the power mode is doze suspend then its the hint that the drawing
634  // into the display has suspended and remain in the low power state and
635  // continue displaying the current state and stop applying display
636  // update from the client
637  if (display_->PowerMode() == HWC2_POWER_MODE_DOZE_SUSPEND)
638  return HWC2::Error::None;
639  for (std::pair<const hwc2_layer_t, IAHWC2::Hwc2Layer> &l : layers_) {
640  if (l.second.IsCursorLayer()) {
641  use_cursor_layer = true;
642  cursor_layer = &l.second;
643  cursor_z_order = l.second.z_order();
644  continue;
645  }
646 
647  switch (l.second.validated_type()) {
648  case HWC2::Composition::Device:
649  z_map.emplace(std::make_pair(l.second.z_order(), &l.second));
650  break;
651  case HWC2::Composition::Client:
652  // Place it at the z_order of the highest client layer
653  use_client_layer = true;
654  client_z_order = std::max(client_z_order, l.second.z_order());
655  break;
656  default:
657  continue;
658  }
659  }
660  if (use_client_layer && client_layer_.GetLayer() &&
661  client_layer_.GetLayer()->GetNativeHandle() &&
662  client_layer_.GetLayer()->GetNativeHandle()->handle_) {
663  z_map.emplace(std::make_pair(client_z_order, &client_layer_));
664  }
665 
666  // Place the cursor at the highest z-order
667  if (use_cursor_layer) {
668  if (z_map.size()) {
669  if (z_map.rbegin()->second->z_order() > cursor_z_order)
670  cursor_z_order = z_map.rbegin()->second->z_order() + 1;
671  else if (client_z_order > cursor_z_order)
672  cursor_z_order = client_z_order + 1;
673  }
674  z_map.emplace(std::make_pair(cursor_z_order, cursor_layer));
675  }
676 
677  std::vector<hwcomposer::HwcLayer *> layers;
678  // now that they're ordered by z, add them to the composition
679  for (std::pair<const uint32_t, IAHWC2::Hwc2Layer *> &l : z_map) {
680  layers.emplace_back(l.second->GetLayer());
681  }
682 
683  if (layers.empty())
684  return HWC2::Error::None;
685 
686  IHOTPLUGEVENTTRACE("PhysicalDisplay called for Display: %p \n", display_);
687 
688  bool success = display_->Present(layers, retire_fence);
689  if (!success) {
690  ALOGE("Failed to set layers in the composition");
691  return HWC2::Error::BadLayer;
692  }
693 
694  ++frame_no_;
695 
696  return HWC2::Error::None;
697 }
