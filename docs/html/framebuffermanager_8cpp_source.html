23 void FrameBufferManager::RegisterGemHandles(const uint32_t &num_planes,

26  FBKey key(num_planes, igem_handles);
27  auto it = fb_map_.find(key);
28  if (it != fb_map_.end()) {

57

59  } else {
60  ETRACE("Handle not found in Cache \n");
61  }

63  lock_.unlock();

67 int FrameBufferManager::RemoveFB(uint32_t num_planes,
68  const uint32_t (&igem_handles)[4]) {
69  lock_.lock();
70 
71  int ret = 0;
72  FBKey key(num_planes, igem_handles);
73 
74  auto it = fb_map_.find(key);
75  if (it != fb_map_.end()) {
76  it->second.fb_ref -= 1;
77  if (it->second.fb_ref == 0) {
78  ret = ReleaseFrameBuffer(it->first, it->second.fb_id, gpu_fd_);
79  fb_map_.erase(it);
80  }
81  }
82 
83  if (it == fb_map_.end()) {
84  if (igem_handles[0] != 0 || igem_handles[1] != 0 || igem_handles[2] != 0 ||
85  igem_handles[3] != 0) {
86  ETRACE("Unable to find fb in cache. %d %d %d %d \n", igem_handles[0],
87  igem_handles[1], igem_handles[2], igem_handles[3]);
88  }
89  }
90 
91  lock_.unlock();
92 
93  return ret;
94 }
