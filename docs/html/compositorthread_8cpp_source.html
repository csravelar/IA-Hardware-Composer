83 bool CompositorThread::Draw(std::vector<DrawState> &states,
84  std::vector<DrawState> &media_states,
85  const std::vector<OverlayLayer> &layers) {
86  states_.swap(states);
87  tasks_lock_.lock();
88 
89  if (!states_.empty()) {
90  std::vector<OverlayBuffer *>().swap(buffers_);
91  buffers_.reserve(layers.size());
92  for (auto &layer : layers) {
93  buffers_.emplace_back(layer.GetBuffer());
94  }
95 
96  tasks_ |= kRender3D;
97  }
98 
99  if (!media_states.empty()) {
100  media_states_.swap(media_states);
101  tasks_ |= kRenderMedia;
102  }
103 
104  // We start of assuming that the draw calls
105  // succeed.
106  draw_succeeded_ = true;
107  tasks_lock_.unlock();
108 
109  // Adding check to avoid waiting in this
110  // thread in certain corner case.
111  if (states_.empty() && media_states_.empty()) {
112  return draw_succeeded_;
113  }
114 
115  Resume();
116  Wait();
117  return draw_succeeded_;
118 }

153 void CompositorThread::HandleReleaseRequest() {
154  ScopedSpinLock lock(tasks_lock_);
155  tasks_ &= ~kReleaseResources;
156 
157  std::vector<ResourceHandle> purged_gl_resources;
158  std::vector<MediaResourceHandle> purged_media_resources;
159  bool has_gpu_resource = false;
160  resource_manager_->GetPurgedResources(
161  purged_gl_resources, purged_media_resources, &has_gpu_resource);
162  size_t purged_size = purged_gl_resources.size();
163 
164  if (purged_size != 0) {
165  if (has_gpu_resource) {
166  Ensure3DRenderer();
167  gpu_resource_handler_->ReleaseGPUResources(purged_gl_resources);
168  }
169 
170  const NativeBufferHandler *handler =
171  resource_manager_->GetNativeBufferHandler();
172 
173  for (size_t i = 0; i < purged_size; i++) {
174  const ResourceHandle &handle = purged_gl_resources.at(i);
175  if (!handle.handle_) {
176  continue;
177  }
178 
179  fb_manager_->RemoveFB(handle.handle_->meta_data_.num_planes_,
180  handle.handle_->meta_data_.gem_handles_);
181 
182  handler->ReleaseBuffer(handle.handle_);
183  handler->DestroyHandle(handle.handle_);
184  }
185  }
186 
187  purged_size = purged_media_resources.size();
188 
189  if (purged_size != 0) {
190  EnsureMediaRenderer();
191  media_renderer_->DestroyMediaResources(purged_media_resources);
192 
193  const NativeBufferHandler *handler =
194  resource_manager_->GetNativeBufferHandler();
195 
196  for (size_t i = 0; i < purged_size; i++) {
197  const MediaResourceHandle &handle = purged_media_resources.at(i);
198  if (!handle.handle_) {
199  continue;
200  }
201 
202  fb_manager_->RemoveFB(handle.handle_->meta_data_.num_planes_,
203  handle.handle_->meta_data_.gem_handles_);
204  handler->ReleaseBuffer(handle.handle_);
205  handler->DestroyHandle(handle.handle_);
206  }
207  }
208 }

210 void CompositorThread::Handle3DDrawRequest() {
211  tasks_lock_.lock();
212  tasks_ &= ~kRender3D;
213  tasks_lock_.unlock();
214 
215  Ensure3DRenderer();
216  if (!gl_renderer_) {
217  draw_succeeded_ = false;
218  return;
219  }
220 
221  gl_renderer_->SetExplicitSyncSupport(disable_explicit_sync_);
222 
223  if (!gpu_resource_handler_->PrepareResources(buffers_)) {
224  ETRACE(
225  "Failed to prepare GPU resources for compositing the frame, "
226  "error: %s",
227  PRINTERROR());
228  draw_succeeded_ = false;
229  return;
230  }
231 
232  size_t size = states_.size();
233  for (size_t i = 0; i < size; i++) {
234  DrawState &draw_state = states_.at(i);
235  for (RenderState &render_state : draw_state.states_) {
236  std::vector<RenderState::LayerState> &layer_state =
237  render_state.layer_state_;
238 
239  for (RenderState::LayerState &temp : layer_state) {
240  temp.handle_ =
241  gpu_resource_handler_->GetResourceHandle(temp.layer_index_);
242  }
243  }
244 
245  const std::vector<int32_t> &fences = draw_state.acquire_fences_;
246  for (int32_t fence : fences) {
247  gl_renderer_->InsertFence(fence);
248  }
249 
250  std::vector<int32_t>().swap(draw_state.acquire_fences_);
251 
252  if (!gl_renderer_->Draw(draw_state.states_, draw_state.surface_)) {
253  ETRACE(
254  "Failed to Draw: "
255  "error: %s",
256  PRINTERROR());
257  draw_succeeded_ = false;
258  break;
259  }
260 
261  if (draw_state.destroy_surface_) {
262  if (draw_succeeded_) {
263  draw_state.retire_fence_ =
264  draw_state.surface_->GetLayer()->ReleaseAcquireFence();
265  }
266 
267  delete draw_state.surface_;
268  }
269  }
270 
271  if (disable_explicit_sync_)
272  gl_renderer_->InsertFence(-1);
273 }

275 void CompositorThread::HandleMediaDrawRequest() {
276  tasks_lock_.lock();
277  tasks_ &= ~kRenderMedia;
278  tasks_lock_.unlock();
279 
280  EnsureMediaRenderer();

290  ETRACE(

293  PRINTERROR());
