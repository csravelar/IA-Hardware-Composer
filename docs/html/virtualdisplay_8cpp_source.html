81 bool VirtualDisplay::Present(std::vector<HwcLayer *> &source_layers,
82  int32_t *retire_fence,
83  PixelUploaderCallback * /*call_back*/,
84  bool handle_constraints) {
85  CTRACE();
86  std::vector<OverlayLayer> layers;
87  std::vector<HwcRect<int>> layers_rects;
88  std::vector<size_t> index;
89  int ret = 0;
90  size_t size = source_layers.size();
91  size_t previous_size = in_flight_layers_.size();
92  bool frame_changed = (size != previous_size);
93  bool layers_changed = frame_changed;
94  *retire_fence = -1;
95  uint32_t z_order = 0;
96 
97  resource_manager_->RefreshBufferCache();
98  for (size_t layer_index = 0; layer_index < size; layer_index++) {
99  HwcLayer *layer = source_layers.at(layer_index);
100  layer->SetReleaseFence(-1);
101  if (!layer->IsVisible())
102  continue;
103 
104  layers.emplace_back();
105  OverlayLayer &overlay_layer = layers.back();
106  OverlayLayer *previous_layer = NULL;
107  if (previous_size > z_order) {
108  previous_layer = &(in_flight_layers_.at(z_order));
109  }
110 
111  overlay_layer.InitializeFromHwcLayer(
112  layer, resource_manager_.get(), previous_layer, z_order, layer_index,
113  height_, kIdentity, handle_constraints, fb_manager_);
114  index.emplace_back(z_order);
115  layers_rects.emplace_back(layer->GetDisplayFrame());
116  z_order++;
117 
118  if (frame_changed) {
119  layer->Validate();
120  continue;
121  }
122 
123  if (!previous_layer || overlay_layer.HasLayerContentChanged() ||
124  overlay_layer.HasDimensionsChanged()) {
125  layers_changed = true;
126  }
127 
128  layer->Validate();
129  }
130 
131  if (layers_changed) {
132  compositor_.BeginFrame(false);
133 
134  // Prepare for final composition.
135  if (!compositor_.DrawOffscreen(
136  layers, layers_rects, index, resource_manager_.get(), width_,
137  height_, output_handle_, acquire_fence_, retire_fence)) {
138  ETRACE("Failed to prepare for the frame composition ret=%d", ret);
139  return false;
140  }
141 
142  acquire_fence_ = 0;
143 
144  in_flight_layers_.swap(layers);
145  }
146 
147  int32_t fence = *retire_fence;
148 
149  if (fence > 0) {
150  for (size_t layer_index = 0; layer_index < size; layer_index++) {
151  HwcLayer *layer = source_layers.at(layer_index);
152  layer->SetReleaseFence(dup(fence));
153  }
154  } else {
155  for (size_t layer_index = 0; layer_index < size; layer_index++) {
156  const OverlayLayer &overlay_layer =
157  in_flight_layers_.at(index.at(layer_index));
158  HwcLayer *layer = source_layers.at(overlay_layer.GetLayerIndex());
159  layer->SetReleaseFence(overlay_layer.ReleaseAcquireFence());
160  }
161  }
162 
163  compositor_.FreeResources();
164 
165  return true;
166 }

168 void VirtualDisplay::SetOutputBuffer(HWCNativeHandle buffer,
169  int32_t acquire_fence) {
170  if (!output_handle_ || output_handle_ != buffer) {
171  const NativeBufferHandler *handler =
172  resource_manager_->GetNativeBufferHandler();
173 
174  if (handle_) {
175  ResourceHandle temp;
176  temp.handle_ = handle_;
177  resource_manager_->MarkResourceForDeletion(temp, false);
178  }
179 
180  delete output_handle_;
181  output_handle_ = buffer;
182  handle_ = 0;
183 
184  if (output_handle_) {
185  handler->CopyHandle(output_handle_, &handle_);
186  }
187  }
188 
189  if (acquire_fence_ > 0) {
190  close(acquire_fence_);
191  acquire_fence_ = -1;
192  }
193 
194  if (acquire_fence > 0) {
195  acquire_fence_ = dup(acquire_fence);
196  }
197 }

205 bool VirtualDisplay::GetDisplayAttribute(uint32_t /*config*/,
206  HWCDisplayAttribute attribute,
207  int32_t *value) {
208  // We always get the values from preferred mode config.
209  switch (attribute) {
210  case HWCDisplayAttribute::kWidth:
211  *value = width_;
212  break;
213  case HWCDisplayAttribute::kHeight:
214  *value = height_;
215  break;
216  case HWCDisplayAttribute::kRefreshRate:
217  // in nanoseconds
218  *value = 16666666;
219  break;
220  case HWCDisplayAttribute::kDpiX:
221  // Dots per 1000 inches
222  *value = 1;
223  break;
224  case HWCDisplayAttribute::kDpiY:
225  // Dots per 1000 inches
226  *value = 1;
227  break;
228  default:
229  *value = -1;
230  return false;
231  }
232 
233  return true;
234 }
