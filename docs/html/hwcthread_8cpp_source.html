73 void HWCThread::HandleWait() {

75  ETRACE("Poll Failed in DisplayManager %s", PRINTERROR());
76  return;
77  }
78 
79  if (fd_handler_.IsReady(event_.get_fd())) {

86 void HWCThread::ProcessThread() {
87  setpriority(PRIO_PROCESS, 0, priority_);
88  prctl(PR_SET_NAME, name_.c_str());
89 
90  while (1) {

94  fd_handler_.RemoveFd(event_.get_fd());
