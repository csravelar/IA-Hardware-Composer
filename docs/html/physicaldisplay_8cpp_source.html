142 void PhysicalDisplay::Connect() {
143  SPIN_LOCK(modeset_lock_);
144 
145  connection_state_ &= ~kDisconnectionInProgress;
146 
147  SPIN_UNLOCK(modeset_lock_);
148 
149  if (source_display_) {
150  // Current display is a cloned display, set the source_display_'s
151  // k_RefreshClonedDisplays flag. This makes clone parent have a
152  // chance to update it's cloned display list
153  PhysicalDisplay *p_clone_parent = (PhysicalDisplay *)source_display_;
154  SPIN_LOCK(p_clone_parent->modeset_lock_);
155  p_clone_parent->display_state_ |= kRefreshClonedDisplays;
156  SPIN_UNLOCK(p_clone_parent->modeset_lock_);
157  }
158 
159  SPIN_LOCK(modeset_lock_);
160  if (connection_state_ & kConnected) {
161  SPIN_UNLOCK(modeset_lock_);
162  return;
163  }
164 
165  connection_state_ |= kConnected;
166  display_state_ &= ~kInitialized;
167  display_state_ |= kNotifyClient;
168  IHOTPLUGEVENTTRACE("PhysicalDisplay::Connect recieved. %p \n", this);
169 
170  if (!display_queue_->Initialize(pipe_, width_, height_, this, fb_manager_)) {
171  ETRACE("Failed to initialize Display Queue.");
172  } else {
173  display_state_ |= kInitialized;
174  }
175 
176  if (display_state_ & kUpdateConfig) {
177  display_state_ &= ~kUpdateConfig;
178  display_queue_->DisplayConfigurationChanged();
179  UpdateDisplayConfig();
180  }
181 
182  UpdatePowerMode();
183 
184  SPIN_UNLOCK(modeset_lock_);
185 }

230 bool PhysicalDisplay::SetPowerMode(uint32_t power_mode) {
231 #ifndef DISABLE_HOTPLUG_NOTIFICATION
232  ScopedSpinLock lock(modeset_lock_);
233 #endif
234 
235  if (power_mode_ == power_mode) {
236  return true;
237  }
238 
239  power_mode_ = power_mode;
240  if (!(connection_state_ & kConnected)) {
241  IHOTPLUGEVENTTRACE(
242  "PhysicalDisplay is not connected, postponing power mode update.");
243  display_state_ |= kPendingPowerMode;
244  return true;
245  } else if (connection_state_ & kDisconnectionInProgress) {
246  IHOTPLUGEVENTTRACE(
247  "PhysicalDisplay diconnection in progress, postponing power mode "
248  "update.");
249  // Don't update power mode in case disconnect is in
250  // progress.
251  display_state_ |= kPendingPowerMode;
252  return true;
253  }
254 
255  return UpdatePowerMode();
256 }

258 bool PhysicalDisplay::UpdatePowerMode() {
259  display_state_ &= ~kPendingPowerMode;
260 
261  if (power_mode_ == kOn) {
262  display_state_ |= kNeedsModeset;
263  display_state_ |= kUpdateDisplay;
264  IHOTPLUGEVENTTRACE(
265  "UpdatePowerMode: Powering on Display: pipe: %d display: "
266  "%p",
267  pipe_, this);
268  PowerOn();
269  } else {
270  IHOTPLUGEVENTTRACE(
271  "UpdatePowerMode: Power mode is not kOn: pipe: %d display: "
272  "%p",
273  pipe_, this);
274  display_state_ &= ~kUpdateDisplay;
275  }
276 
277  if (!(display_state_ & kInitialized))
278  return true;
279 
280  return display_queue_->SetPowerMode(power_mode_);
281 }

283 bool PhysicalDisplay::Present(std::vector<HwcLayer *> &source_layers,
284  int32_t *retire_fence,
285  PixelUploaderCallback *call_back,
286  bool handle_constraints) {
287  CTRACE();
288  SPIN_LOCK(modeset_lock_);
289 
290  bool handle_hotplug_notifications = false;
291  if (display_state_ & kHandlePendingHotPlugNotifications) {
292  display_state_ &= ~kHandlePendingHotPlugNotifications;
293  handle_hotplug_notifications = true;
294  }
295 
296  if (!(display_state_ & kUpdateDisplay)) {
297  bool success = true;
298  if (power_mode_ != kDozeSuspend) {
299  ETRACE("Trying to update an Disconnected Display.%p \n", this);
300  }
301 
302  SPIN_UNLOCK(modeset_lock_);
303 
304  if (handle_hotplug_notifications) {
305  NotifyClientsOfDisplayChangeStatus();
306  }
307 
308  return success;
309  }
310 
311  if (source_display_) {
312  ETRACE("Trying to update display independently when in cloned mode.%p \n",
313  this);
314  }
315 
316  if (display_state_ & kRefreshClonedDisplays) {
317  RefreshClones();
318  }
319 
320  SPIN_UNLOCK(modeset_lock_);
321 
322  if (handle_hotplug_notifications) {
323  NotifyClientsOfDisplayChangeStatus();
324  IHOTPLUGEVENTTRACE("Handle_hoplug_notifications done. %p \n", this);
325  }
326 
327  bool ignore_clone_update = false;
328  bool success = display_queue_->QueueUpdate(source_layers, retire_fence,
329  &ignore_clone_update, call_back,
330  handle_constraints);
331  if (success && !clones_.empty() && !ignore_clone_update) {
332  HandleClonedDisplays(this);
333  }
334 
335  size_t size = source_layers.size();
336  for (size_t layer_index = 0; layer_index < size; layer_index++) {
337  HwcLayer *layer = source_layers.at(layer_index);
338  if (!layer->IsVisible())
339  continue;
340 
341  layer->Validate();
342  }
343 
344  return success;
345 }

510  if (cloned_displays_.empty())

514  size_t size = cloned_displays_.size();

535 void PhysicalDisplay::RefreshClones() {
536  display_state_ &= ~kRefreshClonedDisplays;
537  std::vector<NativeDisplay *>().swap(clones_);
538  if (cloned_displays_.empty())
539  return;
540 
541  size_t size = cloned_displays_.size();
542  for (size_t i = 0; i < size; i++) {
543  NativeDisplay *display = cloned_displays_.at(i);
544  if (!display->IsConnected())
545  continue;
546 
547  clones_.emplace_back(display);
548  }
549 
550  uint32_t primary_width = Width();
551  uint32_t primary_height = Height();
552  for (auto display : clones_) {
553  uint32_t display_width = display->Width();
554  uint32_t display_height = display->Height();
555  if ((primary_width == display_width) && (primary_height == display_height))
556  return;
557 
558  display->UpdateScalingRatio(primary_width, primary_height, display_width,
559  display_height);
560  }
561 }
