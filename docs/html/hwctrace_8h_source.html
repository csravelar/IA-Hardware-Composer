48 #ifdef FUNCTION_CALL_TRACING
49 class TraceFunc {
50  public:
51  explicit TraceFunc(std::string file_name, std::string func_name) {
52  file_name_ = file_name;
53  func_name_ = func_name;
54  ITRACE("Calling ----- %s: %s", file_name_.c_str(), func_name_.c_str());
55  t_ = std::chrono::high_resolution_clock::now();
56  }
57  ~TraceFunc() {
58  std::chrono::high_resolution_clock::time_point t2 =
59  std::chrono::high_resolution_clock::now();
60  ITRACE(
61  "Total time spent in --- %s Time(msec): %lld", func_name_.c_str(),
62  std::chrono::duration_cast<std::chrono::microseconds>(t2 - t_).count());
63  ITRACE("Leaving --- %s: %s", file_name_.c_str(), func_name_.c_str());
64  }
65 
66  private:
67  std::chrono::high_resolution_clock::time_point t_;
68  std::string func_name_;
69  std::string file_name_;
70 };
