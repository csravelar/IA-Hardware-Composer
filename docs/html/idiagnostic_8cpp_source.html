110 status_t BnDiagnostic::onTransact(uint32_t code, const Parcel& data,
111  Parcel* reply, uint32_t flags) {
112  switch (code) {
113  case BpDiagnostic::TRANSACT_READ_LOG_PARCEL: {
114  CHECK_INTERFACE(IDiagnostic, data, reply);
115  status_t err = ReadLogParcel(reply);
116  return err;
117  }
118 
119  case BpDiagnostic::TRANSACT_ENABLE_DISPLAY: {
120  CHECK_INTERFACE(IDiagnostic, data, reply);
121  uint32_t d = data.readInt32();
122  EnableDisplay(d);
123  return NO_ERROR;
124  }
125 
126  case BpDiagnostic::TRANSACT_DISABLE_DISPLAY: {
127  CHECK_INTERFACE(IDiagnostic, data, reply);
128  uint32_t d = data.readInt32();
129  bool bBlank = data.readInt32();
130  DisableDisplay(d, bBlank);
131  return NO_ERROR;
132  }
133 
134  case BpDiagnostic::TRANSACT_MASK_LAYER: {
135  CHECK_INTERFACE(IDiagnostic, data, reply);
136  uint32_t d = data.readInt32();
137  uint32_t layer = data.readInt32();
138  bool bHide = data.readInt32();
139  MaskLayer(d, layer, bHide);
140  return NO_ERROR;
141  }
142 
143  case BpDiagnostic::TRANSACT_DUMP_FRAMES: {
144  CHECK_INTERFACE(IDiagnostic, data, reply);
145  uint32_t d = data.readInt32();
146  int32_t frames = data.readInt32();
147  bool bSync = data.readInt32();
148  DumpFrames(d, frames, bSync);
149  return NO_ERROR;
150  }
151 
152  default:
153  return BBinder::onTransact(code, data, reply, flags);
154  }
155 }
