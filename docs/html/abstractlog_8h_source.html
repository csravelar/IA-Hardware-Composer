31 // This is primarily a debug logging class expected to generate data thats
32 // expected
33 // to be used by the validation team to check that the HWC is operating
34 // correctly.
35 class AbstractLogWrite {
36  public:
37  virtual char* reserve(uint32_t maxSize) = 0;
38  virtual void log(char* endPtr) = 0;
39 
40  static const size_t cStrOffset = sizeof(pid_t) + sizeof(uint64_t);
41 
42  private:
43  // Helper template to write the specified value to the byte buffer and advance
44  // the pointer to the next byte after that value
45  template <typename T>
46  void serialize(char*& pBuffer, const T value) {
47  *(reinterpret_cast<T*>(pBuffer)) = value;
48  pBuffer += (sizeof(T));
49  }
50 
51  // Helper template to read the specified value from the byte buffer, return
52  // the value and advance the pointer to the next byte after that value
53  template <typename T>
54  const T& unserialize(const char*& ptr) {
55  T* data = (T*)ptr;
56  ptr += sizeof(T);
57  return *data;
58  }
59 
60  public:
61  virtual ~AbstractLogWrite() {
62  }
63 
64  // Defined here rather than in .cpp so they can be used by validation
65  // subclass.
66  // The alternative would be a library.
67  const char* addV(const char* description, va_list& args) {
68  // Determine space requirement to flatten the layers
69  size_t logAllocSize = sizeof(int);
70 
71  // ... and an amount for the tid, time, and a fixed allocation for the
72  // string
73  logAllocSize += sizeof(pid_t) + sizeof(uint64_t) + strlen(description) +
74  HWCLOG_STRING_RESERVATION_SIZE;
75 
76  // Allocate a log entry with enough space for all that
77  char* entry = reserve(logAllocSize);
78 
79  if (entry == 0) {
80  return "";
81  }
82 
83  char* ptr = entry;
84 
85  // Write the tid
86  pid_t threadid = gettid();
87  serialize(ptr, threadid);
88 
89  // Write the time
90  nsecs_t timestamp = systemTime(CLOCK_MONOTONIC);
91  serialize(ptr, timestamp);
92 
93  // Write the formatted string
94  int maxlen = entry + logAllocSize - ptr;
95  int len = vsnprintf(ptr, maxlen, description, args) + 1;
96  // NB string will be null-terminated within maxlen characters
97 
98  if (len > maxlen) {
99  len = maxlen;
100  }
101 
102  log(ptr + len);
103  return ptr;
104  }
105 
106  // Defined here rather than in .cpp so they can be used by validation
107  // subclass.
108  // The alternative would be a library.
109  const char* add(const char* fmt, ...) {
110  va_list args;
111  va_start(args, fmt);
112  const char* ptr = addV(fmt, args);
113  va_end(args);
114  return ptr;
115  }
116 
117  const char* unpack(const char* ptr, pid_t& pid, int64_t& timestamp) {
118  pid = unserialize<pid_t>(ptr);
119  timestamp = unserialize<int64_t>(ptr);
120  return ptr;
121  }
122 
123  protected:
124  void logToLogcat(const char* ptr) {
125  pid_t threadid;
126  nsecs_t timestamp;
127  const char* str = unpack(ptr, threadid, timestamp);
128  // Long multi line strings to logcat get truncated when they are too large.
129  // Hence, break up the output into one logcat entry per line of text.
130  std::string inputstr(str);
131  std::string delimiter("\n");
132  size_t pos = 0;
133  std::string token;
134  while ((pos = inputstr.find(delimiter)) != std::string::npos) {
135  token = inputstr.substr(0, pos);
136  ALOGI("%s", token.c_str());
137  inputstr.erase(0, pos + delimiter.length());
138  }
139  }
140 };
