39 class DisplayTimeLine {
40  public:
41  int Init() {
42  timeline_fd_ = open("/dev/sw_sync", O_RDWR);
43  if (timeline_fd_ < 0)
44  return -1;
45 
46  return 0;
47  }
48 
49  ~DisplayTimeLine() {
50  if (timeline_fd_ > 0) {
51  close(timeline_fd_);
52  }
53  }
54 
55  int32_t IncrementTimeLine() {
56  int ret =
57  sw_sync_fence_create(timeline_fd_, "display fence", timeline_pt_ + 1);
58  if (ret < 0) {
59  ALOGE("Failed to create display fence %d %d", ret, timeline_fd_);
60  return ret;
61  }
62 
63  int32_t ret_fd(ret);
64 
65  ret = sw_sync_timeline_inc(timeline_fd_, 1);
66  if (ret) {
67  ALOGE("Failed to increment display sync timeline %d", ret);
68  return ret;
69  }
70 
71  ++timeline_pt_;
72  return ret_fd;
73  }
74 
75  private:
76  int32_t timeline_fd_;
77  int timeline_pt_ = 0;
78 };

171 int IAHwc1Layer::InitFromHwcLayer(hwc_layer_1_t *sf_layer) {
172  if (!hwc_layer_) {
173  hwc_layer_ = new hwcomposer::HwcLayer();
174  }
175 
176  bool surface_damage = true;
177 
178  if (hwc_layer_->GetNativeHandle() &&
179  (hwc_layer_->GetNativeHandle()->handle_ == sf_layer->handle))
180  surface_damage = false;
181 
182  native_handle_.handle_ = sf_layer->handle;
183  hwc_layer_->SetNativeHandle(&native_handle_);
184  hwc_layer_->SetAlpha(sf_layer->planeAlpha);
185  hwc_layer_->SetSourceCrop(hwcomposer::HwcRect<float>(
186  sf_layer->sourceCropf.left, sf_layer->sourceCropf.top,
187  sf_layer->sourceCropf.right, sf_layer->sourceCropf.bottom));
188  hwc_layer_->SetDisplayFrame(hwcomposer::HwcRect<int>(
189  sf_layer->displayFrame.left, sf_layer->displayFrame.top,
190  sf_layer->displayFrame.right, sf_layer->displayFrame.bottom));
191 
192  uint32_t transform = 0;
193  if (sf_layer->transform == HWC_TRANSFORM_ROT_270) {
194  transform = hwcomposer::HWCTransform::kTransform270;
195  } else if (sf_layer->transform == HWC_TRANSFORM_ROT_180) {
196  transform = hwcomposer::HWCTransform::kTransform180;
197  } else {
198  if (sf_layer->transform & HWC_TRANSFORM_FLIP_H)
199  transform |= hwcomposer::HWCTransform::kReflectX;
200  if (sf_layer->transform & HWC_TRANSFORM_FLIP_V)
201  transform |= hwcomposer::HWCTransform::kReflectY;
202  if (sf_layer->transform & HWC_TRANSFORM_ROT_90)
203  transform |= hwcomposer::HWCTransform::kTransform90;
204  }
205 
206  hwc_layer_->SetTransform(transform);
207  hwc_layer_->SetAcquireFence(dup(sf_layer->acquireFenceFd));
208 
209  switch (sf_layer->blending) {
210  case HWC_BLENDING_NONE:
211  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingNone);
212  break;
213  case HWC_BLENDING_PREMULT:
214  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingPremult);
215  break;
216  case HWC_BLENDING_COVERAGE:
217  hwc_layer_->SetBlending(hwcomposer::HWCBlending::kBlendingCoverage);
218  break;
219  default:
220  ALOGE("Invalid blending in hwc_layer_1_t %d", sf_layer->blending);
221  return -EINVAL;
222  }
223 
224  if (surface_damage) {
225  uint32_t num_rects = sf_layer->surfaceDamage.numRects;
226  hwcomposer::HwcRegion hwc_region;
227 
228  for (size_t rect = 0; rect < num_rects; ++rect) {
229  hwc_region.emplace_back(sf_layer->surfaceDamage.rects[rect].left,
230  sf_layer->surfaceDamage.rects[rect].top,
231  sf_layer->surfaceDamage.rects[rect].right,
232  sf_layer->surfaceDamage.rects[rect].bottom);
233  }
234 
235  hwc_layer_->SetSurfaceDamage(hwc_region);
236  } else {
237  hwcomposer::HwcRegion hwc_region;
238  hwc_region.emplace_back(0, 0, 0, 0);
239  hwc_layer_->SetSurfaceDamage(hwc_region);
240  }
241 
242  uint32_t num_rects = sf_layer->visibleRegionScreen.numRects;
243  hwcomposer::HwcRegion visible_region;
244 
245  for (size_t rect = 0; rect < num_rects; ++rect) {
246  visible_region.emplace_back(
247  sf_layer->visibleRegionScreen.rects[rect].left,
248  sf_layer->visibleRegionScreen.rects[rect].top,
249  sf_layer->visibleRegionScreen.rects[rect].right,
250  sf_layer->visibleRegionScreen.rects[rect].bottom);
251  }
252 
253  hwc_layer_->SetVisibleRegion(visible_region);
254  return 0;
255 }

273 static int hwc_prepare(hwc_composer_device_1_t *dev, size_t num_displays,
274  hwc_display_contents_1_t **display_contents) {
275  struct hwc_context_t *ctx = (struct hwc_context_t *)&dev->common;
276  int total_displays = (int)num_displays;
277  bool disable_overlays = ctx->disable_explicit_sync_;
278 
279  for (int i = 0; i < total_displays; ++i) {
280  if (!display_contents[i])
281  continue;
282 
283  if (i == HWC_DISPLAY_VIRTUAL) {
284  disable_overlays = true;
285  } else {
286  disable_overlays = ctx->disable_explicit_sync_;
287  }
288 
289  int num_layers = display_contents[i]->numHwLayers;
290  HwcDisplay *native_display = GetDisplay(ctx, i);
291  native_display->gl_composition_ = disable_overlays;
292 
293  for (int j = 0; j < num_layers; ++j) {
294  hwc_layer_1_t *layer = &display_contents[i]->hwLayers[j];
295 
296  if (!disable_overlays) {
297  switch (layer->compositionType) {
298  case HWC_BACKGROUND:
299  case HWC_SIDEBAND:
300  layer->compositionType = HWC_FRAMEBUFFER;
301  native_display->gl_composition_ = true;
302  break;
303  case HWC_FRAMEBUFFER_TARGET:
304  break;
305  default:
306  layer->compositionType = HWC_OVERLAY;
307  break;
308  }
309  } else {
310  layer->compositionType = HWC_FRAMEBUFFER;
311  }
312  }
313  }
314 
315  return 0;
316 }

318 static int hwc_set(hwc_composer_device_1_t *dev, size_t num_displays,
319  hwc_display_contents_1_t **sf_display_contents) {
320  ATRACE_CALL();
321  struct hwc_context_t *ctx = (struct hwc_context_t *)&dev->common;
322  for (size_t i = 0; i < num_displays; ++i) {
323  hwc_display_contents_1_t *dc = sf_display_contents[i];
324  if (!dc || i == HWC_DISPLAY_VIRTUAL)
325  continue;
326 
327  size_t num_dc_layers = dc->numHwLayers;
328  HwcDisplay *native_display = GetDisplay(ctx, i);
329  dc->retireFenceFd = native_display->timeline_.IncrementTimeLine();
330  hwcomposer::NativeDisplay *display = native_display->display_;
331  std::vector<IAHwc1Layer *> &old_layers = native_display->layers_;
332  std::vector<IAHwc1Layer *> new_layers;
333  size_t size = old_layers.size();
334  std::vector<hwcomposer::HwcLayer *> source_layers;
335  for (size_t j = 0; j < num_dc_layers; ++j) {
336  hwc_layer_1_t *sf_layer = &dc->hwLayers[j];
337  if (!sf_layer || !sf_layer->handle || (sf_layer->flags & HWC_SKIP_LAYER))
338  continue;
339 
340  if (!native_display->gl_composition_ &&
341  (sf_layer->compositionType == HWC_FRAMEBUFFER_TARGET)) {
342  continue;
343  }
344 
345  IAHwc1Layer *new_layer = new IAHwc1Layer();
346  if (size > j) {
347  IAHwc1Layer *old_layer = old_layers.at(j);
348  new_layer->hwc_layer_ = old_layer->hwc_layer_;
349  old_layer->hwc_layer_ = NULL;
350  }
351 
352  new_layer->InitFromHwcLayer(sf_layer);
353  source_layers.emplace_back(new_layer->hwc_layer_);
354  new_layer->index_ = j;
355  new_layers.emplace_back(new_layer);
356  sf_layer->acquireFenceFd = -1;
357  sf_layer->releaseFenceFd = -1;
358  }
359 
360  if (source_layers.empty()) {
361  return 0;
362  }
363 
364  int32_t retire_fence = -1;
365  old_layers.swap(new_layers);
366  size = new_layers.size();
367  for (size_t i = 0; i < size; i++) {
368  IAHwc1Layer *layer = new_layers.at(i);
369  delete layer;
370  }
371 
372  std::vector<IAHwc1Layer *>().swap(new_layers);
373 
374  bool success = display->Present(source_layers, &retire_fence);
375  if (!success) {
376  ALOGE("Failed to set layers in the composition");
377  return -1;
378  }
379 
380  if (retire_fence > 0)
381  close(retire_fence);
382 
383  size = old_layers.size();
384  for (size_t i = 0; i < size; i++) {
385  hwcomposer::HwcLayer *layer = old_layers.at(i)->hwc_layer_;
386  int32_t release_fence = layer->GetReleaseFence();
387  if (release_fence <= 0)
388  continue;
389 
390  hwc_layer_1_t *sf_layer = &dc->hwLayers[old_layers.at(i)->index_];
391  sf_layer->releaseFenceFd = release_fence;
392  }
393 
394  std::vector<hwcomposer::HwcLayer *>().swap(source_layers);
395  }
396 
397  return 0;
398 }

577 static int hwc_device_open(const struct hw_module_t *module, const char *name,
578  struct hw_device_t **dev) {
579  if (strcmp(name, HWC_HARDWARE_COMPOSER)) {
580  ALOGE("Invalid module name- %s", name);
581  return -EINVAL;
582  }
583 
584  std::unique_ptr<hwc_context_t> ctx(new hwc_context_t());
585  if (!ctx) {
586  ALOGE("Failed to allocate hwc context");
587  return -ENOMEM;
588  }
589 
590  char value[PROPERTY_VALUE_MAX];
591  property_get("board.disable.explicit.sync", value, "0");
592  ctx->disable_explicit_sync_ = atoi(value);
593  if (ctx->disable_explicit_sync_)
594  ALOGI("EXPLICIT SYNC support is disabled");
595  else
596  ALOGI("EXPLICIT SYNC support is enabled");
597 
598  if (!ctx->device_.Initialize()) {
599  ALOGE("Can't initialize drm object.");
600  return -1;
601  }
602 
603  const std::vector<hwcomposer::NativeDisplay *> &displays =
604  ctx->device_.GetAllDisplays();
605  ctx->virtual_display_.display_ = ctx->device_.GetVirtualDisplay();
606  ctx->virtual_display_.display_->SetExplicitSyncSupport(
607  ctx->disable_explicit_sync_);
608  ctx->virtual_display_.timeline_.Init();
609 
610  size_t size = displays.size();
611  hwcomposer::NativeDisplay *primary_display = displays.at(0);
612  ctx->primary_display_.display_ = primary_display;
613  ctx->primary_display_.display_id_ = 0;
614  ctx->primary_display_.display_->SetExplicitSyncSupport(
615  ctx->disable_explicit_sync_);
616  ctx->primary_display_.timeline_.Init();
617  // Fetch the number of modes from the display
618  uint32_t num_configs;
619  uint32_t default_config;
620  if (!primary_display->GetDisplayConfigs(&num_configs, NULL))
621  return -1;
622 
623  // Grab the first mode, we'll choose this as the active mode
624  num_configs = 1;
625  if (!primary_display->GetDisplayConfigs(&num_configs, &default_config))
626  return -1;
627 
628  if (!primary_display->SetActiveConfig(default_config)) {
629  ALOGE("Could not find active mode for %d", default_config);
630  return -1;
631  }
632 
633  for (size_t i = 1; i < size; ++i) {
634  ctx->extended_displays_.emplace_back();
635  HwcDisplay &temp = ctx->extended_displays_.back();
636  temp.display_ = displays.at(i);
637  temp.display_id_ = i;
638  temp.timeline_.Init();
639  temp.display_->SetExplicitSyncSupport(ctx->disable_explicit_sync_);
