98 status_t LogEntry::read(sp<IDiagnostic> pDiagnostic) {
99  status_t ret = NO_ERROR;
100 
101  static Parcel* spReply = NULL;
102  if (spReply) {
103  ret = spReply->readInt32();
104 
105  if (ret == NOT_ENOUGH_DATA) {
106  // Marks end of parcel, time to fetch a new one
107  delete spReply;
108  spReply = 0;
109  }
110  }
111 
112  if (spReply == 0) {
113  spReply = new Parcel;
114  pDiagnostic = NULL;
115  // ret = pDiagnostic->readLogParcel(spReply);
116 
117  if (ret >= 0)
118  ret = spReply->readInt32();
119  }
120 
121  if (ret < 0) {
122  delete spReply;
123  spReply = 0;
124  return ret;
125  }
126 
127  spReply->read(*this);
128  return ret;
129 }

131 void LogEntry::print(bool bVeryVerbose, bool bVerbose, bool bFences,
132  bool bBufferManager, bool bQueue) {
133  if (strstr(getDescription(), "SF0 onPrepare Entry") ||
134  strstr(getDescription(), "InputAnalyzer SF0")) {
135  printf("\n\n");
136  }
137 
138  if (!bVeryVerbose && bVerbose == false) {
139  // Strip out any 'verbose' matching strings
140  if (strstr(getDescription(), "onPrepare Entry") ||
141  strstr(getDescription(), "onPrepare Exit") ||
142  strstr(getDescription(), "onSet Exit") ||
143  strncmp(getDescription(), "InternalBuffer", 14) == 0 ||
144  strncmp(getDescription(), "drm", 3) == 0 ||
145  strncmp(getDescription(), "adf", 3) == 0)
146  return;
147  }
148  if (!bVeryVerbose && bFences == false) {
149  // Strip out any 'fence' matching strings.
150  if (strncmp(getDescription(), "Fence:", 6) == 0)
151  return;
152  if (strncmp(getDescription(), "NativeFence:", 12) == 0)
153  return;
154  }
155  if (!bVeryVerbose && bBufferManager == false) {
156  // Strip out any 'buffer manager' matching strings.
157  if (strncmp(getDescription(), "BufferManager:", 14) == 0)
158  return;
159  }
160  if (!bVeryVerbose && bQueue == false) {
161  // Strip out any 'display queue' matching strings.
162  if (strncmp(getDescription(), "Queue:", 6) == 0)
163  return;
164  }
165 
166  printf("%" PRIi64 "s %03" PRIi64 "ms", getTimestamp() / 1000000000,
167  (getTimestamp() % 1000000000) / 1000000);
168  if (bVerbose) {
169  printf(" %06" PRIi64 "ns", getTimestamp() % 1000000);
170  printf(" TID:%d", getTID());
171  }
172  printf(" %s\n", getDescription().string());
173 }
