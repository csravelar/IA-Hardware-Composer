213 void VKProgram::UseProgram(const RenderState &state,
214  unsigned int viewport_width,
215  unsigned int viewport_height) {
216  unsigned layer_count = state.layer_state_.size();
217 
218  size_t vert_ub_size = 4 + 12 * layer_count;
219  RingBuffer::Allocation vert_ub_alloc =
220  ring_buffer_.Allocate(vert_ub_size * sizeof(float), ub_offset_align_);
221  if (!vert_ub_alloc) {
222  ETRACE("Failed to allocate space for vert uniform buffer");
223  return;
224  }
225  float *vert_ub = vert_ub_alloc.get<float>();
226 
227  vert_ub[0] = state.x_ / (float)viewport_width;
228  vert_ub[1] = state.y_ / (float)viewport_height;
229  vert_ub[2] = state.width_ / (float)viewport_width;
230  vert_ub[3] = state.height_ / (float)viewport_height;
231  vert_ub += 4;
232 
233  for (unsigned src_index = 0; src_index < layer_count; src_index++) {
234  const RenderState::LayerState &src = state.layer_state_[src_index];
235 
236  vert_ub[0] = src.crop_bounds_[0];
237  vert_ub[1] = src.crop_bounds_[1];
238  vert_ub[2] = src.crop_bounds_[2] - src.crop_bounds_[0];
239  vert_ub[3] = src.crop_bounds_[3] - src.crop_bounds_[1];
240  vert_ub += 4;
241 
242  std::copy_n(src.texture_matrix_, 2, vert_ub);
243  vert_ub += 4; // 2 data + 2 padding
244  std::copy_n(src.texture_matrix_ + 2, 2, vert_ub);
245  vert_ub += 4; // 2 data + 2 padding
246  }
247 
248  size_t frag_ub_size = 4 * layer_count;
249  RingBuffer::Allocation frag_ub_alloc =
250  ring_buffer_.Allocate(frag_ub_size * sizeof(float), ub_offset_align_);
251  if (!frag_ub_alloc) {
252  ETRACE("failed to allocate space for frag uniform buffer");
253  return;
254  }
255  float *frag_ub = frag_ub_alloc.get<float>();
256 
257  for (unsigned src_index = 0; src_index < layer_count; src_index++) {
258  frag_ub[0] = state.layer_state_[src_index].alpha_;
259  frag_ub[1] = state.layer_state_[src_index].premult_;
260  frag_ub += 4; // 2 data + 2 padding
261  }
262 
263  vert_buf_info_ = {};
264  vert_buf_info_.buffer = uniform_buffer_;
265  vert_buf_info_.offset = vert_ub_alloc.offset();
266  vert_buf_info_.range = vert_ub_size * sizeof(float);
267 
268  frag_buf_info_ = {};
269  frag_buf_info_.buffer = uniform_buffer_;
270  frag_buf_info_.offset = frag_ub_alloc.offset();
271  frag_buf_info_.range = frag_ub_size * sizeof(float);
272 
273  for (unsigned src_index = 0; src_index < layer_count; src_index++) {
274  const RenderState::LayerState &src = state.layer_state_[src_index];
275 
276  VkDescriptorImageInfo image_info = {};
277  image_info.sampler = sampler_;
278  image_info.imageView = src.handle_.image_view;
279  image_info.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
280  src_image_infos_.emplace_back(image_info);
281  }
282 
283  ub_allocs_.emplace_back(std::move(vert_ub_alloc));
284  ub_allocs_.emplace_back(std::move(frag_ub_alloc));
285 }
