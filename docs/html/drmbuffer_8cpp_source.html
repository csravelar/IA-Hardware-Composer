80 void DrmBuffer::Initialize(const HwcBuffer& bo) {
81  width_ = bo.width_;
82  height_ = bo.height_;
83  for (uint32_t i = 0; i < 4; i++) {
84  pitches_[i] = bo.pitches_[i];
85  offsets_[i] = bo.offsets_[i];
86  gem_handles_[i] = bo.gem_handles_[i];
87  }
88 
89  format_ = bo.format_;
90  if (format_ == DRM_FORMAT_NV12_Y_TILED_INTEL || format_ == DRM_FORMAT_NV21)
91  format_ = DRM_FORMAT_NV12;
92  else if (format_ == DRM_FORMAT_YVU420_ANDROID)
93  format_ = DRM_FORMAT_YUV420;
94 
95  tiling_mode_ = bo.tiling_mode_;
96  usage_ = bo.usage_;
97 
98  if (usage_ == hwcomposer::kLayerCursor) {
99  // We support DRM_FORMAT_ARGB8888 for cursor.
100  frame_buffer_format_ = DRM_FORMAT_ARGB8888;
101  } else {
102  frame_buffer_format_ = format_;
103  }
104 
105  fb_manager_->RegisterGemHandles(image_.handle_->meta_data_.num_planes_,
106  image_.handle_->meta_data_.gem_handles_);
107 }

128 const ResourceHandle& DrmBuffer::GetGpuResource(GpuDisplay egl_display,
129  bool external_import) {
130 #if USE_GL
131  if (image_.image_ == 0) {
132  EGLImageKHR image = EGL_NO_IMAGE_KHR;
133  uint32_t total_planes = image_.handle_->meta_data_.num_planes_;
134  // Note: If eglCreateImageKHR is successful for a EGL_LINUX_DMA_BUF_EXT
135  // target, the EGL will take a reference to the dma_buf.
136  if ((usage_ == kLayerVideo) && total_planes > 1) {
137  if (total_planes == 2) {
138  const EGLint attr_list_nv12[] = {
139  EGL_WIDTH,
140  static_cast<EGLint>(width_),
141  EGL_HEIGHT,
142  static_cast<EGLint>(height_),
143  EGL_LINUX_DRM_FOURCC_EXT,
144  static_cast<EGLint>(format_),
145  EGL_DMA_BUF_PLANE0_FD_EXT,
146  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[0]),
147  EGL_DMA_BUF_PLANE0_PITCH_EXT,
148  static_cast<EGLint>(pitches_[0]),
149  EGL_DMA_BUF_PLANE0_OFFSET_EXT,
150  static_cast<EGLint>(offsets_[0]),
151  EGL_DMA_BUF_PLANE1_FD_EXT,
152  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[1]),
153  EGL_DMA_BUF_PLANE1_PITCH_EXT,
154  static_cast<EGLint>(pitches_[1]),
155  EGL_DMA_BUF_PLANE1_OFFSET_EXT,
156  static_cast<EGLint>(offsets_[1]),
157  EGL_NONE,
158  0};
159  image = eglCreateImageKHR(
160  egl_display, EGL_NO_CONTEXT, EGL_LINUX_DMA_BUF_EXT,
161  static_cast<EGLClientBuffer>(nullptr), attr_list_nv12);
162  } else {
163  const EGLint attr_list_yv12[] = {
164  EGL_WIDTH,
165  static_cast<EGLint>(width_),
166  EGL_HEIGHT,
167  static_cast<EGLint>(height_),
168  EGL_LINUX_DRM_FOURCC_EXT,
169  static_cast<EGLint>(format_),
170  EGL_DMA_BUF_PLANE0_FD_EXT,
171  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[0]),
172  EGL_DMA_BUF_PLANE0_PITCH_EXT,
173  static_cast<EGLint>(pitches_[0]),
174  EGL_DMA_BUF_PLANE0_OFFSET_EXT,
175  static_cast<EGLint>(offsets_[0]),
176  EGL_DMA_BUF_PLANE1_FD_EXT,
177  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[1]),
178  EGL_DMA_BUF_PLANE1_PITCH_EXT,
179  static_cast<EGLint>(pitches_[1]),
180  EGL_DMA_BUF_PLANE1_OFFSET_EXT,
181  static_cast<EGLint>(offsets_[1]),
182  EGL_DMA_BUF_PLANE2_FD_EXT,
183  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[2]),
184  EGL_DMA_BUF_PLANE2_PITCH_EXT,
185  static_cast<EGLint>(pitches_[2]),
186  EGL_DMA_BUF_PLANE2_OFFSET_EXT,
187  static_cast<EGLint>(offsets_[2]),
188  EGL_NONE,
189  0};
190  image = eglCreateImageKHR(
191  egl_display, EGL_NO_CONTEXT, EGL_LINUX_DMA_BUF_EXT,
192  static_cast<EGLClientBuffer>(nullptr), attr_list_yv12);
193  }
194  } else if (image_.handle_->meta_data_.fb_modifiers_[0] > 0 &&
195  total_planes == 2) {
196  EGLint modifier_low =
197  static_cast<EGLint>(image_.handle_->meta_data_.fb_modifiers_[1]);
198  EGLint modifier_high =
199  static_cast<EGLint>(image_.handle_->meta_data_.fb_modifiers_[0]);
200  const EGLint image_attrs[] = {
201  EGL_WIDTH,
202  static_cast<EGLint>(width_),
203  EGL_HEIGHT,
204  static_cast<EGLint>(height_),
205  EGL_LINUX_DRM_FOURCC_EXT,
206  static_cast<EGLint>(format_),
207  EGL_DMA_BUF_PLANE0_FD_EXT,
208  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[0]),
209  EGL_DMA_BUF_PLANE0_PITCH_EXT,
210  static_cast<EGLint>(pitches_[0]),
211  EGL_DMA_BUF_PLANE0_OFFSET_EXT,
212  static_cast<EGLint>(offsets_[0]),
213  EGL_DMA_BUF_PLANE0_MODIFIER_LO_EXT,
214  modifier_low,
215  EGL_DMA_BUF_PLANE0_MODIFIER_HI_EXT,
216  modifier_high,
217  EGL_DMA_BUF_PLANE1_FD_EXT,
218  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[1]),
219  EGL_DMA_BUF_PLANE1_PITCH_EXT,
220  static_cast<EGLint>(pitches_[1]),
221  EGL_DMA_BUF_PLANE1_OFFSET_EXT,
222  static_cast<EGLint>(offsets_[1]),
223  EGL_DMA_BUF_PLANE1_MODIFIER_LO_EXT,
224  modifier_low,
225  EGL_DMA_BUF_PLANE1_MODIFIER_HI_EXT,
226  modifier_high,
227  EGL_NONE,
228  };
229 
230  image =
231  eglCreateImageKHR(egl_display, EGL_NO_CONTEXT, EGL_LINUX_DMA_BUF_EXT,
232  static_cast<EGLClientBuffer>(nullptr), image_attrs);
233  } else {
234  const EGLint attr_list[] = {
235  EGL_WIDTH,
236  static_cast<EGLint>(width_),
237  EGL_HEIGHT,
238  static_cast<EGLint>(height_),
239  EGL_LINUX_DRM_FOURCC_EXT,
240  static_cast<EGLint>(format_),
241  EGL_DMA_BUF_PLANE0_FD_EXT,
242  static_cast<EGLint>(image_.handle_->meta_data_.prime_fds_[0]),
243  EGL_DMA_BUF_PLANE0_PITCH_EXT,
244  static_cast<EGLint>(pitches_[0]),
245  EGL_DMA_BUF_PLANE0_OFFSET_EXT,
246  0,
247  EGL_NONE,
248  0};
249  image =
250  eglCreateImageKHR(egl_display, EGL_NO_CONTEXT, EGL_LINUX_DMA_BUF_EXT,
251  static_cast<EGLClientBuffer>(nullptr), attr_list);
252  }
253 
254  if (image == EGL_NO_IMAGE_KHR) {
255  ETRACE("eglCreateKHR failed to create image for DrmBuffer");
256  }
257  image_.image_ = image;
258  }
259 
260  GLenum target = GL_TEXTURE_EXTERNAL_OES;
261  if (!external_import) {
262  target = GL_TEXTURE_2D;
263  }
264 
265  if (!image_.texture_) {
266  GLuint texture;
267  glGenTextures(1, &texture);
268  image_.texture_ = texture;
269  }
270 
271  glBindTexture(target, image_.texture_);
272  glEGLImageTargetTexture2DOES(target, (GLeglImageOES)image_.image_);
273 
274  glBindTexture(target, 0);
275 
276  if (!external_import && image_.fb_ == 0) {
277  glGenFramebuffers(1, &image_.fb_);
278  }
279 #elif USE_VK
280  if (image_.image_ == VK_NULL_HANDLE) {
281  VkDevice dev = egl_display;
282  VkResult res;
283 
284  PFN_vkCreateDmaBufImageINTEL vkCreateDmaBufImageINTEL =
285  (PFN_vkCreateDmaBufImageINTEL)vkGetDeviceProcAddr(
286  dev, "vkCreateDmaBufImageINTEL");
287  if (vkCreateDmaBufImageINTEL == NULL) {
288  ETRACE("vkGetDeviceProcAddr(\"vkCreateDmaBufImageINTEL\") failed\n");
289  }
290 
291  VkFormat vk_format = NativeToVkFormat(format_);
292  if (vk_format == VK_FORMAT_UNDEFINED) {
293  ETRACE("Failed DRM -> Vulkan format conversion\n");
294  }
295 
296  VkExtent3D image_extent = {};
297  image_extent.width = width_;
298  image_extent.height = height_;
299  image_extent.depth = 1;
300 
301  VkDmaBufImageCreateInfo image_create = {};
302  image_create.sType =
303  (enum VkStructureType)VK_STRUCTURE_TYPE_DMA_BUF_IMAGE_CREATE_INFO_INTEL;
304  image_create.fd =
305  static_cast<int>(image_.handle_->meta_data_.prime_fds_[0]);
306  image_create.format = vk_format;
307  image_create.extent = image_extent;
308  image_create.strideInBytes = pitches_[0];
309 
310  res = vkCreateDmaBufImageINTEL(dev, &image_create, NULL, &image_.memory_,
311  &image_.image_);
312  if (res != VK_SUCCESS) {
313  ETRACE("vkCreateDmaBufImageINTEL failed\n");
314  }
315  }
316 #endif
317  return image_;
318 }

320 const MediaResourceHandle& DrmBuffer::GetMediaResource(MediaDisplay display,
321  uint32_t width,
322  uint32_t height) {
323 #ifndef DISABLE_VA
324  if (media_image_.surface_ != VA_INVALID_ID) {
325  if ((previous_width_ == width) && (previous_height_ == height)) {
326  return media_image_;
327  }
328 
329  MediaResourceHandle media_resource;
330  media_resource.surface_ = media_image_.surface_;
331  media_image_.surface_ = VA_INVALID_ID;
332  resource_manager_->MarkMediaResourceForDeletion(media_resource);
333  }
334 
335  previous_width_ = width;
336  previous_height_ = height;
337 
338  VASurfaceAttribExternalBuffers external;
339  memset(&external, 0, sizeof(external));
340  uint32_t rt_format = DrmFormatToRTFormat(format_);
341  uint32_t total_planes = image_.handle_->meta_data_.num_planes_;
342  external.pixel_format = DrmFormatToVAFormat(format_);
343  external.width = width_;
344  external.height = height_;
345  external.num_planes = total_planes;
346 #if VA_MAJOR_VERSION < 1
347  unsigned long prime_fds[total_planes];
348 #else
349  uintptr_t prime_fds[total_planes];
350 #endif
351  for (unsigned int i = 0; i < total_planes; i++) {
352  external.pitches[i] = pitches_[i];
353  external.offsets[i] = offsets_[i];
354  prime_fds[i] = image_.handle_->meta_data_.prime_fds_[i];
355  }
356 
357  external.num_buffers = total_planes;
358  external.buffers = prime_fds;
359 
360  VASurfaceAttrib attribs[2];
361  attribs[0].flags = VA_SURFACE_ATTRIB_SETTABLE;
362  attribs[0].type = VASurfaceAttribMemoryType;
363  attribs[0].value.type = VAGenericValueTypeInteger;
364  attribs[0].value.value.i = VA_SURFACE_ATTRIB_MEM_TYPE_DRM_PRIME;
365 
366  attribs[1].flags = VA_SURFACE_ATTRIB_SETTABLE;
367  attribs[1].type = VASurfaceAttribExternalBufferDescriptor;
368  attribs[1].value.type = VAGenericValueTypePointer;
369  attribs[1].value.value.p = &external;
370 
371  VAStatus ret =
372  vaCreateSurfaces(display, rt_format, external.width, external.height,
373  &media_image_.surface_, 1, attribs, 2);
374  if (ret != VA_STATUS_SUCCESS)
375  ETRACE("Failed to create VASurface from drmbuffer with ret %x", ret);
376 
377  return media_image_;
378 #else
379 
380  // FIXME: when va is disabled, this function should
381  // not been called or left to be defined if other media
382  // backend
383  ETRACE("GetMediaResource is not implemented for this Media Backend.");
384  return media_image_;
385 #endif
386 }
