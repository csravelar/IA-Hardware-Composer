220  template <typename PFN, typename T>
221  static hwc2_function_pointer_t ToHook(T function) {
222  static_assert(std::is_same<PFN, T>::value, "Incompatible fn pointer");
223  return reinterpret_cast<hwc2_function_pointer_t>(function);

226  template <typename T, typename HookType, HookType func, typename... Args>
227  static T DeviceHook(hwc2_device_t *dev, Args... args) {
228  IAHWC2 *hwc = toIAHWC2(dev);
229  return static_cast<T>(((*hwc).*func)(std::forward<Args>(args)...));
230  }
231 
232  template <typename HookType, HookType func, typename... Args>
233  static int32_t DisplayHook(hwc2_device_t *dev, hwc2_display_t display_handle,
234  Args... args) {
235  IAHWC2 *hwc = toIAHWC2(dev);
236  if (display_handle == HWC_DISPLAY_PRIMARY) {
237  HwcDisplay &display = hwc->primary_display_;
238  return static_cast<int32_t>((display.*func)(std::forward<Args>(args)...));
239  }
240 
241  if (display_handle == HWC_DISPLAY_VIRTUAL) {
242  return static_cast<int32_t>(
243  (hwc->virtual_display_.*func)(std::forward<Args>(args)...));
244  }
245 #ifdef NESTED_DISPLAY_SUPPORT
246  if (display_handle == HWC_DISPLAY_NESTED) {
247  return static_cast<int32_t>(
248  (hwc->nested_display_.*func)(std::forward<Args>(args)...));
249  }
250 #endif
251  if (display_handle == HWC_DISPLAY_EXTERNAL) {
252  HwcDisplay *display = hwc->extended_displays_.at(0).get();
253  return static_cast<int32_t>(
254  (display->*func)(std::forward<Args>(args)...));
255  }
256 
257  HwcDisplay *display = hwc->extended_displays_.at(1).get();
258  return static_cast<int32_t>((display->*func)(std::forward<Args>(args)...));
259  }
260 
261  template <typename HookType, HookType func, typename... Args>
262  static int32_t LayerHook(hwc2_device_t *dev, hwc2_display_t display_handle,
263  hwc2_layer_t layer_handle, Args... args) {
264  IAHWC2 *hwc = toIAHWC2(dev);
265  if (display_handle == HWC_DISPLAY_PRIMARY) {
266  HwcDisplay &display = hwc->primary_display_;
267  Hwc2Layer &layer = display.get_layer(layer_handle);
268  return static_cast<int32_t>((layer.*func)(std::forward<Args>(args)...));
269  }
270 
271  if (display_handle == HWC_DISPLAY_VIRTUAL) {
272  Hwc2Layer &layer = hwc->virtual_display_.get_layer(layer_handle);
273  return static_cast<int32_t>((layer.*func)(std::forward<Args>(args)...));
274  }
275 #ifdef NESTED_DISPLAY_SUPPORT
276  if (display_handle == HWC_DISPLAY_NESTED) {
277  Hwc2Layer &layer = hwc->nested_display_.get_layer(layer_handle);
278  return static_cast<int32_t>((layer.*func)(std::forward<Args>(args)...));
279  }
280 #endif
281  if (display_handle == HWC_DISPLAY_EXTERNAL) {
282  HwcDisplay *display = hwc->extended_displays_.at(0).get();
283  Hwc2Layer &layer = display->get_layer(layer_handle);
284  return static_cast<int32_t>((layer.*func)(std::forward<Args>(args)...));
285  }
286 
287  HwcDisplay *display = hwc->extended_displays_.at(1).get();
288  Hwc2Layer &layer = display->get_layer(layer_handle);
289  return static_cast<int32_t>((layer.*func)(std::forward<Args>(args)...));
290  }
