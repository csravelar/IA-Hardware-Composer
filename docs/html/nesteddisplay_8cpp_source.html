191 bool NestedDisplay::Present(std::vector<HwcLayer *> &source_layers,
192  int32_t * /*retire_fence*/,
193  PixelUploaderCallback * /*call_back*/,
194  bool /*handle_constraints*/) {
195 #ifndef NESTED_DISPLAY_SUPPORT
196  HWC_UNUSED(source_layers);
197  return true;
198 #else
199  if (!mconnected)
200  return true;
201 
202  int ret = 0;
203  size_t size = source_layers.size();
204  const uint32_t *pitches;
205  const uint32_t *offsets;
206  HWCNativeHandle sf_handle;
207  size_t buffer_number = 0;
208  size_t info_size = sizeof(vm_buffer_info);
209  size_t header_size = sizeof(vm_header);
210 
211  for (size_t layer_index = 0; layer_index < size; layer_index++) {
212  uint32_t surf_index = 0;
213  HwcLayer *layer = source_layers.at(layer_index);
214  if (!layer->IsVisible())
215  continue;
216 
217  const HwcRect<int> &display_frame = layer->GetDisplayFrame();
218  sf_handle = layer->GetNativeHandle();
219  auto search = mHyperDmaExportedBuffers.find(sf_handle);
220  if (search == mHyperDmaExportedBuffers.end()) {
221  std::shared_ptr<OverlayBuffer> buffer(NULL);
222  buffer = OverlayBuffer::CreateOverlayBuffer();
223  buffer->InitializeFromNativeHandle(sf_handle, resource_manager_.get(),
224  fb_manager_);
225 
226  if (mHyperDmaBuf_Fd > 0 && buffer->GetPrimeFD() > 0) {
227  struct ioctl_hyper_dmabuf_export_remote msg;
228  memset(&msg, 0, sizeof(ioctl_hyper_dmabuf_export_remote));
229 
230  /* TODO: add more flexibility here, instead of hardcoded domain 0*/
231  msg.remote_domain = 0;
232  msg.dmabuf_fd = buffer->GetPrimeFD();
233 
234  ret = ioctl(mHyperDmaBuf_Fd, IOCTL_HYPER_DMABUF_EXPORT_REMOTE, &msg);
235  if (ret) {
236  ETRACE("Hyper DmaBuf: Exporting hyper_dmabuf failed with error %d\n",
237  ret);
238  return false;
239  } else {
240  ITRACE("Hyper DmaBuf: Exporting hyper_dmabuf Done! 0x%x\n",
241  msg.hid.id);
242  mHyperDmaExportedBuffers[sf_handle].surf_index = surf_index++;
243  mHyperDmaExportedBuffers[sf_handle].width = buffer->GetWidth();
244  mHyperDmaExportedBuffers[sf_handle].height = buffer->GetHeight();
245  mHyperDmaExportedBuffers[sf_handle].format = buffer->GetFormat();
246  pitches = buffer->GetPitches();
247  offsets = buffer->GetOffsets();
248  mHyperDmaExportedBuffers[sf_handle].pitch[0] = pitches[0];
249  mHyperDmaExportedBuffers[sf_handle].pitch[1] = pitches[1];
250  mHyperDmaExportedBuffers[sf_handle].pitch[2] = pitches[2];
251  mHyperDmaExportedBuffers[sf_handle].offset[0] = offsets[0];
252  mHyperDmaExportedBuffers[sf_handle].offset[1] = offsets[1];
253  mHyperDmaExportedBuffers[sf_handle].offset[2] = offsets[2];
254  mHyperDmaExportedBuffers[sf_handle].tile_format =
255  buffer->GetTilingMode();
256  mHyperDmaExportedBuffers[sf_handle].rotation = 0;
257  mHyperDmaExportedBuffers[sf_handle].status = 0;
258  mHyperDmaExportedBuffers[sf_handle].counter = 0;
259  mHyperDmaExportedBuffers[sf_handle].hyper_dmabuf_id = msg.hid;
260  mHyperDmaExportedBuffers[sf_handle].surface_id = (uint64_t)sf_handle;
261  strncpy(mHyperDmaExportedBuffers[sf_handle].surface_name, "Cluster",
262  SURFACE_NAME_LENGTH);
263  mHyperDmaExportedBuffers[sf_handle].bbox[0] = display_frame.left;
264  mHyperDmaExportedBuffers[sf_handle].bbox[1] = display_frame.top;
265  mHyperDmaExportedBuffers[sf_handle].bbox[2] = buffer->GetWidth();
266  mHyperDmaExportedBuffers[sf_handle].bbox[3] = buffer->GetHeight();
267  }
268  }
269  }
270  memcpy(buf + sizeof(int) + header_size + info_size * buffer_number,
271  &mHyperDmaExportedBuffers[sf_handle], info_size);
272  buffer_number++;
273  }
274  int *stream_start = reinterpret_cast<int *>(buf);
275  *stream_start = METADATA_STREAM_START;
276  vm_header *header = reinterpret_cast<vm_header *>(buf + sizeof(int));
277  header->n_buffers = buffer_number;
278  header->version = 3;
279  header->output = 0; // need clarify meaning
280  header->counter = 0; // add later
281  header->disp_w = width_;
282  header->disp_h = height_;
283  int *stream_end = reinterpret_cast<int *>(buf + sizeof(int) + header_size +
284  info_size * buffer_number);
285  *stream_end = METADATA_STREAM_END;
286  int msg_size = header_size + info_size * buffer_number + sizeof(int) * 2;
287  int rc;
288  do {
289  rc = HyperCommunicationNetworkSendData(buf, msg_size);
290  } while (rc != msg_size && rc >= 0);
291  memset(buf, 0, METADATA_BUFFER_SIZE);
292 
293  return true;
294 #endif
295 }

428 #ifdef NESTED_DISPLAY_SUPPORT
429 int NestedDisplay::StartSockService() {
430  struct sockaddr_in server_addr;
431  msock_fd = socket(AF_INET, SOCK_STREAM, 0);
432  if (msock_fd < 0) {
433  ETRACE("Cannot create socket fd:%d", msock_fd);
434  return -1;
435  }
436 
437  memset(&server_addr, 0, sizeof(server_addr));
438  int portno = port_;
439  server_addr.sin_family = AF_INET;
440  server_addr.sin_addr.s_addr = INADDR_ANY;
441  server_addr.sin_port = htons(portno);
442 
443  if (bind(msock_fd, (struct sockaddr *)&server_addr, sizeof(server_addr)) <
444  0) {
445  close(msock_fd);
446  msock_fd = -1;
447  return -1;
448  }
449 
450  listen(msock_fd, 1);
451  signal(SIGPIPE, SignalCallbackHandler);
452  return 0;
453 }
