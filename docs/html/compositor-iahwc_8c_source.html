285  pixman_region32_t *damage, void *repaint_data) {
286  struct iahwc_output *output = to_iahwc_output(output_base);
287  struct iahwc_backend *backend = to_iahwc_backend(output->base.compositor);
288  struct wl_event_loop *loop;
289 
290  weston_log("release fence is %d\n", output->release_fence);
291  if (output->release_fence > 0) {
292  wl_event_source_remove(output->release_fence_source);
293  close(output->release_fence);
294  output->release_fence = -1;
295  output->release_fence_source = NULL;
296  }
297 
298  backend->iahwc_present_display(backend->iahwc_device, 0,
299  &output->release_fence);
300 
301  loop = wl_display_get_event_loop(output->base.compositor->wl_display);
302 
303  if (output->release_fence > 0) {
304  output->release_fence_source = wl_event_loop_add_fd(
305  loop, output->release_fence, WL_EVENT_READABLE, frame_done, output);
306  } else {
307  // when release fence is -1, immediately call frame_done
308  wl_event_loop_add_idle(loop, frame_done, output);
309  }
310 
311  lock(&output->spin_lock);
312  output->state_invalid = false;
313  unlock(&output->spin_lock);
314  return 0;
315 }

386  struct weston_mode *target_mode) {
387  struct iahwc_mode *tmp_mode = NULL, *mode;
388 
389  if (output->base.current_mode->width == target_mode->width &&
390  output->base.current_mode->height == target_mode->height &&
391  (output->base.current_mode->refresh == target_mode->refresh ||
392  target_mode->refresh == 0))
393  return (struct iahwc_mode *)output->base.current_mode;
394 
395  wl_list_for_each(mode, &output->base.mode_list, base.link) {
396  if (mode->base.width == target_mode->width &&
397  mode->base.height == target_mode->height) {
398  if (mode->base.refresh == target_mode->refresh ||
399  target_mode->refresh == 0) {
400  return mode;
401  } else if (!tmp_mode)
402  tmp_mode = mode;
403  }
404  }
405 
406  return tmp_mode;

451 static struct gbm_device *create_gbm_device(int fd) {
452  struct gbm_device *gbm;
453 
454  gl_renderer = weston_load_module("gl-renderer.so", "gl_renderer_interface");
455  if (!gl_renderer)
456  return NULL;
457 
458  /* GBM will load a dri driver, but even though they need symbols from
459  * libglapi, in some version of Mesa they are not linked to it. Since
460  * only the gl-renderer module links to it, the call above won't make
461  * these symbols globally available, and loading the DRI driver fails.
462  * Workaround this by dlopen()'ing libglapi with RTLD_GLOBAL. */
463  dlopen("libglapi.so.0", RTLD_LAZY | RTLD_GLOBAL);
464 
465  gbm = gbm_create_device(fd);
466 
467  return gbm;
468 }

470 /* When initializing EGL, if the preferred buffer format isn't available
471  * we may be able to substitute an ARGB format for an XRGB one.
472  *
473  * This returns 0 if substitution isn't possible, but 0 might be a
474  * legitimate format for other EGL platforms, so the caller is
475  * responsible for checking for 0 before calling gl_renderer->create().
476  *
477  * This works around https://bugs.freedesktop.org/show_bug.cgi?id=89689
478  * but it's entirely possible we'll see this again on other implementations.
479  */
480 static int fallback_format_for(uint32_t format) {
481  switch (format) {
482  case GBM_FORMAT_XRGB8888:
483  return GBM_FORMAT_ARGB8888;
484  case GBM_FORMAT_XRGB2101010:
485  return GBM_FORMAT_ARGB2101010;
486  default:
487  return 0;
488  }
489 }

491 static int iahwc_backend_create_gl_renderer(struct iahwc_backend *b) {
492  EGLint format[3] = {
493  b->gbm_format, fallback_format_for(b->gbm_format), 0,
494  };
495  int n_formats = 2;
496 
497  if (format[1])
498  n_formats = 3;
499  if (gl_renderer->display_create(
500  b->compositor, EGL_PLATFORM_GBM_KHR, (void *)b->gbm, NULL,
501  gl_renderer->opaque_attribs, format, n_formats) < 0) {
502  return -1;
503  }
504 
505  return 0;
506 }

593  struct iahwc_backend *b = to_iahwc_backend(ec);

595  struct wl_resource *buffer_resource;
596  struct weston_plane *p = &output->overlay_plane;
597  struct linux_dmabuf_buffer *dmabuf;
598  struct gbm_bo *bo;
599  struct wl_shm_buffer *shmbuf;
600  pixman_region32_t dest_rect, src_rect;
601  pixman_box32_t *box, tbox;
602  wl_fixed_t sx1, sy1, sx2, sy2;
603  int32_t src_x, src_y;
604  uint32_t src_w, src_h;
605  uint32_t dest_x, dest_y;
606  uint32_t dest_w, dest_h;
607  int is_cusor_layer = 0;
608  float x, y;
609 
610  uint32_t overlay_layer_id;
611  if (ev->surface->buffer_ref.buffer == NULL) {
612  return NULL;
613  }
614  buffer_resource = ev->surface->buffer_ref.buffer->resource;
615  shmbuf = wl_shm_buffer_get(buffer_resource);
616 
617  struct iahwc_overlay *plane = 0;
618  plane = iahwc_get_existing_plane(output, layer_index);
619  // Update Damage.
620  bool layer_damaged = true;
621  bool full_damage = false;
622  iahwc_region_t damage_region;
623  damage_region.numRects = 1;
624  struct weston_surface *es = ev->surface;
625  if (!plane) {
626  b->iahwc_create_layer(b->iahwc_device, 0, &overlay_layer_id);
627  full_damage = true;
628  } else if (plane->es == es) {
629  overlay_layer_id = plane->overlay_layer_id;
630  if (!pixman_region32_not_empty(&es->pending.damage_buffer) &&
631  !pixman_region32_not_empty(&es->pending.damage_surface) &&
632  !pixman_region32_not_empty(&es->damage)) {
633  iahwc_rect_t damage_rect = {0, 0, 0, 0};
634  damage_region.rects = &damage_rect;
635  b->iahwc_layer_set_surface_damage(b->iahwc_device, 0, overlay_layer_id,
636  damage_region);
637  layer_damaged = false;
638  } else {
639  pixman_region32_t damage;
640  pixman_region32_init(&damage);
641  pixman_region32_union(&damage, &es->pending.damage_surface, &es->damage);
642  pixman_region32_union(&damage, &es->pending.damage_buffer, &damage);
643  pixman_region32_fini(&damage);
644  pixman_box32_t *damage_extents = pixman_region32_extents(&damage);
645  iahwc_rect_t damage_rect = {damage_extents->x1, damage_extents->y1,
646  damage_extents->x2, damage_extents->y2};
647  damage_region.rects = &damage_rect;
648  b->iahwc_layer_set_surface_damage(b->iahwc_device, 0, overlay_layer_id,
649  damage_region);
650  }
651  } else {
652  overlay_layer_id = plane->overlay_layer_id;
653  // Layer might have changed z-order as surface has changed.
654  // Mark full surface as damaged.
655  full_damage = true;
656  }
657 
658  pixman_region32_clear(&es->pending.damage_buffer);
659  pixman_region32_clear(&es->pending.damage_surface);
660  pixman_region32_clear(&es->damage);
661 
662  if (shmbuf) {
663  if (ev->surface->width <= b->cursor_width &&
664  ev->surface->height <= b->cursor_height) {
665  is_cusor_layer = 1;
666  b->iahwc_layer_set_usage(b->iahwc_device, 0, overlay_layer_id,
667  IAHWC_LAYER_USAGE_CURSOR);
668  } else {
669  b->iahwc_layer_set_usage(b->iahwc_device, 0, overlay_layer_id,
670  IAHWC_LAYER_USAGE_OVERLAY);
671  }
672  }
673 
674  if (is_cusor_layer) {
675  weston_view_to_global_float(ev, 0, 0, &x, &y);

723  dest_y = tbox.y1;
724  dest_w = tbox.x2 - tbox.x1;
725  dest_h = tbox.y2 - tbox.y1;
