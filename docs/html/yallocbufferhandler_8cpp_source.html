74 bool YallocBufferHandler::CreateBuffer(uint32_t w, uint32_t h, int format,
75  HWCNativeHandle *handle,
76  uint32_t layer_type, bool *modifier_used,
77  int64_t /*preferred_modifier*/,
78  bool /*raw_pixel_buffer*/) const {
79  // LOG_I("YallocBufferHandler::CreateBuffer --> enter.\n");
80  // LOG_I("\tw: %d, h: %d, format: %x.\n", w, h, format);
81  // FIXME:: Take Modifiers into use.
82  if (modifier_used)
83  *modifier_used = false;
84 
85  struct yalloc_handle *temp = new struct yalloc_handle();
86 
87  uint32_t pixel_format = 0;
88  if (format != 0) {
89  pixel_format = DrmFormatToHALFormat(format);
90  }
91 
92  if (pixel_format == 0) {
93  pixel_format = YUN_HAL_FORMAT_RGBA_8888;
94  }
95 
96  bool force_normal_usage = false;
97  if ((layer_type == hwcomposer::kLayerVideo) &&
98  !IsSupportedMediaFormat(format)) {
99  ETRACE("Forcing normal usage for Video Layer. \n");
100  force_normal_usage = true;
101  }
102 
103  uint32_t usage = 0;
104  if ((layer_type == hwcomposer::kLayerNormal) || force_normal_usage) {
105  usage |= YALLOC_FLAG_HW_COMPOSER | YALLOC_FLAG_HW_RENDER |
106  YALLOC_FLAG_HW_TEXTURE;
107  } else if (layer_type == hwcomposer::kLayerVideo) {
108  switch (pixel_format) {
109  case YUN_HAL_FORMAT_YCbCr_422_I:
110  case YUN_HAL_FORMAT_Y8:
111  usage |= YALLOC_FLAG_HW_TEXTURE | YALLOC_FLAG_HW_VIDEO_DECODER;
112  break;
113  default:
114  usage |= YALLOC_FLAG_HW_CAMERA_WRITE | YALLOC_FLAG_HW_CAMERA_READ |
115  YALLOC_FLAG_HW_VIDEO_ENCODER | YALLOC_FLAG_HW_TEXTURE;
116  }
117  } else if (layer_type == hwcomposer::kLayerCursor) {
118  usage |= YALLOC_FLAG_CURSOR;
119  }
120 
121  int stride;
122  device_->alloc(device_, w, h, pixel_format, usage, &(temp->target_), &stride);
123 
124  temp->hwc_buffer_ = true;
125 
126  *handle = temp;
127 
128  return true;
129 }
