33 bool GLSurface::InitializeGPUResources() {
34  EGLDisplay egl_display = eglGetCurrentDisplay();
35  // Create EGLImage.
36  const ResourceHandle& import =
37  layer_.GetBuffer()->GetGpuResource(egl_display, false);
38 
39  if (import.image_ == EGL_NO_IMAGE_KHR) {
40  ETRACE("Failed to make EGL image.");
41  return false;
42  }
43 
44  // Bind Fb.
45  fb_ = import.fb_;
46  glBindFramebuffer(GL_FRAMEBUFFER, fb_);
47  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D,
48  import.texture_, 0);
49 
50  fb_ = import.fb_;
51  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
52  if (status != GL_FRAMEBUFFER_COMPLETE) {
53  switch (status) {
54  case (GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT):
55  ETRACE("GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT.");
56  break;
57  case (GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT):
58  ETRACE("GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT.");
59  break;
60  case (GL_FRAMEBUFFER_UNSUPPORTED):
61  ETRACE("GL_FRAMEBUFFER_UNSUPPORTED.");
62  break;
63  default:
64  break;
65  }
66 
67  ETRACE("GL Framebuffer is not complete %d.", import.texture_);
68  return false;
69  }
70 
71  return true;
72 }
